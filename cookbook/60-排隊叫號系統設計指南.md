## 1. 系統架構總覽

### 1.1 四端架構

```

|   取號機 (Kiosk)  | --> |   後端 Server    | --> |   顯示螢幕       |

```

## 2. 取號機介面

### 2.1 設計原則

取號機通常是觸控螢幕（平板或 Kiosk），UI 必須極度簡潔：

| 原則 | 說明 |
|------|------|
| **大按鈕** | 每個服務類型一個大色塊，覆蓋整個螢幕區域 |
| **無需登入** | 點一下就取號，零學習成本 |
| **即時回饋** | 取號後顯示號碼 + 前面等待人數 + 預估時間 |
| **字體超大** | 號碼至少 72px，老人家也能看清楚 |
| **無障礙** | 高對比配色，避免純色盲不友善的紅綠配 |

```javascript
const counterId = 'counter-1'; // 目前窗口 ID

async function callNext() {
  const res = await fetch('/api/queue/call-next', {

  }));

async function recallCurrent() {
  const number = document.getElementById('serving-number').textContent;
  if (number === '—') return;

  ws.send(JSON.stringify({
    type: 'recall',
    number,
    counter: '1 號窗口'
  }));
}

async function skipCurrent() {
  const number = document.getElementById('serving-number').textContent;
  if (number === '—') return;

  await fetch('/api/queue/skip', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ counterId, action: 'skip' })
  });

  document.getElementById('serving-number').textContent = '—';
}

async function completeCurrent() {
  await fetch('/api/queue/complete', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ counterId })
  });
```

## 5. 語音播報

### 5.1 Web Speech API

```javascript
function announceNumber(number, counter) {
  if (!('speechSynthesis' in window)) return;

  // 取消正在播放的語音
  speechSynthesis.cancel();

  // 將號碼拆成可讀格式：A023 → A 零二三
  const readable = formatNumberForSpeech(number);

  const utterance = new SpeechSynthesisUtterance(
    `請 ${readable} 號，到 ${counter} 辦理`
  );

  utterance.lang = 'zh-TW';
  utterance.rate = 0.85;   // 語速稍慢，聽得清楚
  utterance.pitch = 1;
  utterance.volume = 1;

  // 選擇中文語音
  const voices = speechSynthesis.getVoices();
  const zhVoice = voices.find(v => v.lang.startsWith('zh'));
  if (zhVoice) utterance.voice = zhVoice;

  // 播報兩次（重要場所慣例）
  speechSynthesis.speak(utterance);

  setTimeout(() => {
    const repeat = new SpeechSynthesisUtterance(
      `再次提醒，${readable} 號，請到 ${counter}`
    );
    repeat.lang = 'zh-TW';
    repeat.rate = 0.85;
    if (zhVoice) repeat.voice = zhVoice;
    speechSynthesis.speak(repeat);
  }, 3000);
}

function formatNumberForSpeech(number) {
  // 'A023' → 'A 零二三'
  const digits = { '0': '零', '1': '一', '2': '二', '3': '三', '4': '四',
                   '5': '五', '6': '六', '7': '七', '8': '八', '9': '九' };
  return number.split('').map(c => digits[c] || c).join(' ');
}
```

### 6.3 LINE 綁定號碼（Webhook）

```javascript
// LINE Webhook 接收訊息
app.post('/api/line/webhook', (req, res) => {
  const events = req.body.events || [];

  events.forEach(event => {
    if (event.type === 'message' && event.message.type === 'text') {
      const text = event.message.text.trim().toUpperCase();
      const userId = event.source.userId;

      // 客人傳送號碼來綁定，如 "A023"
      if (/^[A-Z]\d{3,4}$/.test(text)) {
        bindTicketToLine(text, userId);
        replyMessage(event.replyToken,
          `已綁定號碼 ${text}，快到號時會通知您！`
        );
      }
    }
  });

  res.sendStatus(200);
});

async function bindTicketToLine(number, lineUserId) {
  // 更新票券的 lineUserId
  await db.query(
    'UPDATE tickets SET line_user_id = $1 WHERE number = $2 AND status = $3',
    [lineUserId, number, 'waiting']
  );
}
```

## 7. 多窗口與多服務台管理

### 7.1 智慧分配策略

```javascript
class QueueManager {
  constructor() {
    this.queues = {};  // serviceType -> Ticket[]
    this.counters = []; // Counter[]
  }

  // 智慧取下一位：考慮服務類型匹配 + VIP 優先
  getNextTicket(counterId) {
    const counter = this.counters.find(c => c.id === counterId);
    if (!counter) return null;

    // 依窗口可服務的類型，找等待最久的票
    let bestTicket = null;
    let longestWait = 0;

    for (const typeId of counter.serviceTypes) {
      const queue = this.queues[typeId] || [];
      const waiting = queue.filter(t => t.status === 'waiting');

      if (waiting.length > 0) {
        const first = waiting[0];
        const waitTime = Date.now() - new Date(first.createdAt).getTime();

        // VIP 類型權重 x2
        const weight = typeId === 'vip' ? 2 : 1;

        if (waitTime * weight > longestWait) {
          longestWait = waitTime * weight;
          bestTicket = first;
        }
      }
    }

    return bestTicket;
  }

  // 預估等待時間
  estimateWait(serviceTypeId) {
    const queue = this.queues[serviceTypeId] || [];
    const waiting = queue.filter(t => t.status === 'waiting').length;
    const serviceType = this.getServiceType(serviceTypeId);
    const activeCounters = this.counters.filter(
      c => c.serviceTypes.includes(serviceTypeId) && c.status !== 'paused'
    ).length;

    if (activeCounters === 0) return 999;

    return Math.ceil((waiting * serviceType.avgServiceTime) / activeCounters);
  }

  // 每日統計
  getDailyStats() {
    const allTickets = Object.values(this.queues).flat();
    const completed = allTickets.filter(t => t.status === 'completed');

    return {
      date: new Date().toISOString().slice(0, 10),
      totalTickets: allTickets.length,
      servedCount: completed.length,
      avgWaitTime: this.calcAvgWait(completed),
      avgServiceTime: this.calcAvgService(completed),
      peakHour: this.findPeakHour(allTickets)
    };
  }

  calcAvgWait(tickets) {
    if (tickets.length === 0) return 0;
    const total = tickets.reduce((sum, t) => {
      return sum + (new Date(t.calledAt) - new Date(t.createdAt)) / 60000;
    }, 0);
    return Math.round(total / tickets.length);
  }

  findPeakHour(tickets) {
    const hours = {};
    tickets.forEach(t => {
      const h = new Date(t.createdAt).getHours();
      hours[h] = (hours[h] || 0) + 1;
    });
    return Object.entries(hours).sort((a, b) => b[1] - a[1])[0]?.[0] + ':00' || 'N/A';
  }
}
```

### 8.1 後端 API（Express）

```javascript
const express = require('express');
const router = express.Router();

// 今日號碼計數器（記憶體內，每日重設）
const numberCounters = {}; // { 'A': 0, 'B': 0 }

// 取號
router.post('/queue/take', (req, res) => {
  const { serviceType } = req.body;
  const type = serviceTypes.find(t => t.id === serviceType);
  if (!type) return res.status(400).json({ error: '無效的服務類型' });

  // 產生號碼
  const prefix = type.prefix;
  numberCounters[prefix] = (numberCounters[prefix] || 0) + 1;
  const number = `${prefix}${String(numberCounters[prefix]).padStart(3, '0')}`;

  const ticket = {
    id: `T-${Date.now()}`,
    number,
    serviceType: type.id,
    status: 'waiting',
    counter: null,
    createdAt: new Date().toISOString(),
    calledAt: null,
    completedAt: null,
    lineUserId: null,
    estimatedWait: queueManager.estimateWait(type.id)
  };

  queueManager.addTicket(ticket);

  // 回傳取號結果
  const waitingAhead = queueManager.getWaitingCount(type.id);
  res.json({ ...ticket, waitingAhead });

  // 廣播佇列更新
  broadcastQueueUpdate();
});

// 叫號
router.post('/queue/call-next', (req, res) => {
  const { counterId } = req.body;
  const ticket = queueManager.getNextTicket(counterId);

  if (!ticket) {
    return res.json({ message: '沒有等待中的號碼' });
  }

  ticket.status = 'calling';
  ticket.calledAt = new Date().toISOString();
  ticket.counter = counterId;

  // 更新窗口狀態
  const counter = queueManager.counters.find(c => c.id === counterId);
  counter.currentTicket = ticket.number;
  counter.status = 'serving';

  res.json(ticket);

  // 廣播叫號到顯示螢幕
  broadcastCall(ticket.number, counter.name);

  // LINE 通知
  notifyCalling(ticket, counter.name);

  // 通知前 3 位
  notifyApproachingTickets(ticket.serviceType);
});

// 完成服務
router.post('/queue/complete', (req, res) => {
  const { counterId } = req.body;
  const counter = queueManager.counters.find(c => c.id === counterId);

  if (counter.currentTicket) {
    const ticket = queueManager.findTicket(counter.currentTicket);
    if (ticket) {
      ticket.status = 'completed';
      ticket.completedAt = new Date().toISOString();
    }
    counter.currentTicket = null;
    counter.status = 'idle';
  }

  res.json({ success: true });
  broadcastQueueUpdate();
});

module.exports = router;
```

```javascript
const cron = require('node-cron');

// 每天凌晨 5:00 重設號碼（早於營業時間）
cron.schedule('0 5 * * *', () => {
  // 儲存昨日統計
  const stats = queueManager.getDailyStats();
  saveDailyStats(stats);

  // 重設
  Object.keys(numberCounters).forEach(k => numberCounters[k] = 0);
  queueManager.reset();

});
```
