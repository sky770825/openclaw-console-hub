## 1. 圖片格式選擇指南

### 格式比較總覽

| 格式 | 壓縮方式 | 透明度 | 動畫 | 瀏覽器支援 | 典型用途 |
|------|----------|--------|------|------------|----------|
| **JPEG** | 有損 | 不支援 | 不支援 | 100% | 照片、大型背景圖 |
| **PNG** | 無損 | 支援 | 不支援 | 100% | 需透明背景的圖、截圖 |
| **WebP** | 有損/無損 | 支援 | 支援 | 97%+ | **通用首選**，照片+圖示皆可 |
| **AVIF** | 有損/無損 | 支援 | 支援 | 92%+ | 極致壓縮、高品質攝影 |
| **SVG** | 向量（XML） | 支援 | 支援 | 100% | 圖標、Logo、插圖 |
| **GIF** | 無損（256色） | 支援 | 支援 | 100% | **已過時**，改用 WebP 或 `<video>` |

### 場景速查表

```
照片 / 攝影作品     → WebP（首選）或 AVIF（追求極致）
需要透明背景        → WebP（首選）或 PNG（相容性需求）
圖標 / Logo        → SVG（向量縮放不失真）
簡單插圖 / 圖表     → SVG
截圖 / 含文字圖片   → PNG（無損不糊字）
動態圖             → <video> 取代 GIF（省 90% 容量）
電商商品圖          → WebP + AVIF fallback
英雄橫幅（Hero）    → WebP，搭配 srcset 響應式
favicon            → SVG（現代瀏覽器）+ PNG 32x32（fallback）
Open Graph 社群圖   → JPEG 1200x630（社群平台不支援 WebP）
```

## 2. 圖片壓縮實戰

### 品質設定建議

| 格式 | 建議品質 | 說明 |
|------|----------|------|
| **WebP** | 75-80 | 80 是甜蜜點，肉眼幾乎無差異 |
| **JPEG** | 80-85 | 低於 75 開始出現明顯色塊 |
| **AVIF** | 45-55 | AVIF 壓縮效率高，50 就很好 |
| **PNG** | 無損 | 用工具移除多餘 metadata 即可 |

> **經驗法則**：同樣視覺品質下，WebP 比 JPEG 小 25-35%，AVIF 比 JPEG 小 40-50%。

### 批次壓縮腳本

#### Sharp 批次壓縮腳本（推薦，Node.js 專案直接用）

```javascript
// scripts/optimize-images.mjs
import sharp from 'sharp';
import { readdir, stat, mkdir } from 'fs/promises';
import path from 'path';

const INPUT_DIR = './src/assets/images/raw';
const OUTPUT_DIR = './src/assets/images/optimized';
const WEBP_QUALITY = 80;
const MAX_WIDTH = 1920; // 最大寬度，超過就縮

async function optimizeAll() {
  await mkdir(OUTPUT_DIR, { recursive: true });
  const files = await readdir(INPUT_DIR);
  const imageFiles = files.filter(f =>
    /\.(jpe?g|png|gif|tiff|bmp)$/i.test(f)
  );

  console.log(`找到 ${imageFiles.length} 張圖片，開始壓縮...`);

  let totalSaved = 0;

  for (const file of imageFiles) {
    const inputPath = path.join(INPUT_DIR, file);
    const baseName = path.parse(file).name;
    const inputStat = await stat(inputPath);

    // 產生 WebP
    const webpPath = path.join(OUTPUT_DIR, `${baseName}.webp`);
    await sharp(inputPath)
      .resize({ width: MAX_WIDTH, withoutEnlargement: true })
      .webp({ quality: WEBP_QUALITY })
      .toFile(webpPath);

    // 產生 AVIF（給支援的瀏覽器）
    const avifPath = path.join(OUTPUT_DIR, `${baseName}.avif`);
    await sharp(inputPath)
      .resize({ width: MAX_WIDTH, withoutEnlargement: true })
      .avif({ quality: 50 })
      .toFile(avifPath);

    // 產生最佳化 JPEG fallback
    const jpgPath = path.join(OUTPUT_DIR, `${baseName}.jpg`);
    await sharp(inputPath)
      .resize({ width: MAX_WIDTH, withoutEnlargement: true })
      .jpeg({ quality: 85, mozjpeg: true })
      .toFile(jpgPath);

    const webpStat = await stat(webpPath);
    const saved = ((1 - webpStat.size / inputStat.size) * 100).toFixed(1);
    totalSaved += inputStat.size - webpStat.size;

    console.log(`  ${file} → WebP 省 ${saved}%`);
  }

  console.log(`\n完成！總共省下 ${(totalSaved / 1024 / 1024).toFixed(2)} MB`);
}

optimizeAll();
```

```bash
# 執行
node scripts/optimize-images.mjs
```

#### ImageMagick 批次壓縮腳本（不需 Node.js）

```bash
#!/bin/bash
# scripts/batch-compress.sh
# 用法：bash batch-compress.sh ./images/raw ./images/optimized

INPUT_DIR="${1:-.}"
OUTPUT_DIR="${2:-./optimized}"
WEBP_QUALITY=80
MAX_WIDTH=1920

mkdir -p "$OUTPUT_DIR"

count=0
saved=0

for img in "$INPUT_DIR"/*.{jpg,jpeg,png,JPG,JPEG,PNG}; do
  [ -f "$img" ] || continue

  filename=$(basename "$img")
  name="${filename%.*}"
  original_size=$(stat -f%z "$img" 2>/dev/null || stat --format=%s "$img" 2>/dev/null)

  # WebP
  magick "$img" -resize "${MAX_WIDTH}x>" -quality $WEBP_QUALITY "$OUTPUT_DIR/${name}.webp"

  # 最佳化 JPEG fallback
  magick "$img" -resize "${MAX_WIDTH}x>" -quality 85 -strip "$OUTPUT_DIR/${name}.jpg"

  webp_size=$(stat -f%z "$OUTPUT_DIR/${name}.webp" 2>/dev/null || stat --format=%s "$OUTPUT_DIR/${name}.webp" 2>/dev/null)
  percent=$(echo "scale=1; (1 - $webp_size / $original_size) * 100" | bc)

  echo "$filename → WebP 省 ${percent}%"
  count=$((count + 1))
  saved=$((saved + original_size - webp_size))
done

echo ""
echo "處理 $count 張圖片，總共省下 $(echo "scale=2; $saved / 1024 / 1024" | bc) MB"
```

### 響應式圖片自動產生腳本

```javascript
// scripts/generate-responsive.mjs
import sharp from 'sharp';
import path from 'path';

const SIZES = [400, 800, 1200, 1600, 2000];
const FORMATS = ['webp', 'avif', 'jpeg'];

async function generateResponsive(inputPath, outputDir) {
  const baseName = path.parse(inputPath).name;

  for (const width of SIZES) {
    for (const format of FORMATS) {
      const ext = format === 'jpeg' ? 'jpg' : format;
      const outputPath = path.join(outputDir, `${baseName}-${width}.${ext}`);

      let pipeline = sharp(inputPath)
        .resize({ width, withoutEnlargement: true });

      switch (format) {
        case 'webp':
          pipeline = pipeline.webp({ quality: 80 });
          break;
        case 'avif':
          pipeline = pipeline.avif({ quality: 50 });
          break;
        case 'jpeg':
          pipeline = pipeline.jpeg({ quality: 85, mozjpeg: true });
          break;
      }

      await pipeline.toFile(outputPath);
      console.log(`  生成 ${outputPath}`);
    }
  }
}

// 用法
generateResponsive('./raw/hero.jpg', './public/images/');
```

## 4. Lazy Loading

### 原生 `loading="lazy"`（最簡單，推薦）

```html
<!-- 首屏圖片：不要加 loading="lazy"，用 fetchpriority="high" 加速 -->
<img
  src="hero.webp"
  alt="首屏主視覺"
  width="1920"
  height="800"
  fetchpriority="high"
  decoding="async"
/>

<!-- 非首屏圖片：加 loading="lazy" -->
<img
  src="product-1.webp"
  alt="商品圖片"
  width="400"
  height="300"
  loading="lazy"
  decoding="async"
/>
```

> **重要規則**：
> - 首屏可見的圖片（Hero、Logo、導航欄圖片）**不要** lazy load
> - `<iframe>` 也支援 `loading="lazy"`
> - 一定要寫 `width` 和 `height`，避免版面跳動（CLS）

### Intersection Observer 自定義（更多控制）

```javascript
// lazy-load.js — 自定義 lazy loading，支援動畫淡入
document.addEventListener('DOMContentLoaded', () => {
  const lazyImages = document.querySelectorAll('img[data-src]');

  if ('IntersectionObserver' in window) {
    const observer = new IntersectionObserver((entries, obs) => {
      entries.forEach(entry => {
        if (entry.isIntersecting) {
          const img = entry.target;

          // 載入圖片
          img.src = img.dataset.src;
          if (img.dataset.srcset) {
            img.srcset = img.dataset.srcset;
          }

          // 淡入動畫
          img.classList.add('loaded');

          // 載入後停止觀察
          obs.unobserve(img);
        }
      });
    }, {
      // 提前 200px 開始載入（使用者滾到之前就開始）
      rootMargin: '200px 0px',
      threshold: 0.01
    });

    lazyImages.forEach(img => observer.observe(img));
  } else {
    // fallback：不支援 IntersectionObserver 的瀏覽器，直接載入
    lazyImages.forEach(img => {
      img.src = img.dataset.src;
      if (img.dataset.srcset) img.srcset = img.dataset.srcset;
    });
  }
});
```

### LQIP（Low Quality Image Placeholder）模糊佔位圖

```javascript
// 用 Sharp 產生模糊佔位圖（10px 寬，base64）
import sharp from 'sharp';

async function generateLQIP(inputPath) {
  const buffer = await sharp(inputPath)
    .resize(10) // 極小尺寸
    .blur()     // 模糊
    .toBuffer();

  const base64 = buffer.toString('base64');
  const mimeType = 'image/jpeg';
  return `data:${mimeType};base64,${base64}`;
}

// 用法
const placeholder = await generateLQIP('product.jpg');
// 產出的 base64 字串只有 ~200-400 bytes
```

## 5. CDN 設定

### Cloudflare 免費方案設定

#### 基本設定（適用所有方案）

1. **DNS 設定**：將網站 DNS 指向 Cloudflare（橘雲開啟代理）
2. **快取規則**：靜態資源自動快取（圖片、CSS、JS）

```
# Cloudflare 預設快取的圖片副檔名
.jpg .jpeg .png .gif .webp .avif .svg .ico .bmp .tif .tiff
```

3. **Page Rules**（免費方案 3 條）：

```
# 規則 1：靜態資源長期快取
URL: example.com/images/*
設定: Cache Level = Cache Everything, Edge Cache TTL = 1 month

# 規則 2：Bypass admin 路徑
URL: example.com/admin/*
設定: Cache Level = Bypass
```

4. **瀏覽器快取 TTL**：

```
Cloudflare Dashboard → Caching → Configuration → Browser Cache TTL
建議設 1 month（2592000 秒）
```

5. **自動壓縮**：

```
Cloudflare Dashboard → Speed → Optimization
開啟：Auto Minify (JS, CSS, HTML)
開啟：Brotli 壓縮
```

#### Cloudflare Polish（Pro 方案以上，$20/月）

```
Cloudflare Dashboard → Speed → Optimization → Image Optimization

Polish 設定：
- Lossless：無損壓縮，移除 metadata
- Lossy：有損壓縮，通常省 30-50%（推薦）
- WebP：自動轉換 WebP 給支援的瀏覽器（一定要開）
```

```javascript
// 或用 Node.js 自己組合
// scripts/build-sprite.mjs
import { readdir, readFile, writeFile } from 'fs/promises';
import path from 'path';

async function buildSprite(iconDir, outputPath) {
  const files = await readdir(iconDir);
  const svgFiles = files.filter(f => f.endsWith('.svg'));

  let symbols = '';

  for (const file of svgFiles) {
    const id = `icon-${path.parse(file).name}`;
    let content = await readFile(path.join(iconDir, file), 'utf-8');

    // 提取 viewBox
    const viewBoxMatch = content.match(/viewBox="([^"]+)"/);
    const viewBox = viewBoxMatch ? viewBoxMatch[1] : '0 0 24 24';

    // 提取 SVG 內容（去掉外層 <svg> 標籤）
    const innerContent = content
      .replace(/<svg[^>]*>/, '')
      .replace(/<\/svg>/, '')
      .trim();

    symbols += `  <symbol id="${id}" viewBox="${viewBox}">\n    ${innerContent}\n  </symbol>\n\n`;
  }

  const sprite = `<svg xmlns="http://www.w3.org/2000/svg" style="display: none;">\n${symbols}</svg>`;

  await writeFile(outputPath, sprite);
  console.log(`Sprite 已產生：${svgFiles.length} 個圖標 → ${outputPath}`);
}

buildSprite('./src/icons/', './public/icons.svg');
```

### Inline SVG vs `<img>` 比較

| 方式 | 優點 | 缺點 | 適用場景 |
|------|------|------|----------|
| **Inline SVG** | CSS 可控制顏色/動畫、不多發 HTTP 請求 | 增加 HTML 大小、不能被快取 | 互動式圖標、需變色的圖標 |

| **CSS `background-image`** | 不占 DOM | 無法操控 SVG 內容 | 純裝飾背景 |

### 背景影片壓縮

#### FFmpeg（命令列，最強大）

```bash
# macOS 安裝
brew install ffmpeg

# 基本壓縮（H.264，適合背景影片）
ffmpeg -i input.mp4 \
  -c:v libx264 \
  -crf 28 \
  -preset slow \
  -an \
  -movflags +faststart \
  -vf "scale=1920:-2" \
  output.mp4

# 參數說明：
#   -c:v libx264    → H.264 編碼（最廣泛支援）
#   -crf 28         → 品質（0=無損，23=預設，28=背景影片夠用，51=最差）
#   -preset slow    → 編碼速度（越慢壓越小：ultrafast/fast/medium/slow/veryslow）
#   -an             → 移除音軌（背景影片不需要聲音）
#   -movflags +faststart → 讓影片能邊下載邊播
#   -vf "scale=1920:-2"  → 縮到 1920px 寬（高度自動等比例，-2 確保偶數）

# WebM 格式（更小，但編碼慢）
ffmpeg -i input.mp4 \
  -c:v libvpx-vp9 \
  -crf 35 \
  -b:v 0 \
  -an \
  -vf "scale=1280:-2" \
  output.webm

# 同時產生 MP4 + WebM
ffmpeg -i input.mp4 -c:v libx264 -crf 28 -an -movflags +faststart -vf "scale=1920:-2" bg.mp4
ffmpeg -i input.mp4 -c:v libvpx-vp9 -crf 35 -b:v 0 -an -vf "scale=1280:-2" bg.webm
```

### 影片處理批次腳本

```bash
#!/bin/bash
# scripts/compress-videos.sh
# 用法：bash compress-videos.sh ./videos/raw ./videos/compressed

INPUT_DIR="${1:-.}"
OUTPUT_DIR="${2:-./compressed}"
CRF=28

mkdir -p "$OUTPUT_DIR"

for video in "$INPUT_DIR"/*.{mp4,mov,avi,mkv,MP4,MOV}; do
  [ -f "$video" ] || continue
  filename=$(basename "$video")
  name="${filename%.*}"

  echo "壓縮：$filename"

  # MP4（H.264）
  ffmpeg -y -i "$video" \
    -c:v libx264 -crf $CRF -preset slow \
    -an -movflags +faststart \
    -vf "scale=1920:-2" \
    "$OUTPUT_DIR/${name}.mp4" 2>/dev/null

  # WebM（VP9）
  ffmpeg -y -i "$video" \
    -c:v libvpx-vp9 -crf 35 -b:v 0 \
    -an \
    -vf "scale=1280:-2" \
    "$OUTPUT_DIR/${name}.webm" 2>/dev/null

  # 擷取 poster 圖
  ffmpeg -y -i "$video" \
    -ss 00:00:01 -vframes 1 \
    -vf "scale=1920:-2" \
    "$OUTPUT_DIR/${name}-poster.jpg" 2>/dev/null

  original=$(stat -f%z "$video" 2>/dev/null || stat --format=%s "$video" 2>/dev/null)
  compressed=$(stat -f%z "$OUTPUT_DIR/${name}.mp4" 2>/dev/null || stat --format=%s "$OUTPUT_DIR/${name}.mp4" 2>/dev/null)
  saved=$(echo "scale=1; (1 - $compressed / $original) * 100" | bc)
  echo "  → MP4 省 ${saved}%"
done
```

```
3. Efficiently encode images
   → JPEG/PNG 是否壓縮過
   → 解法：品質 80-85，用 MozJPEG

4. Defer offscreen images
   → 非首屏圖片是否 lazy load
   → 解法：loading="lazy"

5. Image elements have explicit width and height
   → 是否寫了 width/height（避免 CLS）
   → 解法：<img> 都要寫寬高

6. Largest Contentful Paint (LCP) image
   → 首屏最大圖片載入速度
   → 解法：preload + fetchpriority="high"
```

```javascript
// 貼到瀏覽器 Console，找出所有浪費的大圖
(function detectOversizedImages() {
  const images = document.querySelectorAll('img');
  const issues = [];

  images.forEach(img => {
    if (!img.complete || !img.naturalWidth) return;

    const displayWidth = img.clientWidth;
    const displayHeight = img.clientHeight;
    const naturalWidth = img.naturalWidth;
    const naturalHeight = img.naturalHeight;
    const ratio = naturalWidth / (displayWidth || 1);
    const src = img.currentSrc || img.src;

    // 問題 1：實際尺寸超過顯示尺寸 2 倍以上
    if (ratio > 2) {
      issues.push({
        src: src.substring(src.lastIndexOf('/') + 1),
        problem: '尺寸過大',
        detail: `顯示 ${displayWidth}x${displayHeight}，實際 ${naturalWidth}x${naturalHeight}（${ratio.toFixed(1)}x）`,
        suggestion: `建議尺寸 ${displayWidth * 2}px 寬（2x Retina）`,
      });
    }

    // 問題 2：沒有 loading="lazy" 且不在首屏
    const rect = img.getBoundingClientRect();
    if (rect.top > window.innerHeight && !img.loading) {
      issues.push({
        src: src.substring(src.lastIndexOf('/') + 1),
        problem: '非首屏但沒有 lazy load',
        detail: `距離頂部 ${Math.round(rect.top)}px`,
        suggestion: '加上 loading="lazy"',
      });
    }

    // 問題 3：沒有 width/height 屬性
    if (!img.hasAttribute('width') || !img.hasAttribute('height')) {
      issues.push({
        src: src.substring(src.lastIndexOf('/') + 1),
        problem: '缺少 width/height（會造成 CLS）',
        detail: `建議加上 width="${naturalWidth}" height="${naturalHeight}"`,
        suggestion: '加上明確的寬高屬性',
      });
    }

    // 問題 4：沒有 alt
    if (!img.hasAttribute('alt')) {
      issues.push({
        src: src.substring(src.lastIndexOf('/') + 1),
        problem: '缺少 alt 屬性（SEO + 無障礙）',
        suggestion: '加上描述性 alt 文字',
      });
    }
  });

  if (issues.length === 0) {
    console.log('%c 所有圖片都OK！', 'color: green; font-size: 16px;');
  } else {
    console.log(`%c 發現 ${issues.length} 個圖片問題：`, 'color: red; font-size: 16px;');
    console.table(issues);
  }

  // 額外統計
  const totalImages = images.length;
  const lazyImages = document.querySelectorAll('img[loading="lazy"]').length;
  const webpImages = Array.from(images).filter(i => (i.currentSrc || i.src).includes('.webp')).length;

  console.log(`\n圖片統計：共 ${totalImages} 張，${lazyImages} 張 lazy load，${webpImages} 張 WebP`);
})();
```

#### Node.js 專案大圖掃描腳本

```javascript
// scripts/audit-images.mjs
// 掃描專案目錄中的圖片，找出過大的檔案
import { readdir, stat } from 'fs/promises';
import path from 'path';
import sharp from 'sharp';

const IMAGE_DIR = './public/images';
const MAX_FILE_SIZE_KB = 500;       // 超過 500KB 警告
const MAX_DIMENSION = 2000;         // 超過 2000px 警告
const RECOMMENDED_FORMATS = ['.webp', '.avif', '.svg'];

async function scanDir(dir) {
  const entries = await readdir(dir, { withFileTypes: true });
  const results = [];

  for (const entry of entries) {
    const fullPath = path.join(dir, entry.name);

    if (entry.isDirectory()) {
      results.push(...await scanDir(fullPath));
      continue;
    }

    const ext = path.extname(entry.name).toLowerCase();
    if (!['.jpg', '.jpeg', '.png', '.webp', '.avif', '.gif', '.svg'].includes(ext)) continue;

    const fileStat = await stat(fullPath);
    const sizeKB = fileStat.size / 1024;

    const issues = [];

    // 檢查檔案大小
    if (sizeKB > MAX_FILE_SIZE_KB) {
      issues.push(`檔案過大：${sizeKB.toFixed(0)}KB（建議 < ${MAX_FILE_SIZE_KB}KB）`);
    }

    // 檢查格式
    if (['.jpg', '.jpeg', '.png', '.gif'].includes(ext)) {
      issues.push(`建議轉為 WebP 格式（目前是 ${ext}）`);
    }

    // 檢查尺寸（非 SVG）
    if (ext !== '.svg') {
      try {
        const metadata = await sharp(fullPath).metadata();
        if (metadata.width > MAX_DIMENSION || metadata.height > MAX_DIMENSION) {
          issues.push(`尺寸過大：${metadata.width}x${metadata.height}（建議 < ${MAX_DIMENSION}px）`);
        }
      } catch (e) {
        issues.push(`無法讀取圖片 metadata`);
      }
    }

    if (issues.length > 0) {
      results.push({
        file: fullPath,
        size: `${sizeKB.toFixed(0)}KB`,
        issues,
      });
    }
  }

  return results;
}

const issues = await scanDir(IMAGE_DIR);

if (issues.length === 0) {
  console.log('所有圖片都通過檢查！');
} else {
  console.log(`發現 ${issues.length} 個需要最佳化的圖片：\n`);
  issues.forEach(({ file, size, issues }) => {
    console.log(`${file} (${size})`);
    issues.forEach(i => console.log(`  - ${i}`));
    console.log('');
  });
}
```
