```javascript
// 簡易派單：找最近的空閒外送員
async function assignDriver(order) {
  const { lat, lng } = await getRestaurantLocation(order.restaurant_id);

  const drivers = await db.query(`
    SELECT id, name,
      (6371 * acos(
        cos(radians($1)) * cos(radians(lat)) *
        cos(radians(lng) - radians($2)) +
        sin(radians($1)) * sin(radians(lat))
      )) AS distance_km
    FROM drivers
    WHERE status = 'online'
    ORDER BY distance_km ASC
    LIMIT 5
  `, [lat, lng]);

  if (drivers.rows.length === 0) return null;

  // 派給最近的外送員
  const driver = drivers.rows[0];
  await db.query(
    `UPDATE drivers SET status = 'busy' WHERE id = $1`,
    [driver.id]
  );
  await db.query(
    `UPDATE orders SET driver_id = $1 WHERE id = $2`,
    [driver.id, order.id]
  );

  // 推播通知外送員
  await sendPushNotification(driver.id, {
    title: '新訂單',
    body: `前往 ${order.restaurant_name} 取餐`,
    data: { orderId: order.id }
  });

  return driver;
}
```

### 5.2 距離計算 API

```javascript
// GET /api/restaurants?lat=25.033&lng=121.565&radius=3&sort=distance
router.get('/api/restaurants', async (req, res) => {
  const { lat, lng, radius = 3, sort = 'distance', category, keyword } = req.query;

  let query = `
    SELECT *,
      (6371 * acos(
        cos(radians($1)) * cos(radians(lat)) *
        cos(radians(lng) - radians($2)) +
        sin(radians($1)) * sin(radians(lat))
      )) AS distance_km
    FROM restaurants
    WHERE is_open = true
  `;
  const params = [lat, lng];
  let paramIdx = 3;

  // 距離篩選
  if (radius) {
    query += ` AND (6371 * acos(
      cos(radians($1)) * cos(radians(lat)) *
      cos(radians(lng) - radians($2)) +
      sin(radians($1)) * sin(radians(lat))
    )) <= $${paramIdx}`;
    params.push(radius);
    paramIdx++;
  }

  // 分類篩選
  if (category) {
    query += ` AND $${paramIdx} = ANY(categories)`;
    params.push(category);
    paramIdx++;
  }

  // 關鍵字搜尋
  if (keyword) {
    query += ` AND (name ILIKE $${paramIdx} OR EXISTS (
      SELECT 1 FROM menu_items
      WHERE restaurant_id = restaurants.id AND name ILIKE $${paramIdx}
    ))`;
    params.push(`%${keyword}%`);
    paramIdx++;
  }

  // 排序
  const sortMap = {
    distance: 'distance_km ASC',
    rating: 'rating DESC',
    delivery_fee: 'delivery_fee ASC',
    prep_time: 'avg_prep_time_min ASC'
  };
  query += ` ORDER BY ${sortMap[sort] || 'distance_km ASC'} LIMIT 50`;

  const result = await db.query(query, params);
  res.json({ restaurants: result.rows });
});
```

### 5.3 營業時間判斷

```javascript
function isRestaurantOpen(openingHours) {
  // openingHours 格式：{ "mon": ["11:00-14:00", "17:00-21:00"], "tue": [...], ... }
  const now = new Date();
  const dayNames = ['sun', 'mon', 'tue', 'wed', 'thu', 'fri', 'sat'];
  const today = dayNames[now.getDay()];
  const currentTime = now.getHours() * 60 + now.getMinutes();

  const slots = openingHours[today] || [];
  return slots.some(slot => {
    const [start, end] = slot.split('-').map(t => {
      const [h, m] = t.split(':').map(Number);
      return h * 60 + m;
    });
    return currentTime >= start && currentTime <= end;
  });
}
```

## 7. 購物車與結帳

### 7.1 購物車邏輯

```javascript
// 購物車 state（React Context / Zustand）
const useCartStore = create((set, get) => ({
  restaurantId: null,
  restaurantName: '',
  items: [],           // [{ itemId, name, price, qty, options, note, subtotal }]
  deliveryFee: 0,
  minOrder: 0,

  addItem: (item) => set((state) => {
    // 如果換餐廳，清空購物車
    if (state.restaurantId && state.restaurantId !== item.restaurantId) {
      if (!confirm('更換餐廳將清空購物車，確定嗎？')) return state;
      return {
        restaurantId: item.restaurantId,
        restaurantName: item.restaurantName,
        items: [item],
        deliveryFee: item.deliveryFee,
        minOrder: item.minOrder,
      };
    }

    // 同菜品+同選項 → 合併數量
    const existIdx = state.items.findIndex(
      i => i.itemId === item.itemId && JSON.stringify(i.options) === JSON.stringify(item.options)
    );
    if (existIdx >= 0) {
      const newItems = [...state.items];
      newItems[existIdx].qty += item.qty;
      newItems[existIdx].subtotal = newItems[existIdx].qty * newItems[existIdx].price;
      return { items: newItems };
    }

    return {
      restaurantId: item.restaurantId || state.restaurantId,
      restaurantName: item.restaurantName || state.restaurantName,
      items: [...state.items, item],
    };
  }),

  getTotal: () => {
    const { items, deliveryFee } = get();
    const subtotal = items.reduce((sum, i) => sum + i.subtotal, 0);
    return { subtotal, deliveryFee, total: subtotal + deliveryFee };
  },

  canCheckout: () => {
    const { items, minOrder } = get();
    const subtotal = items.reduce((sum, i) => sum + i.subtotal, 0);
    return items.length > 0 && subtotal >= minOrder;
  },
}));
```

### 7.2 外送費計算

```javascript
function calculateDeliveryFee(distanceKm, orderAmount) {
  // 基礎費率
  let fee = 30; // 基本費 30 元

  if (distanceKm <= 1) fee += 0;
  else if (distanceKm <= 3) fee += Math.ceil((distanceKm - 1) * 10);
  else if (distanceKm <= 5) fee += 20 + Math.ceil((distanceKm - 3) * 15);
  else fee += 50 + Math.ceil((distanceKm - 5) * 20);

  // 滿額免運
  if (orderAmount >= 500) return 0;
  if (orderAmount >= 300) return Math.floor(fee * 0.5);

  return fee;
}
```

## 8. 訂單追蹤系統

### 8.1 訂單狀態機

```

                                                         |
                                                    取餐 |
                                                         v
                                                     picked_up ──配送──> delivering ──送達──> delivered

任何狀態 ──取消──> cancelled（有條件限制）
```

### 8.2 即時追蹤 WebSocket

```javascript
// 後端：訂單狀態推送
const WebSocket = require('ws');
const wss = new WebSocket.Server({ port: 8080 });

// 訂單房間管理
const orderRooms = new Map(); // orderId -> Set<ws>

wss.on('connection', (ws, req) => {
  const orderId = new URL(req.url, 'http://localhost').searchParams.get('orderId');
  if (!orderId) return ws.close();

  if (!orderRooms.has(orderId)) orderRooms.set(orderId, new Set());
  orderRooms.get(orderId).add(ws);

  ws.on('close', () => {
    orderRooms.get(orderId)?.delete(ws);
  });
});

function broadcastOrderUpdate(orderId, data) {
  const clients = orderRooms.get(orderId);
  if (!clients) return;
  const msg = JSON.stringify(data);
  clients.forEach(ws => {
    if (ws.readyState === WebSocket.OPEN) ws.send(msg);
  });
}

// 外送員位置更新
function updateDriverLocation(orderId, lat, lng) {
  broadcastOrderUpdate(orderId, {
    type: 'driver_location',
    lat, lng,
    timestamp: Date.now()
  });
}

// 訂單狀態變更
function updateOrderStatus(orderId, status, message) {
  broadcastOrderUpdate(orderId, {
    type: 'status_update',
    status,
    message,
    timestamp: Date.now()
  });
}
```

### 8.3 預估送達時間

```javascript
function estimateDeliveryTime(order) {
  const prepTime = order.restaurant.avg_prep_time_min || 20;
  const distanceKm = order.distance_km || 3;
  const deliverySpeed = 25; // 機車平均 25 km/h
  const deliveryTime = (distanceKm / deliverySpeed) * 60;
  const buffer = 5; // 緩衝 5 分鐘

  return Math.ceil(prepTime + deliveryTime + buffer);
}
```

```javascript
// 取得使用者目前位置
function getCurrentLocation() {
  return new Promise((resolve, reject) => {
    if (!navigator.geolocation) return reject('不支援定位');
    navigator.geolocation.getCurrentPosition(
      pos => resolve({ lat: pos.coords.latitude, lng: pos.coords.longitude }),
      err => reject(err.message),
      { enableHighAccuracy: true, timeout: 10000 }
    );
  });
}
```

```javascript
async function checkDeliveryAvailable(restaurantId, userLat, userLng) {
```

```html
let map, driverMarker, ws;

function initTrackingMap(restaurantLat, restaurantLng, deliveryLat, deliveryLng) {
  map = new google.maps.Map(document.getElementById('tracking-map'), {
    center: { lat: restaurantLat, lng: restaurantLng },
    zoom: 14,
  });

  // 餐廳標記
  new google.maps.Marker({
    position: { lat: restaurantLat, lng: restaurantLng },
    map, icon: '/icons/restaurant-pin.png', title: '餐廳'
  });

  // 送達地址標記
  new google.maps.Marker({
    position: { lat: deliveryLat, lng: deliveryLng },
    map, icon: '/icons/home-pin.png', title: '送達地址'
  });

  // 外送員標記（動態更新）
  driverMarker = new google.maps.Marker({
    position: { lat: restaurantLat, lng: restaurantLng },
    map, icon: '/icons/driver-pin.png', title: '外送員'
  });
}

function connectOrderTracking(orderId) {
  ws = new WebSocket(`ws://localhost:8080?orderId=${orderId}`);

  ws.onmessage = (event) => {
    const data = JSON.parse(event.data);

    if (data.type === 'status_update') {
      updateStatusTimeline(data.status);
      document.getElementById('status-message').textContent = getStatusMessage(data.status);
    }

    if (data.type === 'driver_location') {
      driverMarker.setPosition({ lat: data.lat, lng: data.lng });
      map.panTo({ lat: data.lat, lng: data.lng });
    }

    if (data.type === 'eta_update') {
      document.getElementById('eta-min').textContent = data.minutes;
    }
  };
}

function updateStatusTimeline(currentStatus) {
  const steps = ['confirmed', 'preparing', 'picked_up', 'delivering', 'delivered'];
  const currentIdx = steps.indexOf(currentStatus);

  steps.forEach((step, i) => {
    const el = document.getElementById(`step-${step}`);
    el.classList.remove('active', 'done');
    if (i < currentIdx) el.classList.add('done');
    if (i === currentIdx) el.classList.add('active');
  });
}

function getStatusMessage(status) {
  const messages = {
    confirmed: '餐廳已確認您的訂單',
    preparing: '餐廳正在準備您的餐點...',
    ready: '餐點已備好，等待外送員取餐',
    picked_up: '外送員已取餐，正前往您的位置',
    delivering: '外送員正在配送中...',
    delivered: '餐點已送達，請享用！',
  };
  return messages[status] || '';
}

// 頁面初始化
async function initTracking(orderId) {
  const res = await fetch(`/api/orders/${orderId}`);
  const order = await res.json();

  initTrackingMap(
    order.restaurant.lat, order.restaurant.lng,
    order.delivery_address.lat, order.delivery_address.lng
  );
  updateStatusTimeline(order.status);
  connectOrderTracking(orderId);
}
```

## 常見問題

| 問題 | 解法 |
|------|------|
| 定位不準 | 用 Google Geocoding 反查地址，讓用戶確認 |
| 餐廳太多載入慢 | 分頁 + 虛擬滾動，每頁 20 家 |
| 外送員位置跳動 | 位置更新頻率 10 秒一次，前端做平滑動畫 |
| 跨餐廳點餐 | 不支援，切換餐廳時提醒清空購物車 |
| 預約單處理 | 加 scheduled_at 欄位，到時間才推給餐廳 |
