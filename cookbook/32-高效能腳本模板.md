# Cookbook 32: 高效能腳本模板庫

> 當需要 create_task 派工寫腳本時，直接從這裡複製對應的模板，修改後使用。這能確保程式碼品質，並大幅提升效率，即使是基礎模型也能產出可靠的腳本。

## 1. 通用 Bash 腳本模板

``bash
#!/bin/bash

# 嚴格模式，出錯立即停止
set -euo pipefail

# --- 常數定義 ---
TARGET_DIR="/Users/sky770825/.openclaw/workspace/output"
LOG_FILE="/tmp/myscript.log"

# --- 函數定義 ---
log() {
  echo "$(date +'%Y-%m-%d %H:%M:%S') - $1" | tee -a "$LOG_FILE"
}

cleanup() {
  log "執行清理工作..."
  # 在此添加清理程式碼，例如刪除臨時檔案
}

# --- 主邏輯 ---
main() {
  # 設置陷阱，腳本結束或被中斷時執行 cleanup 函數
  trap cleanup EXIT

  log "腳本開始執行"

  if [ ! -d "$TARGET_DIR" ]; then
    log "目標目錄不存在，正在建立: $TARGET_DIR"
    mkdir -p "$TARGET_DIR"
  fi

  log "正在執行核心任務..."
  # 在此添加你的核心程式碼
  echo "Hello World" > "$TARGET_DIR/result.txt"

  log "腳本執行成功"
}

# --- 執行入口 ---
main "$@"

`

## 2. Curl API 呼叫與 JSON 處理模板 (需安裝 jq)

`bash
#!/bin/bash
set -euo pipefail

API_URL="https://api.example.com/data"
OUTPUT_FILE="/Users/sky770825/.openclaw/workspace/notes/api_data.json"

log() { echo "$(date +'%Y-%m-%d %H:%M:%S') - $1"; }

main() {
  log "正在呼叫 API: $API_URL"
  RESPONSE=$(curl -s -H "Accept: application/json" "$API_URL")

  # 檢查 curl 是否成功
  if [ $? -ne 0 ]; then
    log "錯誤：Curl 命令執行失敗"
    exit 1
  fi

  # 檢查回傳的是否為有效的 JSON
  if ! echo "$RESPONSE" | jq . > /dev/null 2>&1; then
    log "錯誤：API 未返回有效的 JSON"
    log "收到的回應: $RESPONSE"
    exit 1
  fi

  log "API 呼叫成功，正在解析並儲存資料..."
  # 提取特定欄位，例如 'data.items'
  echo "$RESPONSE" | jq '.data.items' > "$OUTPUT_FILE"

  log "資料已成功儲存到 $OUTPUT_FILE"
}

main "$@"
`
