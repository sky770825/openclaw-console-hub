### 1.1 Rich Menu（圖文選單）完整設定

```typescript
import axios from 'axios';
```

```typescript
// 處理 LINE Webhook 事件
export async function handleLineWebhook(event: WebhookEvent): Promise<void> {
  if (event.type !== 'message' || event.message.type !== 'text') return;

  const userText = event.message.text.trim();
  const replyToken = event.replyToken;

  // 1. 關鍵字匹配
  for (const rule of keywordRules) {
    const matched = rule.exactMatch
      ? rule.keywords.includes(userText)
      : rule.keywords.some((kw) => userText.includes(kw));

    if (matched) {
      await replyMessage(replyToken, [rule.response]);
      if (rule.trackEvent) {
        await trackConversion(rule.trackEvent, event.source.userId || '');
      }
      return;
    }
  }

  // 2. 無匹配 → AI 回覆（用 Gemini Flash 省成本）
  const aiReply = await getAIReply(userText);
  await replyMessage(replyToken, [{ type: 'text', text: aiReply }]);
}

async function replyMessage(
  replyToken: string,
  messages: (TextMessage | FlexMessage)[]
): Promise<void> {
  await axios.post(
    LINE_REPLY_API,
    { replyToken, messages },
    {
      headers: {
        Authorization: `Bearer ${CHANNEL_TOKEN}`,
        'Content-Type': 'application/json',
      },
    }
  );
}

async function getAIReply(userMessage: string): Promise<string> {
  const GEMINI_KEY = process.env.GEMINI_API_KEY!;
  const { data } = await axios.post(
    `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash:generateContent?key=${GEMINI_KEY}`,
    {
      contents: [
        {
          parts: [
            {
              text: `你是一個友善的客服助理。用戶問題：「${userMessage}」\n請用繁體中文回覆，控制在 100 字以內。如果不確定答案，引導用戶聯繫真人客服。`,
            },
          ],
        },
      ],
    }
  );
  return data.candidates?.[0]?.content?.parts?.[0]?.text || '感謝您的訊息，我們會盡快回覆您！';
}
```

### 1.3 推播訊息（Broadcast / Narrowcast / Multicast）

```typescript
// src/marketing/line/push-message.ts
import axios from 'axios';

const LINE_API = 'https://api.line.me/v2/bot/message';
const CHANNEL_TOKEN = process.env.LINE_CHANNEL_ACCESS_TOKEN!;

const headers = {
  Authorization: `Bearer ${CHANNEL_TOKEN}`,
  'Content-Type': 'application/json',
};

// 全體推播（計費！慎用）
export async function broadcastMessage(text: string): Promise<void> {
  await axios.post(`${LINE_API}/broadcast`, {
    messages: [{ type: 'text', text }],
  }, { headers });
}

// 指定用戶群推播
export async function multicastMessage(
  userIds: string[],
  text: string
): Promise<void> {
  // 每次最多 500 人
  const chunks = chunkArray(userIds, 500);
  for (const chunk of chunks) {
    await axios.post(`${LINE_API}/multicast`, {
      to: chunk,
      messages: [{ type: 'text', text }],
    }, { headers });
    // 避免超過 rate limit
    await sleep(100);
  }
}

// Narrowcast：依條件篩選推播（性別、年齡、地區、好友時間）
export async function narrowcastMessage(
  text: string,
  options: {
    gender?: 'male' | 'female';
    ageGroup?: string;     // 例如 'age_25_29'
    regionCode?: string;   // JP region code
  }
): Promise<void> {
  const filter: any = { type: 'operator', and: [] };

  if (options.gender) {
    filter.and.push({
      type: 'demographic',
      demographicFilter: {
        type: 'gender',
        oneOf: [options.gender],
      },
    });
  }

  if (options.ageGroup) {
    filter.and.push({
      type: 'demographic',
      demographicFilter: {
        type: 'age',
        gte: options.ageGroup,
      },
    });
  }

  await axios.post(`${LINE_API}/narrowcast`, {
    messages: [{ type: 'text', text }],
    recipient: filter.and.length > 0 ? filter : undefined,
  }, { headers });
}

// 查詢推播配額
export async function getMessageQuota(): Promise<{
  totalUsage: number;
  remaining: number;
  limit: number;
}> {
  const [quotaRes, consumptionRes] = await Promise.all([
    axios.get(`${LINE_API.replace('/message', '')}/message/quota`, { headers }),
    axios.get(`${LINE_API.replace('/message', '')}/message/quota/consumption`, { headers }),
  ]);

  return {
    totalUsage: consumptionRes.data.totalUsage,
    remaining: quotaRes.data.value - consumptionRes.data.totalUsage,
    limit: quotaRes.data.value,
  };
}

function chunkArray<T>(arr: T[], size: number): T[][] {
  const chunks: T[][] = [];
  for (let i = 0; i < arr.length; i += size) {
    chunks.push(arr.slice(i, i + size));
  }
  return chunks;
}

function sleep(ms: number): Promise<void> {
  return new Promise((r) => setTimeout(r, ms));
}
```

```typescript
// src/marketing/meta/fb-ads-client.ts
import axios, { AxiosInstance } from 'axios';

const META_API_VERSION = 'v19.0';
const META_BASE_URL = `https://graph.facebook.com/${META_API_VERSION}`;

interface MetaAdsConfig {
  accessToken: string;     // 長效 Page Access Token
  adAccountId: string;     // act_XXXXXXXXXX
  pixelId?: string;        // Facebook Pixel ID
  pageId?: string;
}

export class MetaAdsClient {
  private api: AxiosInstance;
  private config: MetaAdsConfig;

  constructor(config: MetaAdsConfig) {
    this.config = config;
    this.api = axios.create({
      baseURL: META_BASE_URL,
      params: { access_token: config.accessToken },
    });
  }

  // ====== Campaign（廣告活動） ======

  async createCampaign(params: {
    name: string;
    objective:
      | 'OUTCOME_AWARENESS'
      | 'OUTCOME_TRAFFIC'
      | 'OUTCOME_ENGAGEMENT'
      | 'OUTCOME_LEADS'
      | 'OUTCOME_SALES';
    dailyBudget?: number;    // 單位：分（100 = NT$1）
    lifetimeBudget?: number;
    status?: 'ACTIVE' | 'PAUSED';
  }): Promise<{ id: string }> {
    const { data } = await this.api.post(`/${this.config.adAccountId}/campaigns`, {
      name: params.name,
      objective: params.objective,
      special_ad_categories: [],
      status: params.status || 'PAUSED',
      daily_budget: params.dailyBudget,
      lifetime_budget: params.lifetimeBudget,
    });
    return data;
  }

  // ====== Ad Set（廣告組合）======

  async createAdSet(params: {
    name: string;
    campaignId: string;
    dailyBudget: number;
    targeting: {
      geoLocations: { countries?: string[]; cities?: { key: string }[] };
      ageMin?: number;
      ageMax?: number;
      genders?: (1 | 2)[];   // 1=男 2=女
      interests?: { id: string; name: string }[];
      customAudiences?: { id: string }[];
    };
    placements?: ('facebook' | 'instagram' | 'audience_network')[];
    startTime?: string;
    endTime?: string;
  }): Promise<{ id: string }> {
    const targeting: any = {
      geo_locations: {
        countries: params.targeting.geoLocations.countries,
        cities: params.targeting.geoLocations.cities,
      },
      age_min: params.targeting.ageMin || 18,
      age_max: params.targeting.ageMax || 65,
    };

    if (params.targeting.genders) {
      targeting.genders = params.targeting.genders;
    }

    if (params.targeting.interests) {
      targeting.flexible_spec = [
        { interests: params.targeting.interests },
      ];
    }

    if (params.targeting.customAudiences) {
      targeting.custom_audiences = params.targeting.customAudiences;
    }

    const { data } = await this.api.post(`/${this.config.adAccountId}/adsets`, {
      name: params.name,
      campaign_id: params.campaignId,
      daily_budget: params.dailyBudget,
      billing_event: 'IMPRESSIONS',
      optimization_goal: 'LINK_CLICKS',
      targeting,
      start_time: params.startTime,
      end_time: params.endTime,
      status: 'PAUSED',
    });
    return data;
  }

  // ====== Ad Creative（廣告素材）======

  async createAdCreative(params: {
    name: string;
    pageId: string;
    imageHash?: string;
    imageUrl?: string;
    headline: string;
    description: string;
    linkUrl: string;
    callToAction?: 'LEARN_MORE' | 'SHOP_NOW' | 'SIGN_UP' | 'CONTACT_US';
  }): Promise<{ id: string }> {
    const objectStorySpec: any = {
      page_id: params.pageId,
      link_data: {
        link: params.linkUrl,
        message: params.description,
        name: params.headline,
        call_to_action: {
          type: params.callToAction || 'LEARN_MORE',
          value: { link: params.linkUrl },
        },
      },
    };

    if (params.imageHash) {
      objectStorySpec.link_data.image_hash = params.imageHash;
    }

    const { data } = await this.api.post(`/${this.config.adAccountId}/adcreatives`, {
      name: params.name,
      object_story_spec: objectStorySpec,
    });
    return data;
  }

  // ====== Ad（廣告）======

  async createAd(params: {
    name: string;
    adSetId: string;
    creativeId: string;
  }): Promise<{ id: string }> {
    const { data } = await this.api.post(`/${this.config.adAccountId}/ads`, {
      name: params.name,
      adset_id: params.adSetId,
      creative: { creative_id: params.creativeId },
      status: 'PAUSED',
    });
    return data;
  }

  // ====== 報表 ======

  async getCampaignInsights(
    campaignId: string,
    dateRange: { since: string; until: string }
  ): Promise<any> {
    const { data } = await this.api.get(`/${campaignId}/insights`, {
      params: {
        fields: 'impressions,reach,clicks,cpc,cpm,ctr,spend,actions,cost_per_action_type',
        time_range: JSON.stringify(dateRange),
        time_increment: 1,
      },
    });
    return data.data;
  }

  // ====== 圖片上傳 ======

  async uploadAdImage(imagePath: string): Promise<string> {
    const FormData = (await import('form-data')).default;
    const fs = await import('fs');
    const form = new FormData();
    form.append('filename', fs.createReadStream(imagePath));

    const { data } = await this.api.post(
      `/${this.config.adAccountId}/adimages`,
      form,
      { headers: form.getHeaders() }
    );
    const images = data.images;
    const hash = Object.values(images)[0] as any;
    return hash.hash;
  }
}
```

```typescript
export function initFacebookPixel(pixelId: string): void {
  if (typeof window === 'undefined') return;

  // 標準 Pixel 初始化
  const script = document.createElement('script');
  script.innerHTML = `
    !function(f,b,e,v,n,t,s)

    fbq('init', '${pixelId}');
    fbq('track', 'PageView');
  `;
```

```typescript
import axios from 'axios';

const META_BASE = 'https://graph.facebook.com/v19.0';

interface IGPublishConfig {
  accessToken: string;
  igBusinessAccountId: string;  // Instagram Business Account ID
}

export class InstagramPublisher {
  private config: IGPublishConfig;

  constructor(config: IGPublishConfig) {
    this.config = config;
  }

  // 發布單張圖片貼文
  async publishPhoto(params: {
    imageUrl: string;      // 必須是公開可存取的 URL
    caption: string;
    hashtags?: string[];
  }): Promise<string> {
    const caption = params.hashtags
      ? `${params.caption}\n\n${params.hashtags.map((h) => `#${h}`).join(' ')}`
      : params.caption;

    // Step 1: 建立 media container
    const { data: container } = await axios.post(
      `${META_BASE}/${this.config.igBusinessAccountId}/media`,
      null,
      {
        params: {
          image_url: params.imageUrl,
          caption,
          access_token: this.config.accessToken,
        },
      }
    );

    // Step 2: 等待 media 處理完成
    await this.waitForMediaReady(container.id);

    // Step 3: 發布
    const { data: publish } = await axios.post(
      `${META_BASE}/${this.config.igBusinessAccountId}/media_publish`,
      null,
      {
        params: {
          creation_id: container.id,
          access_token: this.config.accessToken,
        },
      }
    );

    return publish.id;
  }

  // 發布 Carousel（多張圖片輪播）
  async publishCarousel(params: {
    imageUrls: string[];   // 2-10 張
    caption: string;
  }): Promise<string> {
    // Step 1: 為每張圖建立子 container
    const childIds: string[] = [];
    for (const url of params.imageUrls) {
      const { data } = await axios.post(
        `${META_BASE}/${this.config.igBusinessAccountId}/media`,
        null,
        {
          params: {
            image_url: url,
            is_carousel_item: true,
            access_token: this.config.accessToken,
          },
        }
      );
      childIds.push(data.id);
    }

    // Step 2: 建立 carousel container
    const { data: carousel } = await axios.post(
      `${META_BASE}/${this.config.igBusinessAccountId}/media`,
      null,
      {
        params: {
          media_type: 'CAROUSEL',
          children: childIds.join(','),
          caption: params.caption,
          access_token: this.config.accessToken,
        },
      }
    );

    await this.waitForMediaReady(carousel.id);

    // Step 3: 發布
    const { data: publish } = await axios.post(
      `${META_BASE}/${this.config.igBusinessAccountId}/media_publish`,
      null,
      {
        params: {
          creation_id: carousel.id,
          access_token: this.config.accessToken,
        },
      }
    );

    return publish.id;
  }

  private async waitForMediaReady(
    containerId: string,
    maxRetries = 20
  ): Promise<void> {
    for (let i = 0; i < maxRetries; i++) {
      const { data } = await axios.get(`${META_BASE}/${containerId}`, {
        params: {
          fields: 'status_code',
          access_token: this.config.accessToken,
        },
      });
      if (data.status_code === 'FINISHED') return;
      if (data.status_code === 'ERROR') throw new Error('Media processing failed');
      await new Promise((r) => setTimeout(r, 3000));
    }
    throw new Error('Media processing timeout');
  }
}
```

## 5. Landing Page A/B Testing

### 5.1 A/B Test 引擎

```typescript
// src/marketing/ab-test/ab-engine.ts
import crypto from 'crypto';
import { createClient, SupabaseClient } from '@supabase/supabase-js';

interface ABTestConfig {
  testId: string;
  variants: {
    id: string;
    name: string;
    weight: number;  // 權重百分比（加總需為 100）
  }[];
  startDate: string;
  endDate?: string;
}

interface ABTestResult {
  variantId: string;
  visitors: number;
  conversions: number;
  conversionRate: number;
  revenue: number;
  confidence: number;    // 統計顯著性
}

// Supabase 表結構
/*
CREATE TABLE IF NOT EXISTS ab_tests (
  id TEXT PRIMARY KEY,
  name TEXT NOT NULL,
  variants JSONB NOT NULL,
  start_date TIMESTAMPTZ NOT NULL,
  end_date TIMESTAMPTZ,
  status TEXT DEFAULT 'active',
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS ab_test_assignments (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  test_id TEXT REFERENCES ab_tests(id),
  visitor_id TEXT NOT NULL,
  variant_id TEXT NOT NULL,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(test_id, visitor_id)
);

CREATE TABLE IF NOT EXISTS ab_test_conversions (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  test_id TEXT NOT NULL,
  visitor_id TEXT NOT NULL,
  variant_id TEXT NOT NULL,
  event_type TEXT NOT NULL,
  event_value NUMERIC DEFAULT 0,
  created_at TIMESTAMPTZ DEFAULT NOW()
);
*/

export class ABTestEngine {
  private supabase: SupabaseClient;

  constructor(supabaseUrl: string, supabaseKey: string) {
    this.supabase = createClient(supabaseUrl, supabaseKey);
  }

  // 建立 A/B 測試
  async createTest(config: ABTestConfig): Promise<void> {
    const totalWeight = config.variants.reduce((sum, v) => sum + v.weight, 0);
    if (totalWeight !== 100) {
      throw new Error(`Variant weights must sum to 100, got ${totalWeight}`);
    }

    await this.supabase.from('ab_tests').insert({
      id: config.testId,
      name: config.testId,
      variants: config.variants,
      start_date: config.startDate,
      end_date: config.endDate,
    });
  }

  // 分配訪客到 variant（確定性 hash，同一訪客永遠得到相同結果）
  async assignVariant(testId: string, visitorId: string): Promise<string> {
    // 先查有沒有已分配的
    const { data: existing } = await this.supabase
      .from('ab_test_assignments')
      .select('variant_id')
      .eq('test_id', testId)
      .eq('visitor_id', visitorId)
      .single();

    if (existing) return existing.variant_id;

    // 用 hash 決定分組
    const { data: test } = await this.supabase
      .from('ab_tests')
      .select('variants')
      .eq('id', testId)
      .single();

    if (!test) throw new Error(`Test ${testId} not found`);

    const variants = test.variants as ABTestConfig['variants'];
    const hash = crypto
      .createHash('md5')
      .update(`${testId}:${visitorId}`)
      .digest('hex');
    const hashNum = parseInt(hash.substring(0, 8), 16) % 100;

    let cumulative = 0;
    let assignedVariant = variants[0].id;
    for (const variant of variants) {
      cumulative += variant.weight;
      if (hashNum < cumulative) {
        assignedVariant = variant.id;
        break;
      }
    }

    // 記錄分配
    await this.supabase.from('ab_test_assignments').insert({
      test_id: testId,
      visitor_id: visitorId,
      variant_id: assignedVariant,
    });

    return assignedVariant;
  }

  // 記錄轉換
  async trackConversion(
    testId: string,
    visitorId: string,
    eventType: string,
    value: number = 0
  ): Promise<void> {
    const variantId = await this.assignVariant(testId, visitorId);
    await this.supabase.from('ab_test_conversions').insert({
      test_id: testId,
      visitor_id: visitorId,
      variant_id: variantId,
      event_type: eventType,
      event_value: value,
    });
  }

  // 取得測試結果（含統計顯著性）
  async getResults(testId: string): Promise<ABTestResult[]> {
    const { data: test } = await this.supabase
      .from('ab_tests')
      .select('variants')
      .eq('id', testId)
      .single();

    if (!test) throw new Error(`Test ${testId} not found`);

    const variants = test.variants as ABTestConfig['variants'];
    const results: ABTestResult[] = [];

    for (const variant of variants) {
      const { count: visitors } = await this.supabase
        .from('ab_test_assignments')
        .select('*', { count: 'exact', head: true })
        .eq('test_id', testId)
        .eq('variant_id', variant.id);

      const { data: conversions } = await this.supabase
        .from('ab_test_conversions')
        .select('event_value')
        .eq('test_id', testId)
        .eq('variant_id', variant.id);

      const conversionCount = conversions?.length || 0;
      const totalRevenue = conversions?.reduce(
        (sum, c) => sum + (c.event_value || 0), 0
      ) || 0;
      const visitorCount = visitors || 0;

      results.push({
        variantId: variant.id,
        visitors: visitorCount,
        conversions: conversionCount,
        conversionRate: visitorCount > 0
          ? (conversionCount / visitorCount) * 100
          : 0,
        revenue: totalRevenue,
        confidence: 0, // 下面計算
      });
    }

    // 計算統計顯著性（Z-test for proportions）
    if (results.length === 2) {
      const [a, b] = results;
      const p1 = a.visitors > 0 ? a.conversions / a.visitors : 0;
      const p2 = b.visitors > 0 ? b.conversions / b.visitors : 0;
      const pPooled =
        (a.conversions + b.conversions) / (a.visitors + b.visitors);
      const se = Math.sqrt(
        pPooled * (1 - pPooled) * (1 / a.visitors + 1 / b.visitors)
      );

      if (se > 0) {
        const z = Math.abs(p1 - p2) / se;
        // Z-score to confidence
        const confidence = this.zToConfidence(z);
        results[0].confidence = confidence;
        results[1].confidence = confidence;
      }
    }

    return results;
  }

  // Z-score 轉信心水準
  private zToConfidence(z: number): number {
    // 近似計算
    if (z >= 2.576) return 99;
    if (z >= 1.96) return 95;
    if (z >= 1.645) return 90;
    if (z >= 1.28) return 80;
    return Math.round(z * 40);  // 粗略估算
  }
```

```typescript
        },
```

```typescript
        personalizations,
        from: params.from,
        subject: params.subject,
        templateId: params.templateId,
        sendAt: params.sendAt,
      } as any);

      // 避免 rate limit
      await sleep(200);
    }
  }
```

```typescript
interface DripStep {

  subject: string;
  templateId: string;

}

interface DripCampaignConfig {
  name: string;
  triggerEvent: string;     // 例如 'signup', 'purchase', 'abandoned_cart'
  steps: DripStep[];
}

export class DripCampaignManager {
  private supabase: SupabaseClient;

  constructor(supabaseUrl: string, supabaseKey: string) {
    this.supabase = createClient(supabaseUrl, supabaseKey);
  }

  // 建立 Drip Campaign
  async createCampaign(config: DripCampaignConfig): Promise<string> {
    const { data, error } = await this.supabase
      .from('drip_campaigns')
      .insert({
        name: config.name,
        trigger_event: config.triggerEvent,
        steps: config.steps,
      })
      .select('id')
      .single();

    if (error) throw error;
    return data.id;
  }

  // 將用戶加入 Drip（觸發時呼叫）
  async enrollUser(campaignId: string, email: string): Promise<void> {
    const { data: campaign } = await this.supabase
      .from('drip_campaigns')
      .select('steps')
      .eq('id', campaignId)
      .single();

    if (!campaign) throw new Error('Campaign not found');

    const steps = campaign.steps as DripStep[];
    const firstDelay = steps[0]?.delayHours || 0;
    const nextSendAt = new Date(
      Date.now() + firstDelay * 60 * 60 * 1000
    ).toISOString();

    await this.supabase.from('drip_enrollments').upsert({
      campaign_id: campaignId,
      email,
      current_step: 0,
      status: 'active',
      next_send_at: nextSendAt,
    });
  }

  // 定時執行器（每小時跑一次）

    const now = new Date().toISOString();

    // 找出所有需要發送的 enrollment
    const { data: pendingList, error } = await this.supabase
      .from('drip_enrollments')
      .select('*, drip_campaigns(steps)')
      .eq('status', 'active')
      .lte('next_send_at', now)
      .limit(100);

    if (error || !pendingList) return 0;

    let sentCount = 0;

    for (const enrollment of pendingList) {
      const steps = (enrollment as any).drip_campaigns.steps as DripStep[];
      const currentStep = steps[enrollment.current_step];

      if (!currentStep) {
        // 已完成所有步驟

          .from('drip_enrollments')
          .update({ status: 'completed', completed_at: now })
          .eq('id', enrollment.id);

            .from('drip_enrollments')
            .update({
              current_step: nextStepIndex,

            .from('drip_enrollments')
            .update({ status: 'completed', completed_at: now })
            .eq('id', enrollment.id);
        }

      } catch (err) {

      }
```

### 7.2 最佳發文時段分析

```typescript
// src/marketing/scheduler/best-time-analyzer.ts
import { createClient } from '@supabase/supabase-js';

interface TimeSlotStats {
  dayOfWeek: number;   // 0=Sunday, 6=Saturday
  hour: number;        // 0-23
  avgEngagement: number;
  postCount: number;
}

export class BestTimeAnalyzer {
  private supabase;

  constructor(supabaseUrl: string, supabaseKey: string) {
    this.supabase = createClient(supabaseUrl, supabaseKey);
  }

  // 分析歷史數據，找出最佳發文時段
  async analyzeBestTimes(
    platform: string,
    days = 90
  ): Promise<TimeSlotStats[]> {
    const since = new Date(
      Date.now() - days * 24 * 60 * 60 * 1000
    ).toISOString();

    const { data } = await this.supabase
      .from('scheduled_posts')
      .select('published_at, metadata')
      .eq('platform', platform)
      .eq('status', 'published')
      .gte('published_at', since);

    if (!data?.length) return [];

    // 統計每個時段的平均互動率
    const slotMap = new Map<string, { total: number; count: number }>();

    for (const post of data) {
      if (!post.published_at) continue;
      const date = new Date(post.published_at);
      const key = `${date.getDay()}-${date.getHours()}`;
      const engagement = post.metadata?.engagement || 0;

      const existing = slotMap.get(key) || { total: 0, count: 0 };
      existing.total += engagement;
      existing.count += 1;
      slotMap.set(key, existing);
    }

    const results: TimeSlotStats[] = [];
    for (const [key, stats] of slotMap) {
      const [day, hour] = key.split('-').map(Number);
      results.push({
        dayOfWeek: day,
        hour,
        avgEngagement: stats.count > 0 ? stats.total / stats.count : 0,
        postCount: stats.count,
      });
    }

    return results.sort((a, b) => b.avgEngagement - a.avgEngagement);
  }

  // 推薦下一個最佳發文時間
  async suggestNextPostTime(platform: string): Promise<Date> {
    const bestTimes = await this.analyzeBestTimes(platform);

    if (bestTimes.length === 0) {
      // 預設最佳時段：週三 20:00（台灣社群通用黃金時段）
      const next = new Date();
      next.setDate(next.getDate() + ((3 - next.getDay() + 7) % 7 || 7));
      next.setHours(20, 0, 0, 0);
      return next;
    }

    // 取互動率最高的時段
    const best = bestTimes[0];
    const now = new Date();
    const next = new Date();

    // 找到下一個符合 dayOfWeek + hour 的時間
    const daysUntil = (best.dayOfWeek - now.getDay() + 7) % 7 || 7;
    next.setDate(now.getDate() + daysUntil);
    next.setHours(best.hour, 0, 0, 0);

    return next;
  }
}
```

### 9.1 漏斗追蹤系統（TOFU → MOFU → BOFU）

```typescript
// src/marketing/funnel/funnel-tracker.ts
import { createClient, SupabaseClient } from '@supabase/supabase-js';

// Supabase 表結構
/*
CREATE TABLE IF NOT EXISTS funnel_events (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  visitor_id TEXT NOT NULL,
  stage TEXT NOT NULL,
  event_name TEXT NOT NULL,
  source TEXT,
  metadata JSONB DEFAULT '{}',
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_funnel_stage ON funnel_events(stage, created_at);
CREATE INDEX idx_funnel_visitor ON funnel_events(visitor_id);
*/

type FunnelStage =
  | 'awareness'        // TOFU: 看到廣告/搜尋到
  | 'interest'         // TOFU: 點進 Landing Page
  | 'consideration'    // MOFU: 加入 LINE / 訂閱 Email
  | 'intent'           // MOFU: 加入購物車 / 填表單
  | 'evaluation'       // BOFU: 比較方案 / 詢問客服
  | 'purchase';        // BOFU: 下單 / 付款完成

export class FunnelTracker {
  private supabase: SupabaseClient;

  constructor(supabaseUrl: string, supabaseKey: string) {
    this.supabase = createClient(supabaseUrl, supabaseKey);
  }

  async trackEvent(params: {
    visitorId: string;
    stage: FunnelStage;
    eventName: string;
    source?: string;
    metadata?: Record<string, any>;
  }): Promise<void> {
    await this.supabase.from('funnel_events').insert({
      visitor_id: params.visitorId,
      stage: params.stage,
      event_name: params.eventName,
      source: params.source,
      metadata: params.metadata || {},
    });
  }

  // 漏斗轉換率報表
  async getFunnelReport(
    startDate: string,
    endDate: string
  ): Promise<{
    stage: string;
    visitors: number;
    conversionRate: number;
    dropOff: number;
  }[]> {
    const stages: FunnelStage[] = [
      'awareness', 'interest', 'consideration',
      'intent', 'evaluation', 'purchase',
    ];

    const results: { stage: string; visitors: number; conversionRate: number; dropOff: number }[] = [];
    let previousCount = 0;

    for (const stage of stages) {
      const { count } = await this.supabase
        .from('funnel_events')
        .select('visitor_id', { count: 'exact', head: true })
        .eq('stage', stage)
        .gte('created_at', startDate)
        .lte('created_at', endDate);

      const currentCount = count || 0;
      const conversionRate = previousCount > 0
        ? (currentCount / previousCount) * 100
        : 100;
      const dropOff = previousCount > 0
        ? previousCount - currentCount
        : 0;

      results.push({
        stage,
        visitors: currentCount,
        conversionRate: Math.round(conversionRate * 100) / 100,
        dropOff,
      });

      previousCount = currentCount;
    }

    return results;
  }

  // 找出卡在某個階段的訪客（再行銷用）
  async getStuckVisitors(
    stage: FunnelStage,
    daysStuck: number = 7
  ): Promise<string[]> {
    const cutoff = new Date(
      Date.now() - daysStuck * 24 * 60 * 60 * 1000
    ).toISOString();

    const stageOrder: Record<FunnelStage, number> = {
      awareness: 0, interest: 1, consideration: 2,
      intent: 3, evaluation: 4, purchase: 5,
    };

    const nextStage = Object.entries(stageOrder).find(
      ([, order]) => order === stageOrder[stage] + 1
    )?.[0];

    if (!nextStage) return [];

    // 在此階段有事件，但沒進到下一階段
    const { data: stageVisitors } = await this.supabase
      .from('funnel_events')
      .select('visitor_id')
      .eq('stage', stage)
      .lte('created_at', cutoff);

    if (!stageVisitors?.length) return [];

    const visitorIds = [...new Set(stageVisitors.map((v) => v.visitor_id))];

    const { data: advancedVisitors } = await this.supabase
      .from('funnel_events')
      .select('visitor_id')
      .eq('stage', nextStage)
      .in('visitor_id', visitorIds);

    const advancedSet = new Set(advancedVisitors?.map((v) => v.visitor_id) || []);
    return visitorIds.filter((id) => !advancedSet.has(id));
  }
}
```

| 問題 | 原因 | 解法 |
|------|------|------|
| 進垃圾信箱 | SPF/DKIM/DMARC 未設定 | 設定 DNS TXT records |
| 開信率低 | 標題不吸引 / 發送時間差 | A/B 測試標題 + 最佳時段發送 |
| 退訂率高 | 頻率太高 / 內容不相關 | 分群推送 + 控制頻率 |
| Bounce rate 高 | 名單品質差 | 定期清理 hard bounce + 雙重確認訂閱 |
