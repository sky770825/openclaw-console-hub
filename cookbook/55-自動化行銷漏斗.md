## 1. 行銷漏斗設計（TOFU / MOFU / BOFU）

### 1.1 三層漏斗總覽

```
  ┌─────────────────────────────────────┐
  │          TOFU — 認知階段             │  目標：吸引注意力、獲取 Lead
  │   SEO / 社群 / 廣告 / 內容行銷       │  指標：流量、曝光、訂閱數
  │   轉換：訪客 → 訂閱者（Email）       │  轉換率基準：2-5%
  └───────────────┬─────────────────────┘
                  │
  ┌───────────────▼─────────────────────┐
  │          MOFU — 考慮階段             │  目標：建立信任、教育潛客
  │   Email 序列 / 案例研究 / Webinar    │  指標：開信率、點擊率、參與度
  │   轉換：訂閱者 → 合格潛客（MQL）      │  轉換率基準：15-30%
  └───────────────┬─────────────────────┘
                  │
  ┌───────────────▼─────────────────────┐
  │          BOFU — 決策階段             │  目標：促成購買 / 簽約
  │   免費試用 / Demo / 報價 / 折扣碼     │  指標：轉換率、客單價、回購率
  │   轉換：合格潛客 → 付費客戶           │  轉換率基準：5-15%
  └─────────────────────────────────────┘
```

### 1.3 漏斗階段管理器

```typescript
// funnel-manager.ts — 漏斗階段轉換邏輯

import { Lead, FunnelStage, FunnelEvent } from './funnel-types';

/** 各階段自動進階的分數門檻 */
const STAGE_THRESHOLDS: Record<FunnelStage, number> = {
  tofu: 0,
  mofu: 20,
  bofu: 50,
  customer: 80,
  advocate: 95,
};

/** 各行為對應的加分 */
const EVENT_SCORES: Record<string, number> = {
  page_view: 1,
  form_submit: 10,
  email_open: 3,
  email_click: 5,
  webinar_register: 8,
  webinar_attend: 15,
  demo_request: 25,
  trial_start: 30,
  purchase: 50,
  referral_given: 20,
};

export class FunnelManager {
  /**
   * 根據事件更新 Lead 分數與階段
   */
  static processEvent(lead: Lead, event: FunnelEvent): Lead {
    const scoreIncrement = EVENT_SCORES[event.type] ?? 0;
    const newScore = Math.min(100, lead.score + scoreIncrement);
    const newStage = this.calculateStage(newScore);

    return {
      ...lead,
      score: newScore,
      stage: newStage,
      lastActivityAt: event.timestamp,
      convertedAt: newStage === 'customer' && lead.stage !== 'customer'
        ? event.timestamp
        : lead.convertedAt,
    };
  }

  /**
   * 根據分數決定所在漏斗階段
   */
  static calculateStage(score: number): FunnelStage {
    if (score >= STAGE_THRESHOLDS.advocate) return 'advocate';
    if (score >= STAGE_THRESHOLDS.customer) return 'customer';
    if (score >= STAGE_THRESHOLDS.bofu) return 'bofu';
    if (score >= STAGE_THRESHOLDS.mofu) return 'mofu';
    return 'tofu';
  }

  /**
   * 判斷 Lead 是否停滯（超過 N 天沒有活動）
   */
  static isStagnant(lead: Lead, maxDaysInactive: number = 14): boolean {
    const daysSinceActivity = Math.floor(
      (Date.now() - new Date(lead.lastActivityAt).getTime()) / (1000 * 60 * 60 * 24)
    );
    return daysSinceActivity > maxDaysInactive;
  }

  /**
   * 批次計算漏斗各階段指標
   */
  static calculateMetrics(leads: Lead[]): FunnelMetrics[] {
    const stages: FunnelStage[] = ['tofu', 'mofu', 'bofu', 'customer', 'advocate'];

    return stages.map((stage, index) => {
      const leadsInStage = leads.filter(l => l.stage === stage);
      const nextStage = stages[index + 1];
      const leadsInNext = nextStage
        ? leads.filter(l => l.stage === nextStage)
        : [];

      const conversionRate = leadsInStage.length > 0
        ? leadsInNext.length / leadsInStage.length
        : 0;

      const avgTime = this.avgDaysInStage(leadsInStage);

      return {
        stage,
        totalLeads: leadsInStage.length,
        conversionRate: Math.round(conversionRate * 10000) / 100,
        avgTimeInStage: avgTime,
        dropOffRate: Math.round((1 - conversionRate) * 10000) / 100,
      };
    });
  }

  private static avgDaysInStage(leads: Lead[]): number {
    if (leads.length === 0) return 0;
    const totalDays = leads.reduce((sum, lead) => {
      const start = new Date(lead.firstTouchAt).getTime();
      const end = new Date(lead.lastActivityAt).getTime();
      return sum + (end - start) / (1000 * 60 * 60 * 24);
    }, 0);
    return Math.round(totalDays / leads.length);
  }
}
```

### 2.2 著陸頁 API（接收 Lead）

```typescript
// landing-page-api.ts — 著陸頁表單後端

import express, { Request, Response } from 'express';
import { createClient } from '@supabase/supabase-js';

const router = express.Router();

const supabase = createClient(
  process.env.SUPABASE_URL!,
  process.env.SUPABASE_SERVICE_KEY!
);

interface LeadFormData {
  email: string;
  name?: string;
  phone?: string;
  source: string;
  leadMagnetId: string;
  utmSource?: string;
  utmMedium?: string;
  utmCampaign?: string;
}

/**
 * POST /api/leads/capture
 * 接收著陸頁表單提交，建立 Lead 並觸發自動化流程
 */
router.post('/capture', async (req: Request, res: Response) => {
  try {
    const formData: LeadFormData = req.body;

    // 1) 驗證必填欄位
    if (!formData.email || !formData.leadMagnetId) {
      return res.status(400).json({ error: 'email 和 leadMagnetId 為必填' });
    }

    // 2) 檢查是否已存在（避免重複）
    const { data: existing } = await supabase
      .from('leads')
      .select('id, stage, score')
      .eq('email', formData.email.toLowerCase())
      .single();

    let leadId: string;

    if (existing) {
      // 已存在 → 更新活動時間，加分
      const { error } = await supabase
        .from('leads')
        .update({
          last_activity_at: new Date().toISOString(),
          score: Math.min(100, existing.score + 10),
          metadata: { lastLeadMagnet: formData.leadMagnetId },
        })
        .eq('id', existing.id);

      if (error) throw error;
      leadId = existing.id;
    } else {
      // 新 Lead → 插入
      const { data: newLead, error } = await supabase
        .from('leads')
        .insert({
          email: formData.email.toLowerCase(),
          name: formData.name,
          phone: formData.phone,
          source: formData.source || 'lead_magnet',
          stage: 'tofu',
          score: 10,
          tags: [formData.leadMagnetId],
          first_touch_at: new Date().toISOString(),
          last_activity_at: new Date().toISOString(),
          metadata: {
            leadMagnetId: formData.leadMagnetId,
            utmSource: formData.utmSource,
            utmMedium: formData.utmMedium,
            utmCampaign: formData.utmCampaign,
          },
        })
        .select('id')
        .single();

      if (error) throw error;
      leadId = newLead!.id;
    }

    // 3) 記錄漏斗事件
    await supabase.from('funnel_events').insert({
      lead_id: leadId,
      type: 'form_submit',
      stage: 'tofu',
      data: { leadMagnetId: formData.leadMagnetId, source: formData.source },
    });

    // 4) 觸發 n8n webhook（啟動 Drip Campaign）
    await triggerN8nWebhook('lead-captured', {
      leadId,
      email: formData.email,
      name: formData.name,
      leadMagnetId: formData.leadMagnetId,
    });

    res.json({ success: true, leadId, isNew: !existing });
  } catch (err) {
    console.error('[Lead Capture Error]', err);
    res.status(500).json({ error: '系統錯誤，請稍後再試' });
  }
});

/**
 * 觸發 n8n Webhook
 */
async function triggerN8nWebhook(path: string, data: Record<string, unknown>) {
  const n8nUrl = process.env.N8N_WEBHOOK_URL || 'http://localhost:5678';
  try {
    await fetch(`${n8nUrl}/webhook/${path}`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(data),
    });
  } catch (err) {
    console.error(`[n8n Webhook Error] ${path}:`, err);
  }
}

export default router;
```

## 3. 自動 Email 序列（Drip Campaign）

### 3.1 Drip Campaign 設計原則

```
  Day 0 ──→ 歡迎信 + Lead Magnet 下載連結
  Day 1 ──→ 自我介紹 + 品牌故事
  Day 3 ──→ 教育內容（解決一個痛點）
  Day 5 ──→ 案例研究 / 客戶見證
  Day 7 ──→ 進階技巧 + 免費資源
  Day 10 ─→ 常見問題解答（克服異議）
  Day 14 ─→ 限時優惠 / 行動呼籲（CTA）
```

### 3.2 Email 序列引擎

```typescript
// drip-engine.ts — Drip Campaign 自動化引擎

import { createClient, SupabaseClient } from '@supabase/supabase-js';

interface EmailStep {
  delayDays: number;
  subject: string;
  templateId: string;
  conditions?: StepCondition[];
}

interface StepCondition {
  type: 'opened_previous' | 'clicked_previous' | 'has_tag' | 'min_score';
  value: string | number | boolean;
  negate?: boolean;
}

interface SequenceConfig {
  id: string;
  name: string;
  triggerEvent: string;
  steps: EmailStep[];
}

export class DripEngine {
  private supabase: SupabaseClient;
  private emailSender: EmailSender;

  constructor() {
    this.supabase = createClient(
      process.env.SUPABASE_URL!,
      process.env.SUPABASE_SERVICE_KEY!
    );
    this.emailSender = new EmailSender();
  }

  /**
   * 啟動新 Lead 的 Drip 序列
   */
  async enrollLead(leadId: string, sequenceId: string): Promise<void> {
    // 檢查是否已在此序列中
    const { data: existing } = await this.supabase
      .from('email_sends')
      .select('id')
      .eq('lead_id', leadId)
      .eq('sequence_id', sequenceId)
      .limit(1);

    if (existing && existing.length > 0) {
      console.log(`Lead ${leadId} 已在序列 ${sequenceId} 中，跳過`);
      return;
    }

    // 取得序列設定
    const { data: sequence } = await this.supabase
      .from('email_sequences')
      .select('*')
      .eq('id', sequenceId)
      .single();

    if (!sequence || !sequence.is_active) return;

    const steps: EmailStep[] = sequence.steps;

    // 為每個步驟建立排程紀錄
    for (let i = 0; i < steps.length; i++) {
      const step = steps[i];
      const sendAt = new Date();
      sendAt.setDate(sendAt.getDate() + step.delayDays);

      await this.supabase.from('email_sends').insert({
        lead_id: leadId,
        sequence_id: sequenceId,
        step_index: i,
        subject: step.subject,
        status: 'scheduled',
        scheduled_at: sendAt.toISOString(),
      });
    }

    console.log(`Lead ${leadId} 已加入序列 ${sequenceId}，共 ${steps.length} 封`);
  }

  /**
   * 處理到期的排程 Email（每小時由 cron 觸發）
   */
  async processScheduledEmails(): Promise<number> {
    const now = new Date().toISOString();

    // 取得所有到期的排程
    const { data: pendingEmails } = await this.supabase
      .from('email_sends')
      .select(`
        id, lead_id, sequence_id, step_index, subject,
        leads!inner(email, name, tags, score, stage)
      `)
      .eq('status', 'scheduled')
      .lte('scheduled_at', now)
      .limit(50);

    if (!pendingEmails || pendingEmails.length === 0) return 0;

    let sentCount = 0;

    for (const email of pendingEmails) {
      try {
        // 取得序列設定中的條件
        const { data: sequence } = await this.supabase
          .from('email_sequences')
          .select('steps')
          .eq('id', email.sequence_id)
          .single();

        const steps: EmailStep[] = sequence?.steps || [];
        const currentStep = steps[email.step_index];

        // 檢查前置條件
        if (currentStep?.conditions) {
          const conditionsMet = await this.checkConditions(
            email.lead_id,
            email.sequence_id,
            email.step_index,
            currentStep.conditions,
            (email as any).leads
          );

          if (!conditionsMet) {
            await this.supabase
              .from('email_sends')
              .update({ status: 'skipped' })
              .eq('id', email.id);
            continue;
          }
        }

        // 發送 Email
        const lead = (email as any).leads;
        await this.emailSender.send({
          to: lead.email,
          name: lead.name,
          subject: email.subject,
          templateId: currentStep?.templateId || 'default',
          variables: {
            name: lead.name || '朋友',
            score: lead.score,
            stage: lead.stage,
          },
        });

        // 更新狀態
        await this.supabase
          .from('email_sends')
          .update({ status: 'sent', sent_at: new Date().toISOString() })
          .eq('id', email.id);

        // 記錄漏斗事件
        await this.supabase.from('funnel_events').insert({
          lead_id: email.lead_id,
          type: 'email_sent',
          stage: lead.stage,
          data: { sequenceId: email.sequence_id, stepIndex: email.step_index },
        });

        sentCount++;
      } catch (err) {
        console.error(`[Drip] 發送失敗 emailId=${email.id}:`, err);
        await this.supabase
          .from('email_sends')
          .update({ status: 'failed' })
          .eq('id', email.id);
      }
    }

    return sentCount;
  }

  /**
   * 檢查步驟條件是否滿足
   */
  private async checkConditions(
    leadId: string,
    sequenceId: string,
    stepIndex: number,
    conditions: StepCondition[],
    lead: any
  ): Promise<boolean> {
    for (const cond of conditions) {
      let result = false;

      switch (cond.type) {
        case 'opened_previous': {
          const { data } = await this.supabase
            .from('email_sends')
            .select('opened_at')
            .eq('lead_id', leadId)
            .eq('sequence_id', sequenceId)
            .eq('step_index', stepIndex - 1)
            .single();
          result = !!data?.opened_at;
          break;
        }
        case 'clicked_previous': {
          const { data } = await this.supabase
            .from('email_sends')
            .select('clicked_at')
            .eq('lead_id', leadId)
            .eq('sequence_id', sequenceId)
            .eq('step_index', stepIndex - 1)
            .single();
          result = !!data?.clicked_at;
          break;
        }
        case 'has_tag':
          result = lead.tags?.includes(cond.value as string);
          break;
        case 'min_score':
          result = lead.score >= (cond.value as number);
          break;
      }

      if (cond.negate) result = !result;
      if (!result) return false;
    }

    return true;
  }
}

/**
 * Email 發送器（支援 Resend / SendGrid / SMTP）
 */
class EmailSender {
  private provider: 'resend' | 'sendgrid' | 'smtp';

  constructor() {
    this.provider = (process.env.EMAIL_PROVIDER as any) || 'resend';
  }

  async send(options: {
    to: string;
    name?: string;
    subject: string;
    templateId: string;
    variables: Record<string, unknown>;
  }): Promise<void> {
    switch (this.provider) {
      case 'resend':
        await this.sendViaResend(options);
        break;
      case 'sendgrid':
        await this.sendViaSendGrid(options);
        break;
      case 'smtp':
        await this.sendViaSmtp(options);
        break;
```

```typescript
            firstname: contact.firstname,
            lastname: contact.lastname,
            phone: contact.phone,
            lifecyclestage: contact.lifecyclestage || 'subscriber',
            lead_score: contact.lead_score,
```

```typescript
// tagging-engine.ts — 自動標籤系統

import { createClient, SupabaseClient } from '@supabase/supabase-js';

interface TagRule {
  id: string;
  name: string;
  conditions: TagCondition[];
  tagsToAdd: string[];
  tagsToRemove?: string[];
  priority: number;
}

interface TagCondition {
  field: 'score' | 'stage' | 'source' | 'event_count' | 'days_inactive' | 'total_purchases';
  operator: 'eq' | 'gt' | 'lt' | 'gte' | 'lte' | 'contains' | 'in';
  value: string | number | string[];
}

export class TaggingEngine {
  private supabase: SupabaseClient;
  private rules: TagRule[];

  constructor() {
    this.supabase = createClient(
      process.env.SUPABASE_URL!,
      process.env.SUPABASE_SERVICE_KEY!
    );
    this.rules = DEFAULT_TAG_RULES;
  }

  /**
   * 對單一 Lead 執行所有標籤規則
   */
  async evaluateLead(leadId: string): Promise<string[]> {
    const { data: lead } = await this.supabase
      .from('leads')
      .select('*')
      .eq('id', leadId)
      .single();

    if (!lead) return [];

    // 取得 Lead 的事件統計
    const stats = await this.getLeadStats(leadId);

    const tagsToAdd: Set<string> = new Set(lead.tags || []);
    const tagsToRemove: Set<string> = new Set();

    // 依優先序評估每條規則
    const sortedRules = [...this.rules].sort((a, b) => a.priority - b.priority);

    for (const rule of sortedRules) {
      const match = this.evaluateConditions(rule.conditions, lead, stats);

      if (match) {
        rule.tagsToAdd.forEach(tag => tagsToAdd.add(tag));
        rule.tagsToRemove?.forEach(tag => tagsToRemove.delete(tag));
      }
    }

    // 移除要刪的標籤
    tagsToRemove.forEach(tag => tagsToAdd.delete(tag));

    const finalTags = Array.from(tagsToAdd);

    // 更新 Lead 標籤
    await this.supabase
      .from('leads')
      .update({ tags: finalTags })
      .eq('id', leadId);

    return finalTags;
  }

  /**
   * 批次重新計算所有 Lead 標籤（每日 cron）
   */
  async recomputeAllTags(): Promise<{ processed: number; updated: number }> {
    const { data: leads } = await this.supabase
      .from('leads')
      .select('id')
      .order('last_activity_at', { ascending: false })
      .limit(1000);

    let updated = 0;
    for (const lead of leads || []) {
      const newTags = await this.evaluateLead(lead.id);
      if (newTags.length > 0) updated++;
    }

    return { processed: leads?.length || 0, updated };
  }

  /**
   * 用標籤查詢分群
   */
  async getLeadsByTag(tag: string): Promise<any[]> {
    const { data } = await this.supabase
      .from('leads')
      .select('*')
      .contains('tags', [tag])
      .order('score', { ascending: false });

    return data || [];
  }

  /**
   * 取得 Lead 統計數據
   */
  private async getLeadStats(leadId: string) {
    const { data: events } = await this.supabase
      .from('funnel_events')
      .select('type')
      .eq('lead_id', leadId);

    const eventCounts: Record<string, number> = {};
    for (const e of events || []) {
      eventCounts[e.type] = (eventCounts[e.type] || 0) + 1;
    }

    return {
      totalEvents: events?.length || 0,
      eventCounts,
      purchaseCount: eventCounts['purchase'] || 0,
    };
  }

  /**
   * 評估條件
   */
  private evaluateConditions(
    conditions: TagCondition[],
    lead: any,
    stats: any
  ): boolean {
    return conditions.every(cond => {
      let fieldValue: any;

      switch (cond.field) {
        case 'score': fieldValue = lead.score; break;
        case 'stage': fieldValue = lead.stage; break;
        case 'source': fieldValue = lead.source; break;
        case 'event_count': fieldValue = stats.totalEvents; break;
        case 'total_purchases': fieldValue = stats.purchaseCount; break;
        case 'days_inactive': {
          const ms = Date.now() - new Date(lead.last_activity_at).getTime();
          fieldValue = Math.floor(ms / (1000 * 60 * 60 * 24));
          break;
        }
      }

      switch (cond.operator) {
        case 'eq': return fieldValue === cond.value;
        case 'gt': return fieldValue > cond.value;
        case 'lt': return fieldValue < cond.value;
        case 'gte': return fieldValue >= cond.value;
        case 'lte': return fieldValue <= cond.value;
        case 'contains': return String(fieldValue).includes(String(cond.value));
        case 'in': return (cond.value as string[]).includes(String(fieldValue));
        default: return false;
      }
    });
  }
}
```

```typescript
// cart-abandonment.ts — 購物車遺棄提醒系統

import { createClient, SupabaseClient } from '@supabase/supabase-js';

interface CartItem {
  productId: string;
  name: string;
  price: number;
  quantity: number;
  imageUrl?: string;
}

export class CartAbandonmentTracker {
  private supabase: SupabaseClient;

  constructor() {
    this.supabase = createClient(
      process.env.SUPABASE_URL!,
      process.env.SUPABASE_SERVICE_KEY!
    );
  }

  /**
   * 記錄購物車狀態（前端每次更新購物車時呼叫）
   */
  async trackCart(email: string, items: CartItem[]): Promise<void> {
    const cartTotal = items.reduce((sum, item) => sum + item.price * item.quantity, 0);

    // Upsert：同 email 只保留最新一筆未回收的
    const { data: existing } = await this.supabase
      .from('cart_abandonment')
      .select('id')
      .eq('email', email)
      .eq('recovered', false)
      .single();

    if (existing) {
      await this.supabase
        .from('cart_abandonment')
        .update({
          cart_items: items,
          cart_total: cartTotal,
          abandoned_at: new Date().toISOString(),
          reminder_count: 0,
        })
        .eq('id', existing.id);
    } else {
      await this.supabase.from('cart_abandonment').insert({
        email,
        cart_items: items,
        cart_total: cartTotal,
        lead_id: await this.findLeadId(email),
      });
    }
  }

  /**
   * 標記購物車已回收（結帳完成時呼叫）
   */
  async markRecovered(email: string): Promise<void> {
    await this.supabase
      .from('cart_abandonment')
      .update({
        recovered: true,
        recovered_at: new Date().toISOString(),
      })
      .eq('email', email)
      .eq('recovered', false);
  }

  /**
   * 取得需要發送提醒的遺棄購物車（由 cron 定期呼叫）
   */
  async getCartsForReminder(): Promise<{
    firstReminder: any[];
    secondReminder: any[];
    thirdReminder: any[];
  }> {
    const now = Date.now();
    const THIRTY_MIN = 30 * 60 * 1000;
    const TWENTY_FOUR_HR = 24 * 60 * 60 * 1000;
    const SEVENTY_TWO_HR = 72 * 60 * 60 * 1000;

    const { data: allCarts } = await this.supabase
      .from('cart_abandonment')
      .select('*')
      .eq('recovered', false)
      .lt('reminder_count', 3);

    const carts = allCarts || [];

    return {
      firstReminder: carts.filter(c => {
        const elapsed = now - new Date(c.abandoned_at).getTime();
        return c.reminder_count === 0 && elapsed >= THIRTY_MIN;
      }),
      secondReminder: carts.filter(c => {
        const elapsed = now - new Date(c.abandoned_at).getTime();
        return c.reminder_count === 1 && elapsed >= TWENTY_FOUR_HR;
      }),
      thirdReminder: carts.filter(c => {
        const elapsed = now - new Date(c.abandoned_at).getTime();
        return c.reminder_count === 2 && elapsed >= SEVENTY_TWO_HR;
      }),
    };
  }

  /**
   * 發送遺棄提醒（配合 Drip Engine 使用）
   */
  async sendReminders(): Promise<{ sent: number; recovered: number }> {
    const batches = await this.getCartsForReminder();
    let sent = 0;

    const reminderConfigs = [
      {
        carts: batches.firstReminder,
        subject: '您的購物車還在等您回來 🛒',
        templateId: 'cart_reminder',
        discountCode: undefined,
      },
      {
        carts: batches.secondReminder,
        subject: '專屬 10% 折扣 — 限時 24 小時',
        templateId: 'cart_reminder',
        discountCode: 'COMEBACK10',
      },
      {
        carts: batches.thirdReminder,
        subject: '最後機會！您的購物車商品即將售罄',
        templateId: 'cart_reminder',
        discountCode: 'LASTCHANCE15',
      },
    ];

    for (const config of reminderConfigs) {
      for (const cart of config.carts) {
        try {
          // 觸發 n8n webhook 發送 Email
          await fetch(`${process.env.N8N_WEBHOOK_URL}/webhook/cart-reminder`, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
              email: cart.email,
              subject: config.subject,
              templateId: config.templateId,
              cartItems: cart.cart_items,
              cartTotal: cart.cart_total,
              discountCode: config.discountCode,
              cartUrl: `${process.env.SITE_URL}/cart?recover=${cart.id}`,
            }),
          });

          // 更新提醒次數
          await this.supabase
            .from('cart_abandonment')
            .update({ reminder_count: cart.reminder_count + 1 })
            .eq('id', cart.id);

          sent++;
        } catch (err) {
          console.error(`[Cart Reminder] 發送失敗 ${cart.email}:`, err);
        }
      }
    }

    // 統計回收數
    const { count: recovered } = await this.supabase
      .from('cart_abandonment')
      .select('*', { count: 'exact', head: true })
      .eq('recovered', true)
      .gte('recovered_at', new Date(Date.now() - 24 * 60 * 60 * 1000).toISOString());

    return { sent, recovered: recovered || 0 };
  }

  /**
   * 購物車回收統計
   */
  async getRecoveryStats(): Promise<{
    totalAbandoned: number;
    totalRecovered: number;
    recoveryRate: number;
    revenueRecovered: number;
  }> {
    const { count: totalAbandoned } = await this.supabase

      .select('*', { count: 'exact', head: true });

    const { data: recovered } = await this.supabase
      .from('cart_abandonment')
      .select('cart_total')
      .eq('recovered', true);

    const revenueRecovered = recovered?.reduce((sum, r) => sum + Number(r.cart_total), 0) || 0;

    return {
      totalAbandoned: totalAbandoned || 0,
      totalRecovered,
      recoveryRate: totalAbandoned
        ? Math.round((totalRecovered / totalAbandoned) * 10000) / 100
        : 0,
      revenueRecovered,
    };
  }

  private async findLeadId(email: string): Promise<string | null> {
    const { data } = await this.supabase
      .from('leads')
      .select('id')
      .eq('email', email)
      .single();
    return data?.id || null;
  }
```

## 7. 客戶生命週期價值計算

### 7.1 LTV 計算模型

```typescript
// ltv-calculator.ts — 客戶生命週期價值（LTV/CLV）計算

import { createClient, SupabaseClient } from '@supabase/supabase-js';

interface CustomerMetrics {
  customerId: string;
  email: string;
  totalRevenue: number;          // 歷史總營收
  totalOrders: number;           // 總訂單數
  avgOrderValue: number;         // 平均客單價
  purchaseFrequency: number;     // 年購買頻率
  customerLifespan: number;      // 客戶壽命（月）
  historicLTV: number;           // 歷史 LTV
  predictedLTV: number;          // 預測 LTV
  acquisitionCost: number;       // 獲客成本（CAC）
  ltvCacRatio: number;           // LTV:CAC 比率
  segment: 'platinum' | 'gold' | 'silver' | 'bronze' | 'at-risk';
}

export class LTVCalculator {
  private supabase: SupabaseClient;

  constructor() {
    this.supabase = createClient(
      process.env.SUPABASE_URL!,
      process.env.SUPABASE_SERVICE_KEY!
    );
  }

  /**
   * 計算單一客戶的 LTV
   */
  async calculateCustomerLTV(customerId: string): Promise<CustomerMetrics> {
    // 取得客戶購買紀錄
    const { data: orders } = await this.supabase
      .from('funnel_events')
      .select('data, created_at')
      .eq('lead_id', customerId)
      .eq('type', 'purchase')
      .order('created_at', { ascending: true });

    const purchaseHistory = orders || [];
    const totalRevenue = purchaseHistory.reduce(
      (sum, o) => sum + (Number(o.data?.amount) || 0), 0
    );
    const totalOrders = purchaseHistory.length;
    const avgOrderValue = totalOrders > 0 ? totalRevenue / totalOrders : 0;

    // 計算購買頻率（每年幾次）
    let purchaseFrequency = 0;
    if (purchaseHistory.length >= 2) {
      const firstPurchase = new Date(purchaseHistory[0].created_at).getTime();
      const lastPurchase = new Date(purchaseHistory[purchaseHistory.length - 1].created_at).getTime();
      const daysBetween = (lastPurchase - firstPurchase) / (1000 * 60 * 60 * 24);
      purchaseFrequency = daysBetween > 0
        ? (totalOrders / daysBetween) * 365
        : totalOrders;
    } else {
      purchaseFrequency = totalOrders;
    }

    // 客戶壽命（月）
    const { data: lead } = await this.supabase
      .from('leads')
      .select('first_touch_at, last_activity_at, metadata')
      .eq('id', customerId)
      .single();

    const lifespanMs = lead
      ? new Date(lead.last_activity_at).getTime() - new Date(lead.first_touch_at).getTime()
      : 0;
    const customerLifespan = Math.max(1, lifespanMs / (1000 * 60 * 60 * 24 * 30));

    // 歷史 LTV
    const historicLTV = totalRevenue;

    // 預測 LTV（簡單模型：AOV x 年頻率 x 預期壽命年數）
    const expectedLifespanYears = 2.5; // 預設預期壽命 2.5 年
    const predictedLTV = avgOrderValue * purchaseFrequency * expectedLifespanYears;

    // 獲客成本
    const acquisitionCost = Number(lead?.metadata?.cac) || 500; // 預設 NT$500
    const ltvCacRatio = acquisitionCost > 0 ? predictedLTV / acquisitionCost : 0;

    // 分群
    const segment = this.determineSegment(predictedLTV, purchaseFrequency, customerLifespan);

    return {
      customerId,
      email: lead?.metadata?.email || '',
      totalRevenue,
      totalOrders,
      avgOrderValue: Math.round(avgOrderValue),
      purchaseFrequency: Math.round(purchaseFrequency * 10) / 10,
      customerLifespan: Math.round(customerLifespan * 10) / 10,
      historicLTV: Math.round(historicLTV),
      predictedLTV: Math.round(predictedLTV),
      acquisitionCost,
      ltvCacRatio: Math.round(ltvCacRatio * 100) / 100,
      segment,
    };
  }

  /**
   * 批次計算所有客戶 LTV 並更新分群標籤
   */
  async batchCalculateLTV(): Promise<{
    processed: number;
    avgLTV: number;
    segmentBreakdown: Record<string, number>;
  }> {
    const { data: customers } = await this.supabase
      .from('leads')
      .select('id')
      .in('stage', ['customer', 'advocate']);

    const results: CustomerMetrics[] = [];

    for (const customer of customers || []) {
      const metrics = await this.calculateCustomerLTV(customer.id);
      results.push(metrics);

      // 更新 Lead 的 metadata 和標籤
      await this.supabase
        .from('leads')
        .update({
          metadata: {
            ltv: metrics.predictedLTV,
            ltvSegment: metrics.segment,
            ltvCacRatio: metrics.ltvCacRatio,
          },
        })
        .eq('id', customer.id);
    }

    // 統計
    const avgLTV = results.length > 0
      ? results.reduce((sum, r) => sum + r.predictedLTV, 0) / results.length
      : 0;

    const segmentBreakdown: Record<string, number> = {};
    for (const r of results) {
      segmentBreakdown[r.segment] = (segmentBreakdown[r.segment] || 0) + 1;
    }

    return {
      processed: results.length,
      avgLTV: Math.round(avgLTV),
      segmentBreakdown,
    };
  }

  /**
   * 取得 LTV 排行榜（前 N 名高價值客戶）
   */
  async getTopCustomers(limit: number = 20): Promise<CustomerMetrics[]> {
    const { data: customers } = await this.supabase
      .from('leads')
      .select('id')
      .in('stage', ['customer', 'advocate'])
      .order('score', { ascending: false })
      .limit(limit);

    const results: CustomerMetrics[] = [];

    for (const customer of customers || []) {
      results.push(await this.calculateCustomerLTV(customer.id));
    }

    return results.sort((a, b) => b.predictedLTV - a.predictedLTV);
  }

  /**
   * 根據 LTV 和行為決定客戶分群
   */
  private determineSegment(
    ltv: number,
    frequency: number,
    lifespanMonths: number
  ): 'platinum' | 'gold' | 'silver' | 'bronze' | 'at-risk' {
    // 長期不活躍 → at-risk
    if (lifespanMonths > 3 && frequency < 0.5) return 'at-risk';

    // 依 LTV 分群（可依產業調整門檻）
    if (ltv >= 50000) return 'platinum';  // NT$50,000+
    if (ltv >= 20000) return 'gold';      // NT$20,000+
    if (ltv >= 5000) return 'silver';     // NT$5,000+
    return 'bronze';
  }
}
```

### 7.2 LTV 相關 KPI 公式速查

| 指標 | 公式 | 健康基準 |
|------|------|---------|
| **平均客單價 (AOV)** | 總營收 / 總訂單數 | 依產業（台灣電商平均 NT$1,200-2,500） |
| **購買頻率** | 總訂單數 / 不重複客戶數 / 年 | > 2 次/年 |
| **客戶壽命** | 1 / 流失率 | > 24 個月 |
| **LTV（歷史）** | 累計消費總額 | 持續增長 |
| **LTV（預測）** | AOV x 頻率 x 壽命 | > 3x CAC |
| **CAC** | 總行銷費用 / 新客數 | < LTV/3 |
| **LTV:CAC** | LTV / CAC | > 3:1（健康）、> 5:1（優秀） |
| **回購率** | 回購客戶 / 總客戶 | > 30%（電商）、> 60%（訂閱） |
| **淨推薦值 (NPS)** | 推薦者% - 貶低者% | > 50（優秀） |

### 9.1 儀表板 API（聚合查詢）

```typescript
// funnel-dashboard.ts — 漏斗儀表板資料聚合

import { createClient, SupabaseClient } from '@supabase/supabase-js';

export class FunnelDashboard {
  private supabase: SupabaseClient;

  constructor() {
    this.supabase = createClient(
      process.env.SUPABASE_URL!,
      process.env.SUPABASE_SERVICE_KEY!
    );
  }

  /**
   * 取得完整儀表板資料
   */
  async getDashboardData(): Promise<DashboardData> {
    const [
      funnelOverview,
      recentLeads,
      emailStats,
      cartStats,
      sourceBreakdown,
      dailyTrend,
    ] = await Promise.all([
      this.getFunnelOverview(),
      this.getRecentLeads(10),
      this.getEmailStats(),
      this.getCartStats(),
      this.getSourceBreakdown(),
      this.getDailyTrend(30),
    ]);

    return {
      funnelOverview,
      recentLeads,
      emailStats,
      cartStats,
      sourceBreakdown,
      dailyTrend,
      generatedAt: new Date().toISOString(),
    };
  }

  /**
   * 漏斗階段總覽
   */
  private async getFunnelOverview() {
    const { data: leads } = await this.supabase
      .from('leads')
      .select('stage, score');

    const stages = ['tofu', 'mofu', 'bofu', 'customer', 'advocate'];
    const overview = stages.map(stage => {
      const inStage = (leads || []).filter(l => l.stage === stage);
      return {
        stage,
        count: inStage.length,
        avgScore: inStage.length > 0
          ? Math.round(inStage.reduce((s, l) => s + l.score, 0) / inStage.length)
          : 0,
      };
    });

    return {
      totalLeads: leads?.length || 0,
      stages: overview,
    };
  }

  /**
   * 最近加入的 Lead
   */
  private async getRecentLeads(limit: number) {
    const { data } = await this.supabase
      .from('leads')
      .select('id, email, name, source, stage, score, created_at')
      .order('created_at', { ascending: false })
      .limit(limit);

    return data || [];
  }

  /**
   * Email 發送統計
   */
  private async getEmailStats() {
    const { data: sends } = await this.supabase
      .from('email_sends')
      .select('status, opened_at, clicked_at');

    const total = sends?.length || 0;
    const sent = sends?.filter(s => s.status === 'sent').length || 0;
    const opened = sends?.filter(s => s.opened_at).length || 0;
    const clicked = sends?.filter(s => s.clicked_at).length || 0;

    return {
      totalSent: sent,
      totalScheduled: sends?.filter(s => s.status === 'scheduled').length || 0,
      openRate: sent > 0 ? Math.round((opened / sent) * 10000) / 100 : 0,
      clickRate: sent > 0 ? Math.round((clicked / sent) * 10000) / 100 : 0,
      clickToOpenRate: opened > 0 ? Math.round((clicked / opened) * 10000) / 100 : 0,
    };
  }

  /**
   * 購物車遺棄統計
   */
  private async getCartStats() {
    const { data: carts } = await this.supabase
      .from('cart_abandonment')
      .select('recovered, cart_total');

    const total = carts?.length || 0;
    const recovered = carts?.filter(c => c.recovered).length || 0;
    const lostRevenue = carts
      ?.filter(c => !c.recovered)
      .reduce((sum, c) => sum + Number(c.cart_total), 0) || 0;
    const recoveredRevenue = carts
      ?.filter(c => c.recovered)
      .reduce((sum, c) => sum + Number(c.cart_total), 0) || 0;

    return {
      totalAbandoned: total,
      totalRecovered: recovered,
      recoveryRate: total > 0 ? Math.round((recovered / total) * 10000) / 100 : 0,
      lostRevenue: Math.round(lostRevenue),
      recoveredRevenue: Math.round(recoveredRevenue),
    };
  }

  /**
   * Lead 來源分布
   */
  private async getSourceBreakdown() {
    const { data: leads } = await this.supabase
      .from('leads')
      .select('source');

    const breakdown: Record<string, number> = {};
    for (const lead of leads || []) {
      breakdown[lead.source] = (breakdown[lead.source] || 0) + 1;
    }

    return Object.entries(breakdown)
      .map(([source, count]) => ({ source, count }))
      .sort((a, b) => b.count - a.count);
  }

  /**
   * 每日新增 Lead 趨勢（過去 N 天）
   */
  private async getDailyTrend(days: number) {
    const startDate = new Date();
    startDate.setDate(startDate.getDate() - days);

    const { data: leads } = await this.supabase
      .from('leads')
      .select('created_at, stage')
      .gte('created_at', startDate.toISOString());

    // 按日期分組
    const dailyMap: Record<string, { total: number; converted: number }> = {};

    for (const lead of leads || []) {
      const day = new Date(lead.created_at).toISOString().split('T')[0];
      if (!dailyMap[day]) dailyMap[day] = { total: 0, converted: 0 };
      dailyMap[day].total++;
      if (lead.stage === 'customer' || lead.stage === 'advocate') {
        dailyMap[day].converted++;
      }
    }

    return Object.entries(dailyMap)
      .map(([date, data]) => ({ date, ...data }))
      .sort((a, b) => a.date.localeCompare(b.date));
  }
}

interface DashboardData {
  funnelOverview: any;
  recentLeads: any[];
  emailStats: any;
  cartStats: any;
  sourceBreakdown: any[];
  dailyTrend: any[];
  generatedAt: string;
}
```

## 10. 常見問題與排查

### 10.1 Email 送達率低

| 問題 | 可能原因 | 解決方案 |
|------|---------|---------|
| 進垃圾信箱 | 未設定 SPF/DKIM/DMARC | 參考 cookbook/36 設定 DNS 驗證 |
| 開信率 < 15% | 主旨不吸引 / 發送時間差 | A/B 測試主旨，調整為當地上班時間 |
| 退訂率 > 2% | 發送頻率太高 / 內容不相關 | 降低頻率，改善分群精準度 |
| 軟退信 | 信箱已滿 | 3 次軟退信後標記為 inactive |
| 硬退信 | Email 地址無效 | 立即標記 do-not-contact，清理名單 |

### 10.2 漏斗轉換率低

| 階段 | 正常基準 | 偏低時排查 |
|------|---------|-----------|
| TOFU → MOFU | 15-30% | 檢查 Lead Magnet 品質、著陸頁轉換率 |
| MOFU → BOFU | 10-25% | 檢查 Email 序列設計、內容相關性 |
| BOFU → Customer | 5-15% | 檢查報價頁、CTA、是否缺臨門一腳 |
| 整體轉換率 | 1-5% | 全漏斗檢視，找出最大瓶頸階段 |

### 10.3 購物車回收率低


```typescript
// crm-sync-healthcheck.ts — CRM 同步健康檢查

export async function checkCRMSync(): Promise<{
  hubspot: { status: string; error?: string };
  notion: { status: string; error?: string };
}> {
  const results = { hubspot: { status: 'unknown' }, notion: { status: 'unknown' } } as any;

    results.notion = { status: 'error', error: String(err) };
```
