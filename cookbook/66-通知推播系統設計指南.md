## 1. 通知管道總覽

| 管道 | 到達率 | 成本 | 延遲 | 適用場景 |
|------|--------|------|------|----------|
| Email | 高（95%+） | 低 | 秒～分 | 行銷電子報、交易通知、報表 |
| SMS | 極高（99%） | 高 | 秒 | OTP 驗證、緊急通知 |
| LINE | 高（台灣） | 中 | 秒 | 台灣客戶推播、互動 |
| Telegram | 高 | 免費 | 即時 | 團隊通知、系統告警 |
| Web Push | 中（60-80%） | 免費 | 秒 | 網站訪客回流、即時提醒 |
| App Push | 高 | 免費 | 秒 | App 用戶推播 |

### 多管道策略

```
優先級：事件發生 →
  1. 即時管道（Telegram / Web Push）— 立即觸達
  2. 主要管道（LINE / Email）— 確保送達
  3. 備用管道（SMS）— 重要通知的最後防線
```

### 資料模型

```sql
-- 訊息模板
CREATE TABLE notification_templates (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name VARCHAR(100) NOT NULL,
  channel VARCHAR(20) NOT NULL,  -- email / sms / line / telegram / web_push
  subject VARCHAR(200),          -- Email 主旨
  body TEXT NOT NULL,             -- 內容（支援變數 {{name}}）
  body_html TEXT,                 -- HTML 版本（Email 用）
  variables JSONB DEFAULT '[]',  -- 可用變數列表
  is_active BOOLEAN DEFAULT TRUE,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- 受眾群組
CREATE TABLE audiences (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name VARCHAR(100) NOT NULL,
  rules JSONB NOT NULL,           -- 篩選規則
  exclude_rules JSONB DEFAULT '{}',
  estimated_count INT DEFAULT 0,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 推播任務
CREATE TABLE campaigns (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name VARCHAR(200) NOT NULL,
  template_id UUID REFERENCES notification_templates(id),
  audience_id UUID REFERENCES audiences(id),
  channel VARCHAR(20) NOT NULL,
  status VARCHAR(20) DEFAULT 'draft',  -- draft / scheduled / sending / sent / cancelled
  scheduled_at TIMESTAMPTZ,
  sent_at TIMESTAMPTZ,
  variables_data JSONB DEFAULT '{}',
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 發送紀錄
CREATE TABLE notification_logs (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  campaign_id UUID REFERENCES campaigns(id),
  recipient_id UUID,
  recipient_address VARCHAR(200),  -- email / phone / line_uid / chat_id
  channel VARCHAR(20) NOT NULL,
  status VARCHAR(20) DEFAULT 'queued',  -- queued / sent / delivered / opened / clicked / failed / bounced
  sent_at TIMESTAMPTZ,
  delivered_at TIMESTAMPTZ,
  opened_at TIMESTAMPTZ,
  clicked_at TIMESTAMPTZ,
  error_message TEXT,
  metadata JSONB DEFAULT '{}'
);

-- 退訂名單
CREATE TABLE unsubscribes (
  id SERIAL PRIMARY KEY,
  recipient_address VARCHAR(200) NOT NULL,
  channel VARCHAR(20) NOT NULL,
  unsubscribed_at TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(recipient_address, channel)
);
```

```typescript
interface TemplateVariable {
  key: string;       // 變數名：name, company, amount
  label: string;     // 顯示名：客戶姓名
  fallback?: string; // 預設值：valued customer

function renderTemplate(template: string, variables: Record<string, string>): string {
  return template.replace(/\{\{(\w+)(?:\|([^}]+))?\}\}/g, (match, key, fallback) => {
    return variables[key] || fallback || match;
  });
}

// 使用範例
```

## 4. 受眾管理與分群

### 篩選規則結構

```typescript
interface AudienceRule {
  field: string;       // customer.tags / customer.source / deal.value
  operator: 'eq' | 'neq' | 'gt' | 'lt' | 'contains' | 'in' | 'between';
  value: any;
}

// 範例：VIP 客戶 + 有成交 + 排除已退訂
const vipAudience = {
  name: 'VIP 客戶推播',
  rules: [
    { field: 'tags', operator: 'contains', value: 'VIP' },
    { field: 'deal_count', operator: 'gt', value: 0 },
    { field: 'status', operator: 'eq', value: 'active' },
  ],
  exclude_rules: {
    unsubscribed: true,  // 自動排除退訂者
  },
};
```

### 動態受眾查詢

```typescript
function buildAudienceQuery(rules: AudienceRule[]): { sql: string; params: any[] } {
  const conditions: string[] = [];
  const params: any[] = [];
  let paramIdx = 1;

  for (const rule of rules) {
    switch (rule.operator) {
      case 'eq':
        conditions.push(`${rule.field} = $${paramIdx++}`);
        params.push(rule.value);
        break;
      case 'gt':
        conditions.push(`${rule.field} > $${paramIdx++}`);
        params.push(rule.value);
        break;
      case 'contains':
        conditions.push(`$${paramIdx++} = ANY(${rule.field})`);
        params.push(rule.value);
        break;
      case 'in':
        conditions.push(`${rule.field} = ANY($${paramIdx++})`);
        params.push(rule.value);
        break;
      case 'between':
        conditions.push(`${rule.field} BETWEEN $${paramIdx++} AND $${paramIdx++}`);
        params.push(rule.value[0], rule.value[1]);
        break;
    }
  }

  // 排除退訂者
  const sql = `
    SELECT c.id, c.name, c.email, c.phone
    FROM customers c
    WHERE ${conditions.join(' AND ')}
      AND c.email NOT IN (SELECT recipient_address FROM unsubscribes WHERE channel = 'email')
  `;

  return { sql, params };
}
```

### 預估人數

```typescript
// GET /api/audiences/:id/estimate
app.get('/api/audiences/:id/estimate', async (req, res) => {
  const audience = await db.query('SELECT * FROM audiences WHERE id = $1', [req.params.id]);
  const { sql, params } = buildAudienceQuery(audience.rows[0].rules);
  const countSql = sql.replace('SELECT c.id, c.name, c.email, c.phone', 'SELECT COUNT(*) AS total');
  const result = await db.query(countSql, params);
  res.json({ estimated_count: parseInt(result.rows[0].total) });
});
```

## 5. 排程與觸發發送

### 發送模式

| 模式 | 說明 | 實作 |
|------|------|------|
| 立即發送 | 建立後馬上發 | 直接進 queue |
| 排程發送 | 指定時間發 | cron / BullMQ delayed job |
| 事件觸發 | 特定動作後自動發 | Webhook / n8n trigger |

### 排程引擎（BullMQ）

```typescript
import { Queue, Worker } from 'bullmq';

const notificationQueue = new Queue('notifications', {
  connection: { host: 'localhost', port: 6379 }
});

// 排程發送
async function scheduleCampaign(campaignId: string, scheduledAt: Date) {
  const delay = scheduledAt.getTime() - Date.now();
  await notificationQueue.add('send-campaign', { campaignId }, {
    delay: Math.max(delay, 0),
    attempts: 3,
    backoff: { type: 'exponential', delay: 60000 },
  });
}

// Worker 處理發送
const worker = new Worker('notifications', async (job) => {
  const { campaignId } = job.data;
  const campaign = await getCampaign(campaignId);
  const template = await getTemplate(campaign.template_id);
  const recipients = await getAudienceMembers(campaign.audience_id);

  // 批次發送（每批 100 人，間隔 1 秒避免 rate limit）
  for (let i = 0; i < recipients.length; i += 100) {
    const batch = recipients.slice(i, i + 100);
    await Promise.all(batch.map(r => sendNotification(campaign.channel, template, r)));
    if (i + 100 < recipients.length) await sleep(1000);
  }

  await db.query("UPDATE campaigns SET status = 'sent', sent_at = NOW() WHERE id = $1", [campaignId]);
}, { connection: { host: 'localhost', port: 6379 } });
```

### 時區處理

```typescript
// 依收件人時區，在「當地時間 10:00」發送
function getLocalSendTime(targetLocalHour: number, recipientTimezone: string): Date {
  const now = new Date();
  // 用 Intl API 取得收件人當地時間
  const localNow = new Date(now.toLocaleString('en-US', { timeZone: recipientTimezone }));
  const sendTime = new Date(localNow);
  sendTime.setHours(targetLocalHour, 0, 0, 0);

  // 如果今天已過，排到明天
  if (sendTime <= localNow) sendTime.setDate(sendTime.getDate() + 1);

  // 轉回 UTC
  const offset = localNow.getTime() - now.getTime();
  return new Date(sendTime.getTime() - offset);
}
```

### 成效報表 SQL

```sql
-- 單一 campaign 成效
SELECT
  COUNT(*) AS total_sent,
  COUNT(*) FILTER (WHERE status IN ('delivered','opened','clicked')) AS delivered,
  COUNT(*) FILTER (WHERE status IN ('opened','clicked')) AS opened,
  COUNT(*) FILTER (WHERE status = 'clicked') AS clicked,
  COUNT(*) FILTER (WHERE status = 'failed') AS failed,
  COUNT(*) FILTER (WHERE status = 'bounced') AS bounced,
  ROUND(COUNT(*) FILTER (WHERE status IN ('delivered','opened','clicked'))::numeric / NULLIF(COUNT(*), 0) * 100, 1) AS delivery_rate,
  ROUND(COUNT(*) FILTER (WHERE status IN ('opened','clicked'))::numeric / NULLIF(COUNT(*) FILTER (WHERE status IN ('delivered','opened','clicked')), 0) * 100, 1) AS open_rate,
  ROUND(COUNT(*) FILTER (WHERE status = 'clicked')::numeric / NULLIF(COUNT(*) FILTER (WHERE status IN ('opened','clicked')), 0) * 100, 1) AS click_rate
FROM notification_logs
WHERE campaign_id = $1;
```

## 7. n8n / Webhook 整合

### 事件觸發推播（n8n workflow）

```
Webhook Trigger（POST /webhook/new-order）
  │
  ├→ 查詢客戶資料（HTTP Request → /api/customers/:id）
  │
  ├→ 渲染模板（Code Node）
  │     template.replace('{{name}}', customer.name)
  │
  ├→ 發送 LINE 推播（HTTP Request → LINE API）
  │
  └→ 記錄發送紀錄（HTTP Request → POST /api/notification-logs）
```

### Webhook 觸發範例

```typescript
// 訂單成立時自動推播
app.post('/api/orders', async (req, res) => {

async function triggerNotification(event: string, data: Record<string, any>) {
  // 查找綁定此事件的模板

    "SELECT * FROM notification_templates WHERE trigger_event = $1 AND is_active = TRUE",

  );

    await sendNotification(tpl.channel, rendered, data.recipient);
```

```html
<script>
let rules = [];

function addRule() {
  const ruleId = Date.now();
  rules.push({ id: ruleId, field: 'tags', operator: 'contains', value: '' });
  renderRules();
}

function renderRules() {
  document.getElementById('ruleList').innerHTML = rules.map((r, i) => `
    <div class="rule-row" data-id="${r.id}">
      ${i > 0 ? '<span class="rule-connector">AND</span>' : ''}
      <select onchange="updateRule(${r.id}, 'field', this.value)">
        <option value="tags" ${r.field === 'tags' ? 'selected' : ''}>標籤</option>
        <option value="source" ${r.field === 'source' ? 'selected' : ''}>來源</option>
        <option value="status" ${r.field === 'status' ? 'selected' : ''}>狀態</option>
        <option value="deal_count" ${r.field === 'deal_count' ? 'selected' : ''}>交易數</option>
        <option value="total_revenue" ${r.field === 'total_revenue' ? 'selected' : ''}>總營收</option>
      </select>
      <select onchange="updateRule(${r.id}, 'operator', this.value)">
        <option value="eq">等於</option>
        <option value="neq">不等於</option>
        <option value="gt">大於</option>
        <option value="lt">小於</option>
        <option value="contains">包含</option>
      </select>
      <input type="text" value="${r.value}" onchange="updateRule(${r.id}, 'value', this.value)" />
      <button onclick="removeRule(${r.id})" class="btn-danger-sm">x</button>
    </div>
  `).join('');
}

function updateRule(id, key, value) {
  const rule = rules.find(r => r.id === id);
  if (rule) rule[key] = value;
}

function removeRule(id) {
  rules = rules.filter(r => r.id !== id);
  renderRules();
}

    body: JSON.stringify({ rules }),
  });
  const data = await res.json();
  document.getElementById('estimatedCount').textContent = data.estimated_count.toLocaleString() + ' 人';
}
```
