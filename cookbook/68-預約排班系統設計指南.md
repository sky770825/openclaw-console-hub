## 2. 預約流程設計

### 2.1 標準預約流程

```
步驟 1：選擇服務項目（剪髮 / 染髮 / 護髮...）
   ↓
步驟 2：選擇服務人員（可選「不指定」）
   ↓
步驟 3：選擇日期（月曆元件）
   ↓
步驟 4：選擇時段（只顯示可選時段）
   ↓
步驟 5：填寫個人資料 + 備註
   ↓
步驟 6：確認預約內容 → 送出
   ↓
步驟 7：收到確認訊息（LINE / SMS / Email）
```

### 2.2 關鍵 UX 原則

| 原則 | 說明 |
|------|------|
| 即時回饋 | 選完服務 + 人員後，日曆即時更新可選日期 |
| 最少步驟 | 如果只有一種服務，跳過步驟 1 |
| 不可選灰色 | 已滿時段灰色、不可點擊，避免挫敗感 |
| 回退自由 | 任何步驟都可以回上一步修改 |
| 確認頁 | 送出前必須有完整摘要讓用戶確認 |

### 3.3 月曆 JS 邏輯

```javascript
let currentYear = 2026;
let currentMonth = 2; // 0-indexed (2 = March)
let selectedDate = null;

// 從 API 取得該月可用日期
async function fetchAvailability(year, month) {
  const res = await fetch(
    `/api/availability?year=${year}&month=${month + 1}&staffId=${selectedStaff}`
  );
  return await res.json();
  // 回傳格式：{ "2026-03-07": { available: 5, total: 10 }, ... }
}

async function renderCalendar() {
  const grid = document.getElementById('calendarGrid');
  grid.innerHTML = '';

  const availability = await fetchAvailability(currentYear, currentMonth);
  const firstDay = new Date(currentYear, currentMonth, 1).getDay();
  const daysInMonth = new Date(currentYear, currentMonth + 1, 0).getDate();
  const today = new Date();

  // 填充前面空白
  for (let i = 0; i < firstDay; i++) {
    grid.innerHTML += '<div class="calendar-day empty"></div>';
  }

  for (let day = 1; day <= daysInMonth; day++) {
    const dateStr = `${currentYear}-${String(currentMonth + 1).padStart(2, '0')}-${String(day).padStart(2, '0')}`;
    const date = new Date(currentYear, currentMonth, day);
    const isPast = date < new Date(today.getFullYear(), today.getMonth(), today.getDate());
    const dayData = availability[dateStr];

    let className = 'calendar-day';
    if (isPast) {
      className += ' past';
    } else if (!dayData || dayData.available === 0) {
      className += ' full';
    } else {
      className += ' available';
    }

    if (dateStr === selectedDate) className += ' selected';
    if (date.toDateString() === today.toDateString()) className += ' today';

    const el = document.createElement('div');
    el.className = className;
    el.textContent = day;

    if (!isPast && dayData && dayData.available > 0) {
      el.onclick = () => selectDate(dateStr);
    }

    grid.appendChild(el);
  }

  document.getElementById('monthLabel').textContent =
    `${currentYear} 年 ${currentMonth + 1} 月`;
}

function changeMonth(delta) {
  currentMonth += delta;
  if (currentMonth > 11) { currentMonth = 0; currentYear++; }
  if (currentMonth < 0) { currentMonth = 11; currentYear--; }
  renderCalendar();
}

function selectDate(dateStr) {
  selectedDate = dateStr;
  renderCalendar();
  loadTimeSlots(dateStr); // 載入該日時段
}
```

### 4.2 時段產生邏輯

```javascript
/**
 * 產生某天可用時段
 * @param date - 日期字串 '2026-03-07'
 * @param serviceDuration - 服務時長（分鐘）
 * @param slotInterval - 時段間隔（分鐘）
 */
function generateTimeSlots(date, serviceDuration = 60, slotInterval = 30) {
  const dayOfWeek = ['sun','mon','tue','wed','thu','fri','sat'][new Date(date).getDay()];
  const hours = businessHours[dayOfWeek];
  if (!hours) return []; // 休息日

  const slots = [];
  let current = parseTime(hours.open);
  const close = parseTime(hours.close);
  const breakStart = hours.breakStart ? parseTime(hours.breakStart) : null;
  const breakEnd = hours.breakEnd ? parseTime(hours.breakEnd) : null;

  while (current + serviceDuration <= close) {
    // 跳過午休
    if (breakStart && current >= breakStart && current < breakEnd) {
      current = breakEnd;
      continue;
    }
    // 服務時間不能跨午休
    if (breakStart && current < breakStart && current + serviceDuration > breakStart) {
      current += slotInterval;
      continue;
    }

    slots.push({
      time: formatTime(current),
      endTime: formatTime(current + serviceDuration),
      capacity: 1  // 每時段可服務人數（可依人員數量調整）
    });

    current += slotInterval;
  }
  return slots;
}

function parseTime(str) {
  const [h, m] = str.split(':').map(Number);
  return h * 60 + m;
}

function formatTime(minutes) {
  const h = String(Math.floor(minutes / 60)).padStart(2, '0');
  const m = String(minutes % 60).padStart(2, '0');
  return `${h}:${m}`;
}
```

```sql
-- 查詢某天有上班的員工
SELECT s.id, s.name FROM staff s
JOIN staff_schedules ss ON s.id = ss.staff_id
WHERE ss.day_of_week = EXTRACT(DOW FROM DATE '2026-03-10')
  AND ss.is_active = true
  AND s.id NOT IN (
    SELECT staff_id FROM leave_requests
    WHERE status = 'approved'
    AND '2026-03-10' BETWEEN start_date AND end_date
```

## 7. 提醒機制

### 7.1 提醒時間點

| 時機 | 方式 | 內容 |
|------|------|------|
| 預約成功 | LINE + Email | 預約確認 + 詳細資訊 |

### 7.2 排程提醒實作（node-cron）

```javascript
const cron = require('node-cron');

// 每小時跑一次，找出「1 天後」的預約
cron.schedule('0 * * * *', async () => {
  const tomorrow = new Date();
  tomorrow.setDate(tomorrow.getDate() + 1);
  const dateStr = tomorrow.toISOString().split('T')[0];

  const bookings = await db.query(`
    SELECT b.*, c.line_user_id, c.phone, c.name as customer_name,
           s.name as service_name, st.name as staff_name
    FROM bookings b
    JOIN customers c ON b.customer_id = c.id
    JOIN services s ON b.service_id = s.id
    LEFT JOIN staff st ON b.staff_id = st.id
    WHERE b.date = $1
    AND b.status = 'confirmed'
    AND b.reminder_1day_sent = false
  `, [dateStr]);

  for (const booking of bookings) {
    // LINE 推播提醒
    if (booking.line_user_id) {
      await linePush(booking.line_user_id, {
        type: 'flex', altText: '預約提醒',
        contents: buildReminderFlex(booking)
      });
    }

    // 標記已提醒
    await db.query(
      'UPDATE bookings SET reminder_1day_sent = true WHERE id = $1',
      [booking.id]
    );
  }
});

function buildReminderFlex(booking) {
  return {
    type: 'bubble',
    body: {
      type: 'box', layout: 'vertical',
      contents: [
        { type: 'text', text: '預約提醒', weight: 'bold', size: 'lg', color: '#2563eb' },
        { type: 'separator', margin: 'md' },
        { type: 'text', text: `服務：${booking.service_name}`, margin: 'md', size: 'sm' },
        { type: 'text', text: `日期：${booking.date}`, size: 'sm' },
        { type: 'text', text: `時間：${booking.time_slot}`, size: 'sm' },
        { type: 'text', text: `人員：${booking.staff_name || '不指定'}`, size: 'sm' }
      ]
    },
    footer: {
      type: 'box', layout: 'horizontal', spacing: 'sm',
      contents: [
        { type: 'button', style: 'primary', action: { type: 'message', label: '確認出席', text: '確認出席' } },
        { type: 'button', style: 'secondary', action: { type: 'message', label: '取消預約', text: '取消預約' } }
      ]
    }
  };
}
```

## 8. 防重複預約與併發處理

### 8.1 問題情境

```
用戶 A 看到 14:00 有空位（剩 1 位）
用戶 B 也看到 14:00 有空位（剩 1 位）
兩人同時按下「確認預約」→ 超賣！
```

### 8.2 解法一：PostgreSQL Advisory Lock（推薦）

```javascript
async function createBooking(bookingData) {
  const { date, timeSlot, staffId, serviceId, customerId } = bookingData;

  // 用 date + timeSlot + staffId 產生唯一 lock key
  const lockKey = hashToInt(`${date}-${timeSlot}-${staffId}`);

  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    // 取得 Advisory Lock（等待模式，排隊）
    await client.query('SELECT pg_advisory_xact_lock($1)', [lockKey]);

    // 檢查該時段剩餘名額
    const { rows } = await client.query(`
      SELECT COUNT(*) as booked FROM bookings
      WHERE date = $1 AND time_slot = $2 AND staff_id = $3
      AND status IN ('confirmed', 'pending')
    `, [date, timeSlot, staffId]);

    const capacity = await getSlotCapacity(staffId, date, timeSlot);

    if (parseInt(rows[0].booked) >= capacity) {
      await client.query('ROLLBACK');
      return { success: false, error: '該時段已滿，請選擇其他時段。' };
    }

    // 建立預約
    const result = await client.query(`
      INSERT INTO bookings (date, time_slot, staff_id, service_id, customer_id, status)
      VALUES ($1, $2, $3, $4, $5, 'confirmed')
      RETURNING id
    `, [date, timeSlot, staffId, serviceId, customerId]);

    await client.query('COMMIT');
    return { success: true, bookingId: result.rows[0].id };
  } catch (err) {
    await client.query('ROLLBACK');
    throw err;
  } finally {
    client.release();
  }
}

function hashToInt(str) {
  let hash = 0;
  for (let i = 0; i < str.length; i++) {
    hash = ((hash << 5) - hash) + str.charCodeAt(i);
    hash |= 0;
  }
  return Math.abs(hash);
}
```

```sql
CREATE TABLE services (
  id SERIAL PRIMARY KEY,
  name VARCHAR(100) NOT NULL,

  price DECIMAL(10,2),
  description TEXT,
  is_active BOOLEAN DEFAULT true,
  sort_order INT DEFAULT 0
);

-- 員工
CREATE TABLE staff (
  id SERIAL PRIMARY KEY,
  name VARCHAR(50) NOT NULL,
  title VARCHAR(50),            -- 頭銜（設計師/助理）
  avatar_url TEXT,
  is_active BOOLEAN DEFAULT true
);

-- 員工可提供的服務
CREATE TABLE staff_services (
  staff_id INT REFERENCES staff(id),
  service_id INT REFERENCES services(id),
  PRIMARY KEY (staff_id, service_id)
);

-- 客戶
CREATE TABLE customers (
  id SERIAL PRIMARY KEY,
  name VARCHAR(50) NOT NULL,
  phone VARCHAR(20),
  email VARCHAR(100),
  line_user_id VARCHAR(50),
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 預約紀錄
CREATE TABLE bookings (
  id SERIAL PRIMARY KEY,
  customer_id INT NOT NULL REFERENCES customers(id),
  service_id INT NOT NULL REFERENCES services(id),
  staff_id INT REFERENCES staff(id),
  date DATE NOT NULL,
  time_slot TIME NOT NULL,
  end_time TIME NOT NULL,
  status VARCHAR(20) DEFAULT 'confirmed',  -- confirmed/cancelled/completed/no_show
  note TEXT,
  reminder_1day_sent BOOLEAN DEFAULT false,
  reminder_1hour_sent BOOLEAN DEFAULT false,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- 索引（查詢效能關鍵）
CREATE INDEX idx_bookings_date_staff ON bookings(date, staff_id);
```

### 9.2 可用時段 API

```javascript
// GET /api/timeslots?date=2026-03-10&serviceId=1&staffId=1
app.get('/api/timeslots', async (req, res) => {
  const { date, serviceId, staffId } = req.query;

  // 1. 取得服務時長
  const service = await db.queryOne('SELECT duration FROM services WHERE id = $1', [serviceId]);

  // 2. 產生該天所有時段
  const allSlots = generateTimeSlots(date, service.duration);

  // 3. 查詢已預約數量
  const booked = await db.query(`
    SELECT time_slot, COUNT(*) as count FROM bookings
    WHERE date = $1 AND staff_id = $2 AND status IN ('confirmed','pending')
    GROUP BY time_slot
  `, [date, staffId]);

  const bookedMap = {};
  booked.forEach(b => { bookedMap[b.time_slot] = parseInt(b.count); });

  // 4. 計算剩餘
  const capacity = 1; // 每個員工同時段服務 1 人
  const result = allSlots.map(slot => ({
    time: slot.time,
    endTime: slot.endTime,
    remaining: capacity - (bookedMap[slot.time] || 0)
  }));

  res.json(result);
});
```
