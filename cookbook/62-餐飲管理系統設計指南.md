## 2. 桌位管理

### 2.1 桌位狀態機

```
available（空桌）
    |
    +-- 入座 --> occupied（用餐中）
    |                |
    |                +-- 結帳 --> cleaning（待清潔）
    |                |                |
    |                |                +-- 清潔完成 --> available
    |                |
    |                +-- 合併桌位 --> occupied（更新 merged_with）
    |                +-- 分桌 --> 兩張 occupied
    |
    +-- 預約 --> reserved（已預約）
                     |
                     +-- 到場 --> occupied
                     +-- 未到/取消 --> available
```

### 2.2 桌位操作 API

```javascript
// 入座
router.post('/api/tables/:id/seat', async (req, res) => {
  const { id } = req.params;
  const { guestCount, serverName } = req.body;

  // 建立新訂單
  const orderNumber = await generateDailyOrderNumber();
  const order = await db.query(`
    INSERT INTO restaurant_orders (order_number, table_id, guest_count, server_name)
    VALUES ($1, $2, $3, $4) RETURNING *
  `, [orderNumber, id, guestCount, serverName]);

  // 更新桌位狀態
  await db.query(`
    UPDATE tables SET status = 'occupied', current_order_id = $1, updated_at = now()
    WHERE id = $2
  `, [order.rows[0].id, id]);

  res.json({ table: id, order: order.rows[0] });
});

// 合併桌位
router.post('/api/tables/merge', async (req, res) => {
  const { tableIds } = req.body; // ['A1', 'A2']
  const mainTable = tableIds[0];
  const mergedTables = tableIds.slice(1);

  await db.query(`
    UPDATE tables SET merged_with = $1 WHERE number = $2
  `, [mergedTables, mainTable]);

  // 其他桌標記為合併中
  for (const t of mergedTables) {
    await db.query(`
      UPDATE tables SET status = 'occupied', merged_with = ARRAY[$1] WHERE number = $2
    `, [mainTable, t]);
  }

  res.json({ mainTable, merged: mergedTables });
});

// 結帳 + 清桌
router.post('/api/tables/:id/checkout', async (req, res) => {
  const { id } = req.params;
  const { paymentMethod } = req.body;

  const table = await db.query(`SELECT * FROM tables WHERE id = $1`, [id]);
  const orderId = table.rows[0].current_order_id;

  // 結帳
  await db.query(`
    UPDATE restaurant_orders
    SET status = 'completed', payment_method = $1, payment_status = 'paid', closed_at = now()
    WHERE id = $2
  `, [paymentMethod, orderId]);

  // 桌位改為待清潔
  await db.query(`
    UPDATE tables SET status = 'cleaning', current_order_id = NULL WHERE id = $1
  `, [id]);

  // 解除合併桌位
  if (table.rows[0].merged_with?.length) {
    for (const t of table.rows[0].merged_with) {
      await db.query(`
        UPDATE tables SET status = 'cleaning', merged_with = NULL WHERE number = $1
      `, [t]);
    }
  }

  // 庫存扣減
  await deductIngredientStock(orderId);

  res.json({ success: true });
});
```

### 2.3 翻桌率計算

```javascript
async function getTableTurnoverRate(date) {
  const result = await db.query(`
    SELECT
      t.number,
      t.capacity,
      COUNT(o.id) AS order_count,
      AVG(EXTRACT(EPOCH FROM (o.closed_at - o.opened_at)) / 60) AS avg_dining_min
    FROM tables t
    LEFT JOIN restaurant_orders o ON o.table_id = t.id
      AND o.status = 'completed'
      AND DATE(o.opened_at) = $1
    GROUP BY t.id, t.number, t.capacity
    ORDER BY t.number
  `, [date]);

  return result.rows.map(r => ({
    table: r.number,
    capacity: r.capacity,
    orders: r.order_count,
    turnoverRate: r.order_count,  // 翻桌率 = 當日完成訂單數
    avgDiningMin: Math.round(r.avg_dining_min || 0),
  }));
}
```

```javascript
async function checkDishAvailability(dishId) {
  const dish = await db.query(`SELECT ingredients FROM dishes WHERE id = $1`, [dishId]);
  const ingredients = dish.rows[0]?.ingredients || [];

  for (const ing of ingredients) {
    const stock = await db.query(
      `SELECT current_stock FROM ingredients WHERE id = $1`, [ing.ingredient_id]
    );
    if (!stock.rows[0] || stock.rows[0].current_stock < ing.amount) {
      // 食材不足，自動標記售完
      await db.query(`UPDATE dishes SET is_available = false WHERE id = $1`, [dishId]);
      return { available: false, reason: `${ing.name} 庫存不足` };
    }
  }
  return { available: true };
}
```

## 4. 訂單看板

### 4.1 看板分類

| 看板 | 顯示內容 | 使用者 |
|------|---------|--------|
| 外場看板 | 桌號 + 品項 + 上菜進度 | 服務生 |
| 廚房看板 | 待製作 → 製作中 → 完成 | 廚師 |
| 外帶看板 | 外帶/外送訂單 + 取餐號 | 櫃台 |
| 總覽看板 | 全部訂單 + 即時統計 | 店長 |

### 4.2 訂單品項狀態

```
pending（待製作）→ cooking（製作中）→ ready（可出餐）→ served（已上菜）
```

### 4.3 加點 / 退點

```javascript
// 加點
router.post('/api/orders/:id/add-items', async (req, res) => {
  const { id } = req.params;
  const { items } = req.body; // [{dish_id, qty, note}]

  const order = await db.query(`SELECT * FROM restaurant_orders WHERE id = $1`, [id]);
  const existingItems = order.rows[0].items || [];

  const newItems = items.map(item => ({
    ...item,
    status: 'pending',
    added_at: new Date().toISOString(),
  }));

  const allItems = [...existingItems, ...newItems];
  const subtotal = allItems.reduce((sum, i) => sum + (i.price * i.qty), 0);

  await db.query(`
    UPDATE restaurant_orders
    SET items = $1, subtotal = $2, total = $2 - discount + service_charge
    WHERE id = $3
  `, [JSON.stringify(allItems), subtotal, id]);

  // 推送到廚房看板
  broadcastToKitchen({ type: 'new_items', orderId: id, items: newItems });

  res.json({ success: true });
});

// 退點（未製作才能退）
router.post('/api/orders/:id/remove-item', async (req, res) => {
  const { id } = req.params;
  const { itemIndex, reason } = req.body;

  const order = await db.query(`SELECT * FROM restaurant_orders WHERE id = $1`, [id]);
  const items = order.rows[0].items;

  if (items[itemIndex].status !== 'pending') {
    return res.status(400).json({ error: '已開始製作，無法退點' });
  }

  items.splice(itemIndex, 1);
  const subtotal = items.reduce((sum, i) => sum + (i.price * i.qty), 0);

  await db.query(`
    UPDATE restaurant_orders SET items = $1, subtotal = $2, total = $2 - discount + service_charge
    WHERE id = $3
  `, [JSON.stringify(items), subtotal, id]);

  res.json({ success: true });
});
```

## 5. 營業報表

### 5.1 日報表

```javascript
router.get('/api/reports/daily', async (req, res) => {
  const { date } = req.query; // 2026-03-07

  const result = await db.query(`
    SELECT
      COUNT(*) AS total_orders,
      COUNT(CASE WHEN order_type = 'dine_in' THEN 1 END) AS dine_in_count,
      COUNT(CASE WHEN order_type = 'takeout' THEN 1 END) AS takeout_count,
      COUNT(CASE WHEN order_type = 'delivery' THEN 1 END) AS delivery_count,
      SUM(total) AS revenue,
      AVG(total) AS avg_order_amount,
      SUM(guest_count) AS total_guests,
      ROUND(AVG(total::DECIMAL / NULLIF(guest_count, 0)), 0) AS avg_per_guest
    FROM restaurant_orders
    WHERE status = 'completed'
      AND DATE(opened_at) = $1
  `, [date]);

  // 時段分析
  const hourly = await db.query(`
    SELECT
      EXTRACT(HOUR FROM opened_at) AS hour,
      COUNT(*) AS orders,
      SUM(total) AS revenue
    FROM restaurant_orders
    WHERE status = 'completed' AND DATE(opened_at) = $1
    GROUP BY EXTRACT(HOUR FROM opened_at)
    ORDER BY hour
  `, [date]);

  // 熱門菜品 TOP 10
  const topDishes = await db.query(`
    SELECT
      item->>'name' AS name,
      SUM((item->>'qty')::INT) AS total_qty,
      SUM((item->>'price')::INT * (item->>'qty')::INT) AS total_revenue
    FROM restaurant_orders,
      jsonb_array_elements(items) AS item
    WHERE status = 'completed' AND DATE(opened_at) = $1
    GROUP BY item->>'name'
    ORDER BY total_qty DESC
    LIMIT 10
  `, [date]);

  res.json({
    summary: result.rows[0],
    hourly: hourly.rows,
    topDishes: topDishes.rows,
  });
});
```

### 5.2 週/月報表

```javascript
router.get('/api/reports/range', async (req, res) => {
  const { start, end } = req.query;

  const daily = await db.query(`
    SELECT
      DATE(opened_at) AS date,
      COUNT(*) AS orders,
      SUM(total) AS revenue,
      AVG(total) AS avg_amount,
      SUM(guest_count) AS guests
    FROM restaurant_orders
    WHERE status = 'completed'
      AND DATE(opened_at) BETWEEN $1 AND $2
    GROUP BY DATE(opened_at)
    ORDER BY date
  `, [start, end]);

  // 翻桌率
  const turnover = await db.query(`
    SELECT
      DATE(o.opened_at) AS date,
      ROUND(COUNT(o.id)::DECIMAL / (SELECT COUNT(*) FROM tables), 1) AS turnover_rate
    FROM restaurant_orders o
    WHERE o.status = 'completed'
      AND o.order_type = 'dine_in'
      AND DATE(o.opened_at) BETWEEN $1 AND $2
    GROUP BY DATE(o.opened_at)
    ORDER BY date
  `, [start, end]);

  res.json({ daily: daily.rows, turnover: turnover.rows });
});
```

### 5.3 客單價與翻桌率定義

| 指標 | 公式 | 說明 |
|------|------|------|
| 營業額 | SUM(total) | 日/週/月總收入 |
| 客單價 | 營業額 / 訂單數 | 每筆訂單平均金額 |
| 人均消費 | 營業額 / 總來客數 | 每人平均消費 |
| 翻桌率 | 當日內用訂單數 / 桌位數 | 每桌平均服務幾組客人 |
| 毛利率 | (營業額 - 食材成本) / 營業額 | 扣除原料後的利潤率 |

```javascript
// 取得庫存不足的食材
router.get('/api/inventory/alerts', async (req, res) => {
  const result = await db.query(`
    SELECT id, name, unit, current_stock, safety_stock, supplier,
      CASE
        WHEN current_stock = 0 THEN 'out_of_stock'
        WHEN current_stock <= safety_stock THEN 'low_stock'
        WHEN current_stock <= safety_stock * 1.5 THEN 'warning'
      END AS alert_level
    FROM ingredients
    WHERE current_stock <= safety_stock * 1.5
    ORDER BY
      CASE WHEN current_stock = 0 THEN 0 ELSE 1 END,
      current_stock / NULLIF(safety_stock, 0) ASC
  `);

  res.json({ alerts: result.rows });
});
```

### 6.3 自動叫貨建議

```javascript
async function generateRestockSuggestion() {
  // 分析過去 7 天每日平均用量
  const usage = await db.query(`
    SELECT
      ingredient_id,
      i.name,
      i.unit,
      i.current_stock,
      i.safety_stock,
      i.supplier,
      i.cost_per_unit,
      ABS(SUM(amount)) / 7.0 AS avg_daily_usage
    FROM inventory_logs l
    JOIN ingredients i ON i.id = l.ingredient_id
    WHERE l.type = 'consume'
      AND l.created_at >= now() - INTERVAL '7 days'
    GROUP BY l.ingredient_id, i.id
  `);

  return usage.rows.map(item => {
    const daysRemaining = item.current_stock / (item.avg_daily_usage || 1);
    const suggestAmount = Math.ceil(item.avg_daily_usage * 7) - item.current_stock; // 補到 7 天用量
    return {
      ...item,
      daysRemaining: Math.round(daysRemaining),
      suggestAmount: Math.max(0, suggestAmount),
      estimatedCost: Math.max(0, suggestAmount) * item.cost_per_unit,
      urgent: daysRemaining < 2,
    };
  }).filter(i => i.suggestAmount > 0)
    .sort((a, b) => a.daysRemaining - b.daysRemaining);
}
```

```javascript
  const existing = await db.query(`
    SELECT id FROM schedules WHERE staff_id = $1 AND date = $2
  `, [staffId, date]);

  if (existing.rows.length > 0) {
    return res.status(400).json({ error: '該員工當天已有排班' });
  }

  const result = await db.query(`
    INSERT INTO schedules (staff_id, date, shift, start_time, end_time)
    VALUES ($1, $2, $3, $4, $5) RETURNING *
  `, [staffId, date, shift, template.start, template.end]);

  res.json(result.rows[0]);
});

// 打卡
router.post('/api/schedules/:id/check-in', async (req, res) => {
  const now = new Date().toTimeString().slice(0, 5);
  await db.query(`
    UPDATE schedules SET actual_start = $1, status = 'checked_in' WHERE id = $2
  `, [now, req.params.id]);
  res.json({ success: true, checkedIn: now });
});
```

### 7.3 工時統計

```javascript
router.get('/api/staff/:id/hours', async (req, res) => {
  const { id } = req.params;
  const { month } = req.query; // 2026-03

  const result = await db.query(`
    SELECT
      s.name,
      s.hourly_rate,
      COUNT(sc.id) AS total_shifts,
      SUM(
        EXTRACT(EPOCH FROM (sc.actual_end::TIME - sc.actual_start::TIME)) / 3600
      ) AS total_hours,
      COUNT(CASE WHEN sc.status = 'absent' THEN 1 END) AS absent_count
    FROM staff s
    LEFT JOIN schedules sc ON sc.staff_id = s.id
      AND TO_CHAR(sc.date, 'YYYY-MM') = $2
      AND sc.status IN ('checked_out', 'absent')
    WHERE s.id = $1
    GROUP BY s.id
  `, [id, month]);

  const data = result.rows[0];
  if (data) {
    data.estimated_pay = Math.round(data.total_hours * data.hourly_rate);
  }

  res.json(data);
});
```

```html
let kitchenWs;

async function loadKitchenBoard() {
  const res = await fetch('/api/orders/kitchen-board');
  const { orders } = await res.json();
  renderBoard(orders);
}

function renderBoard(orders) {
  const columns = { pending: [], cooking: [], ready: [] };

  orders.forEach(order => {
    order.items.forEach((item, idx) => {
      if (['pending', 'cooking', 'ready'].includes(item.status)) {
        columns[item.status].push({ order, item, itemIdx: idx });
      }
    });
  });

  Object.keys(columns).forEach(status => {
    const container = document.getElementById(`cards-${status}`);
    document.getElementById(`count-${status}`).textContent = columns[status].length;

    container.innerHTML = columns[status].map(({ order, item, itemIdx }) => `
      <div class="order-card ${getUrgencyClass(item)}" data-order="${order.id}" data-idx="${itemIdx}">
        <div class="card-header">
          <span class="table-num">${order.table_number || '外帶'}</span>
          <span class="order-num">#${order.order_number}</span>
          <span class="time">${getElapsedTime(item.created_at || order.opened_at)}</span>
        </div>
        <div class="card-body">
          <h3>${item.name} x${item.qty}</h3>
          ${item.note ? `<p class="note">${item.note}</p>` : ''}
        </div>
        <div class="card-actions">
          ${status === 'pending' ? `<button onclick="startCooking('${order.id}', ${itemIdx})">開始製作</button>` : ''}
          ${status === 'cooking' ? `<button onclick="markReady('${order.id}', ${itemIdx})">製作完成</button>` : ''}
          ${status === 'ready' ? `<button onclick="markServed('${order.id}', ${itemIdx})">已出餐</button>` : ''}

      </div>

function getUrgencyClass(item) {
  const elapsed = (Date.now() - new Date(item.created_at).getTime()) / 60000;
  if (elapsed > 20) return 'urgent';
  if (elapsed > 10) return 'warning';
  return '';
}

function getElapsedTime(time) {
  const min = Math.floor((Date.now() - new Date(time).getTime()) / 60000);
  return `${min} 分`;
}

    headers: { 'Content-Type': 'application/json' },

  loadKitchenBoard();

  await fetch(`/api/orders/${orderId}/items/${itemIdx}/ready`, { method: 'POST' });

    if (data.type === 'new_items' || data.type === 'order_update') {
      loadKitchenBoard(); // 有新單就刷新
```
