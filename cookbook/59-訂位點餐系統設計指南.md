### 1.4 候位邏輯

```javascript
class WaitingQueue {
  constructor() {
    this.queue = []; // { id, name, phone, partySize, joinedAt }
  }

  enqueue(customer) {
    const ticket = {
      id: `W-${Date.now()}`,
      ...customer,
      joinedAt: new Date().toISOString(),
      estimatedWait: this.estimateWait(customer.partySize)
    };
    this.queue.push(ticket);
    return ticket;
  }

  dequeue() {
    return this.queue.shift();
  }

  estimateWait(partySize) {
    // 依前面排隊人數和平均用餐時間估算
    const ahead = this.queue.length;
    const avgDiningMinutes = 60;
    const turnoverRate = 0.3; // 每分鐘約 30% 機率有桌空出
    return Math.ceil(ahead / turnoverRate) + 5; // 分鐘
  }

  getPosition(ticketId) {
    const idx = this.queue.findIndex(t => t.id === ticketId);
    return idx === -1 ? null : idx + 1;
  }
}
```

    status: 'pending',

## 5. QR Code 掃碼點餐

### 5.1 流程設計

```
客人入座 → 掃桌上 QR Code → 開啟點餐頁面（帶桌號參數）
         → 瀏覽菜單 → 選餐客製化 → 確認送出
         → 後廚看板即時收到 → 製作完成出餐
         → 客人可隨時掃碼加點
```

### 5.2 QR Code 生成

```javascript
// 每桌一個專屬 QR Code，帶桌號參數
// 安裝: npm install qrcode

const QRCode = require('qrcode');

async function generateTableQR(tableId, tableName) {
  const url = `https://your-restaurant.com/order?table=${tableId}`;
  const qrDataUrl = await QRCode.toDataURL(url, {
    width: 300,
    margin: 2,
    color: { dark: '#000', light: '#fff' }
  });
  return qrDataUrl;
}

// 批次生成所有桌的 QR Code 並列印
async function printAllTableQR(tables) {
  for (const table of tables) {
    const qr = await generateTableQR(table.id, table.name);
    console.log(`桌 ${table.name}: ${qr}`);
    // 可輸出為 PDF 列印
  }
}
```

    });
}
