```typescript
// src/services/dalle.ts
import OpenAI from 'openai';
import fs from 'fs/promises';
import path from 'path';

const openai = new OpenAI({
  apiKey: process.env.OPENAI_API_KEY,
});

/**
 * DALL-E 3 圖片生成參數
 */
interface DalleGenerateOptions {
  prompt: string;
  /** 圖片尺寸：1024x1024 | 1024x1792（直）| 1792x1024（橫） */
  size?: '1024x1024' | '1024x1792' | '1792x1024';
  /** 品質：standard（$0.04）| hd（$0.08）更細節、更一致 */
  quality?: 'standard' | 'hd';
  /** 風格：vivid（鮮豔誇張）| natural（自然寫實） */
  style?: 'vivid' | 'natural';
  /** 回傳格式：url（預設，1 小時過期）| b64_json（base64） */
  responseFormat?: 'url' | 'b64_json';
}

interface DalleResult {
  url?: string;
  base64?: string;
  revisedPrompt: string;
  model: string;
  cost: number;
}

/**
 * 使用 DALL-E 3 生成圖片
 */
export async function generateImage(options: DalleGenerateOptions): Promise<DalleResult> {
  const {
    prompt,
    size = '1024x1024',
    quality = 'standard',
    style = 'vivid',
    responseFormat = 'url',
  } = options;

  console.log(`[DALL-E 3] 生成圖片: "${prompt.slice(0, 50)}..."`);

  const response = await openai.images.generate({
    model: 'dall-e-3',
    prompt,
    n: 1, // DALL-E 3 每次只能生成 1 張
    size,
    quality,
    style,
    response_format: responseFormat,
  });

  const image = response.data[0];

  // 計算費用
  const costMap: Record<string, Record<string, number>> = {
    standard: { '1024x1024': 0.04, '1024x1792': 0.08, '1792x1024': 0.08 },
    hd: { '1024x1024': 0.08, '1024x1792': 0.12, '1792x1024': 0.12 },
  };
  const cost = costMap[quality]?.[size] ?? 0.04;

  return {
    url: image.url,
    base64: image.b64_json,
    revisedPrompt: image.revised_prompt ?? prompt,
    model: 'dall-e-3',
    cost,
  };
}

/**
 * 生成圖片並直接存檔
 */
export async function generateAndSave(
  prompt: string,
  outputPath: string,
  options?: Partial<DalleGenerateOptions>
): Promise<{ filePath: string; revisedPrompt: string; cost: number }> {
  const result = await generateImage({
    prompt,
    responseFormat: 'b64_json',
    ...options,
  });

  if (!result.base64) {
    throw new Error('未取得 base64 資料');
  }

  const buffer = Buffer.from(result.base64, 'base64');
  const dir = path.dirname(outputPath);
  await fs.mkdir(dir, { recursive: true });
  await fs.writeFile(outputPath, buffer);

  console.log(`[DALL-E 3] 已儲存: ${outputPath} (${(buffer.length / 1024).toFixed(1)} KB)`);

  return {
    filePath: outputPath,
    revisedPrompt: result.revisedPrompt,
    cost: result.cost,
  };
}
```

```typescript
 * 從 URL 下載圖片到本地

export async function downloadImage(imageUrl: string, outputPath: string): Promise<string> {
  const dir = path.dirname(outputPath);
  await fs.promises.mkdir(dir, { recursive: true });

  return new Promise((resolve, reject) => {
    const file = fs.createWriteStream(outputPath);
    https.get(imageUrl, (response) => {
      response.pipe(file);
      file.on('finish', () => {
        file.close();
        console.log(`[Download] 已儲存: ${outputPath}`);
        resolve(outputPath);
      });
    }).on('error', (err) => {
      fs.unlink(outputPath, () => {}); // 清理失敗的檔案
      reject(err);
    });
  });
}
```

```typescript
// src/services/runpod-sd.ts

interface RunPodSDOptions {
  prompt: string;
  negativePrompt?: string;
  width?: number;
  height?: number;
  steps?: number;
  cfgScale?: number;
  sampler?: string;
  seed?: number;
  model?: string;
}

interface RunPodResult {
  images: string[];    // base64 encoded
  status: string;
  executionTime: number;
}

/**
 * RunPod Serverless Stable Diffusion
 * 適合大量生成、自建模型的場景
 */
export async function generateWithRunPod(options: RunPodSDOptions): Promise<RunPodResult> {
  const {
    prompt,
    negativePrompt = 'low quality, blurry, watermark',
    width = 1024,
    height = 1024,
    steps = 30,
    cfgScale = 7,
    sampler = 'DPM++ 2M Karras',
    seed = -1,
  } = options;

  const endpointId = process.env.RUNPOD_ENDPOINT_ID;
  const apiKey = process.env.RUNPOD_API_KEY;

  // 1. 提交任務（異步）
  const runResponse = await fetch(
    `https://api.runpod.ai/v2/${endpointId}/run`,
    {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        Authorization: `Bearer ${apiKey}`,
      },
      body: JSON.stringify({
        input: {
          prompt,
          negative_prompt: negativePrompt,
          width,
          height,
          num_inference_steps: steps,
          cfg_scale: cfgScale,
          sampler_name: sampler,
          seed,
        },
      }),
    }
  );

  const runData = await runResponse.json() as { id: string };
  const jobId = runData.id;
  console.log(`[RunPod] 任務已提交: ${jobId}`);

  // 2. 輪詢等待結果
  const startTime = Date.now();
  const maxWait = 120_000; // 最多等 2 分鐘

  while (Date.now() - startTime < maxWait) {
    const statusResponse = await fetch(
      `https://api.runpod.ai/v2/${endpointId}/status/${jobId}`,
      {
        headers: { Authorization: `Bearer ${apiKey}` },
      }
    );

    const statusData = await statusResponse.json() as {
      status: string;
      output?: { images: string[] };
    };

    if (statusData.status === 'COMPLETED' && statusData.output) {
      return {
        images: statusData.output.images,
        status: 'COMPLETED',
        executionTime: Date.now() - startTime,
      };
    }

    if (statusData.status === 'FAILED') {
      throw new Error(`RunPod 任務失敗: ${jobId}`);
    }

    // 等 2 秒再查
    await new Promise((r) => setTimeout(r, 2000));
  }

  throw new Error(`RunPod 任務逾時: ${jobId}`);
}

/**
 * 將 RunPod 回傳的 base64 圖片存檔
 */
export async function saveRunPodImages(
  images: string[],
  outputDir: string,
  prefix: string = 'runpod'
): Promise<string[]> {
  const fs = await import('fs/promises');
  await fs.mkdir(outputDir, { recursive: true });

  const savedPaths: string[] = [];

  for (let i = 0; i < images.length; i++) {
    const buffer = Buffer.from(images[i], 'base64');
    const filePath = `${outputDir}/${prefix}-${Date.now()}-${i}.png`;
    await fs.writeFile(filePath, buffer);
    savedPaths.push(filePath);
    console.log(`[RunPod] 已儲存: ${filePath}`);
  }

  return savedPaths;
}
```

## 3. AI 背景移除

### 3.1 使用 Replicate（remove.bg 替代方案）

```typescript
// src/services/background-removal.ts
import Replicate from 'replicate';
import sharp from 'sharp';
import fs from 'fs/promises';
import path from 'path';

const replicate = new Replicate({
  auth: process.env.REPLICATE_API_TOKEN,
});

/**
 * AI 背景移除 — 使用 RMBG（開源，Replicate 上免費層級可用）
 */
export async function removeBackground(
  inputPath: string,
  outputPath?: string
): Promise<string> {
  // 讀取圖片轉 base64 或使用 URL
  const imageBuffer = await fs.readFile(inputPath);
  const base64 = `data:image/png;base64,${imageBuffer.toString('base64')}`;

  console.log(`[BG Remove] 處理中: ${inputPath}`);

  const output = await replicate.run(
    'cjwbw/rembg:fb8af171cfa1616ddcf1242c093f9c46bcada5ad4cf6f2fbe8b81b330ec5c003',
    {
      input: {
        image: base64,
      },
    }
  );

  // 下載結果
  const resultUrl = output as string;
  const response = await fetch(resultUrl);
  const resultBuffer = Buffer.from(await response.arrayBuffer());

  const outPath = outputPath ?? inputPath.replace(/\.\w+$/, '-nobg.png');
  const dir = path.dirname(outPath);
  await fs.mkdir(dir, { recursive: true });
  await fs.writeFile(outPath, resultBuffer);

  console.log(`[BG Remove] 完成: ${outPath}`);
  return outPath;
}

/**
 * 背景移除 + 替換新背景
 */
export async function replaceBackground(
  inputPath: string,
  backgroundPath: string,
  outputPath: string
): Promise<string> {
  // 1. 先移除背景
  const noBgPath = inputPath.replace(/\.\w+$/, '-temp-nobg.png');
  await removeBackground(inputPath, noBgPath);

  // 2. 取得前景圖的尺寸
  const foreground = sharp(noBgPath);
  const fgMeta = await foreground.metadata();

  // 3. 將背景調整為相同尺寸
  const background = sharp(backgroundPath).resize(
    fgMeta.width,
    fgMeta.height,
    { fit: 'cover' }
  );

  // 4. 合成
  const bgBuffer = await background.toBuffer();
  const fgBuffer = await fs.readFile(noBgPath);

  await sharp(bgBuffer)
    .composite([{ input: fgBuffer, blend: 'over' }])
    .toFile(outputPath);

  // 5. 清理暫存檔
  await fs.unlink(noBgPath).catch(() => {});

  console.log(`[BG Replace] 完成: ${outputPath}`);
  return outputPath;
}
```

### 3.2 使用 Sharp 純本地處理（簡單背景）

```typescript
// src/services/simple-bg-removal.ts
import sharp from 'sharp';

/**
 * 本地簡易背景移除 — 適用於純色背景（如白底產品圖）
 * 不需要 API，完全本地處理
 */
export async function removeSimpleBackground(
  inputPath: string,
  outputPath: string,
  options: {
    /** 要移除的背景色（預設白色） */
    backgroundColor?: { r: number; g: number; b: number };
    /** 容差值 0-255（越大越寬鬆） */
    tolerance?: number;
  } = {}
): Promise<string> {
  const { backgroundColor = { r: 255, g: 255, b: 255 }, tolerance = 30 } = options;

  const { data, info } = await sharp(inputPath)
    .raw()
    .ensureAlpha()
    .toBuffer({ resolveWithObject: true });

  const pixels = new Uint8Array(data);

  for (let i = 0; i < pixels.length; i += 4) {
    const r = pixels[i];
    const g = pixels[i + 1];
    const b = pixels[i + 2];

    const diffR = Math.abs(r - backgroundColor.r);
    const diffG = Math.abs(g - backgroundColor.g);
    const diffB = Math.abs(b - backgroundColor.b);

    if (diffR <= tolerance && diffG <= tolerance && diffB <= tolerance) {
      pixels[i + 3] = 0; // 設為透明
    }
  }

  await sharp(Buffer.from(pixels), {
    raw: { width: info.width, height: info.height, channels: 4 },
  })
    .png()
    .toFile(outputPath);

  console.log(`[Simple BG] 完成: ${outputPath}`);
  return outputPath;
}
```

```typescript
        image: base64,

  const dir = path.dirname(outPath);
  await fs.mkdir(dir, { recursive: true });

  const newSize = resultBuffer.length;
  console.log(
    `[Upscale] 完成: ${outPath} ` +
    `(${(originalSize / 1024).toFixed(0)}KB → ${(newSize / 1024).toFixed(0)}KB)`
  );

  return outPath;
}

/**
 * 批次放大整個資料夾的圖片
 */
export async function batchUpscale(
  inputDir: string,
  outputDir: string,
  options: { scale?: 2 | 4; concurrency?: number } = {}
): Promise<string[]> {
  const { scale = 2, concurrency = 3 } = options;

  await fs.mkdir(outputDir, { recursive: true });

  const files = await fs.readdir(inputDir);
  const imageFiles = files.filter((f) =>
    /\.(jpg|jpeg|png|webp)$/i.test(f)
  );

  console.log(`[Batch Upscale] 找到 ${imageFiles.length} 張圖片`);

  const results: string[] = [];

  // 控制並行數量
  for (let i = 0; i < imageFiles.length; i += concurrency) {
    const batch = imageFiles.slice(i, i + concurrency);
    const promises = batch.map(async (file) => {
      const inputPath = path.join(inputDir, file);
      const outputPath = path.join(outputDir, file.replace(/\.\w+$/, `.png`));
      return upscaleImage(inputPath, outputPath, { scale });
    });

    const batchResults = await Promise.allSettled(promises);
    for (const r of batchResults) {
      if (r.status === 'fulfilled') {
        results.push(r.value);
      } else {
        console.error(`[Batch Upscale] 失敗:`, r.reason);
      }
    }

    console.log(`[Batch Upscale] 進度: ${Math.min(i + concurrency, imageFiles.length)}/${imageFiles.length}`);
  }

  return results;
}
```

```typescript
import sharp from 'sharp';

/**
 * 非 AI 的圖片放大 — 使用 Lanczos3 插值
 * 品質不如 AI，但完全免費、即時處理
 */
export async function simpleUpscale(
  inputPath: string,
  outputPath: string,
  scale: number = 2
): Promise<string> {

  const newWidth = Math.round((metadata.width ?? 512) * scale);

      fit: 'fill',

    .sharpen({ sigma: 0.5 }) // 輕微銳化
    .toFile(outputPath);

  console.log(`[Simple Upscale] ${scale}x: ${newWidth}x${newHeight} → ${outputPath}`);
  return outputPath;
}
```

```typescript
 * 圖片處理管線設定
 */
interface ImagePipelineConfig {
  /** 輸出資料夾 */
  outputDir: string;
  /** 要產生的尺寸（寬度，高度自動等比） */
  sizes: { name: string; width: number }[];
  /** 輸出格式 */
  formats: ('webp' | 'avif' | 'jpeg' | 'png')[];
  /** 品質設定 */
  quality: {
    webp?: number;   // 預設 80
    avif?: number;   // 預設 50
    jpeg?: number;   // 預設 85
    png?: number;    // 預設 90（壓縮等級）
  };
  /** 是否產生 placeholder（模糊小圖，用於 LQIP） */
  generatePlaceholder?: boolean;
  /** 是否保留 EXIF 資料 */
  keepMetadata?: boolean;
}

interface ProcessedImage {
  original: string;
  outputs: {
    format: string;
    size: string;
    path: string;
    width: number;
    height: number;
    fileSize: number;
  }[];
  placeholder?: string; // base64 data URI
  totalSaved: number;   // 相對原圖節省的位元組
}

/**
 * 預設的響應式圖片管線設定
 */
export const defaultPipelineConfig: ImagePipelineConfig = {
  outputDir: './output',
  sizes: [
    { name: 'sm', width: 480 },
    { name: 'md', width: 768 },
    { name: 'lg', width: 1024 },
    { name: 'xl', width: 1440 },
    { name: '2xl', width: 1920 },
  ],
  formats: ['webp', 'avif'],
  quality: {
    webp: 80,
    avif: 50,
    jpeg: 85,
  },
  generatePlaceholder: true,
  keepMetadata: false,
};

/**
 * 處理單張圖片 — 產生多尺寸、多格式的輸出
 */
export async function processImage(
  inputPath: string,
  config: Partial<ImagePipelineConfig> = {}
): Promise<ProcessedImage> {
  const cfg = { ...defaultPipelineConfig, ...config };
  await fs.mkdir(cfg.outputDir, { recursive: true });

  const originalBuffer = await fs.readFile(inputPath);
  const originalSize = originalBuffer.length;
  const baseName = path.basename(inputPath, path.extname(inputPath));

  const metadata = await sharp(inputPath).metadata();
  const originalWidth = metadata.width ?? 1920;

  const outputs: ProcessedImage['outputs'] = [];
  let totalOutputSize = 0;

  // 產生各尺寸 x 各格式的組合
  for (const size of cfg.sizes) {
    // 跳過比原圖大的尺寸
    if (size.width > originalWidth) continue;

    for (const format of cfg.formats) {
      const outputName = `${baseName}-${size.name}.${format}`;
      const outputPath = path.join(cfg.outputDir, outputName);

      let pipeline = sharp(inputPath).resize(size.width, undefined, {
        fit: 'inside',
        withoutEnlargement: true,
      });

      if (!cfg.keepMetadata) {
        pipeline = pipeline.withMetadata({});
      }

      switch (format) {
        case 'webp':
          pipeline = pipeline.webp({ quality: cfg.quality.webp ?? 80 });
          break;
        case 'avif':
          pipeline = pipeline.avif({ quality: cfg.quality.avif ?? 50 });
          break;
        case 'jpeg':
          pipeline = pipeline.jpeg({
            quality: cfg.quality.jpeg ?? 85,
            mozjpeg: true,
          });
          break;
        case 'png':
          pipeline = pipeline.png({
            compressionLevel: 9,
            adaptiveFiltering: true,
          });
          break;
      }

      const outputBuffer = await pipeline.toBuffer();
      await fs.writeFile(outputPath, outputBuffer);

      const outputMeta = await sharp(outputBuffer).metadata();

      outputs.push({
        format,
        size: size.name,
        path: outputPath,
        width: outputMeta.width ?? size.width,
        height: outputMeta.height ?? 0,
        fileSize: outputBuffer.length,
      });

      totalOutputSize += outputBuffer.length;
    }
  }

  // 產生 LQIP placeholder
  let placeholder: string | undefined;
  if (cfg.generatePlaceholder) {
    const placeholderBuffer = await sharp(inputPath)
      .resize(20, undefined, { fit: 'inside' })
      .blur(10)
      .webp({ quality: 20 })
      .toBuffer();

    placeholder = `data:image/webp;base64,${placeholderBuffer.toString('base64')}`;
  }

  const avgOutputSize = totalOutputSize / outputs.length;
  const saved = originalSize - avgOutputSize;

  console.log(
    `[Pipeline] ${baseName}: ${outputs.length} 個輸出，` +
    `平均節省 ${((saved / originalSize) * 100).toFixed(1)}%`
  );

  return {
    original: inputPath,
    outputs,
    placeholder,
    totalSaved: saved,
  };
}

/**
 * 批次處理整個資料夾
 */
export async function processDirectory(
  inputDir: string,
  config: Partial<ImagePipelineConfig> = {}
): Promise<ProcessedImage[]> {
  const files = await fs.readdir(inputDir);
  const imageFiles = files.filter((f) =>
    /\.(jpg|jpeg|png|webp|tiff|avif)$/i.test(f)
  );

  console.log(`[Pipeline] 找到 ${imageFiles.length} 張圖片，開始處理...`);

  const results: ProcessedImage[] = [];

  for (const file of imageFiles) {
    try {
      const result = await processImage(
        path.join(inputDir, file),
        config
      );
      results.push(result);
    } catch (err) {
      console.error(`[Pipeline] 處理 ${file} 失敗:`, err);
    }
  }

  // 統計
  const totalOriginal = results.reduce(
    (sum, r) => sum + (r.totalSaved > 0 ? r.totalSaved : 0),
    0
  );
  console.log(
    `[Pipeline] 完成！${results.length} 張圖片，` +
    `總共節省 ${(totalOriginal / 1024 / 1024).toFixed(1)} MB`
  );

  return results;
}
```

```typescript
  for (const o of outputs) {

  // AVIF source（如果有）

  if (avifSet) {

      .map((o) => `${o.path} ${o.width}w`)
      .join(', ');
    html += `  <source type="image/avif" srcset="${srcset}" />\n`;
  }

  // WebP source（如果有）
  const webpSet = byFormat.get('webp');
  if (webpSet) {
    const srcset = webpSet
      .map((o) => `${o.path} ${o.width}w`)
      .join(', ');
    html += `  <source type="image/webp" srcset="${srcset}" />\n`;
  }

  // Fallback img
  const fallback = byFormat.get('jpeg') ?? byFormat.get('png') ?? outputs;
  const fallbackSrc = fallback[fallback.length - 1]?.path ?? '';
  const sizes = '(max-width: 480px) 480px, (max-width: 768px) 768px, (max-width: 1024px) 1024px, 1440px';

  if (placeholder) {
    html += `  <img src="${fallbackSrc}" alt="${alt}" sizes="${sizes}" loading="lazy" `;
    html += `style="background-image:url(${placeholder});background-size:cover" />\n`;
  } else {
    html += `  <img src="${fallbackSrc}" alt="${alt}" sizes="${sizes}" loading="lazy" />\n`;
  }

  html += '</picture>';

  return html;
}
```

### 5.3 Express 即時縮圖中間件

```typescript
// src/middleware/image-resize.ts
import { Request, Response, NextFunction } from 'express';
import sharp from 'sharp';
import fs from 'fs/promises';
import path from 'path';
import crypto from 'crypto';

/**
 * 即時圖片處理中間件
 *
 * 用法：GET /images/photo.jpg?w=800&h=600&format=webp&q=80
 *
 * 參數：
 *   w — 寬度
 *   h — 高度
 *   format — 輸出格式（webp, avif, jpeg, png）
 *   q — 品質 1-100
 *   fit — 裁切方式（cover, contain, fill, inside, outside）
 */
export function imageResizeMiddleware(
  imageDir: string,
  cacheDir: string = '/tmp/image-cache'
) {
  return async (req: Request, res: Response, next: NextFunction) => {
    // 只處理圖片路徑
    if (!req.path.match(/\.(jpg|jpeg|png|webp|avif|gif)$/i)) {
      return next();
    }

    const imagePath = path.join(imageDir, req.path);

    // 確認原始檔案存在
    try {
      await fs.access(imagePath);
    } catch {
      return next();
    }

    const width = req.query.w ? parseInt(req.query.w as string) : undefined;
    const height = req.query.h ? parseInt(req.query.h as string) : undefined;
    const format = (req.query.format as string) || 'webp';
    const quality = req.query.q ? parseInt(req.query.q as string) : 80;
    const fit = (req.query.fit as keyof sharp.FitEnum) || 'cover';

    // 沒有任何處理參數，直接送原始檔
    if (!width && !height && !req.query.format) {
      return next();
    }

    // 快取 key
    const cacheKey = crypto
      .createHash('md5')
      .update(`${req.path}-${width}-${height}-${format}-${quality}-${fit}`)
      .digest('hex');
    const cachePath = path.join(cacheDir, `${cacheKey}.${format}`);

    // 檢查快取
    try {
      const cached = await fs.readFile(cachePath);
      res.type(`image/${format === 'jpg' ? 'jpeg' : format}`);
      res.set('Cache-Control', 'public, max-age=31536000, immutable');
      res.set('X-Image-Cache', 'HIT');
      return res.send(cached);
    } catch {
      // 快取不存在，繼續處理
    }

    try {
      let pipeline = sharp(imagePath);

      if (width || height) {
        pipeline = pipeline.resize(width, height, {
          fit,
          withoutEnlargement: true,
        });
      }

      switch (format) {
        case 'webp':
          pipeline = pipeline.webp({ quality });
          break;
        case 'avif':
          pipeline = pipeline.avif({ quality });
          break;
        case 'jpeg':
        case 'jpg':
          pipeline = pipeline.jpeg({ quality, mozjpeg: true });
          break;
        case 'png':
          pipeline = pipeline.png({ compressionLevel: 9 });
          break;
      }

      const outputBuffer = await pipeline.toBuffer();

      // 存快取
      await fs.mkdir(cacheDir, { recursive: true });
      await fs.writeFile(cachePath, outputBuffer);

      res.type(`image/${format === 'jpg' ? 'jpeg' : format}`);
      res.set('Cache-Control', 'public, max-age=31536000, immutable');
      res.set('X-Image-Cache', 'MISS');
      res.send(outputBuffer);
    } catch (err) {
      console.error(`[Image Resize] 錯誤:`, err);
      next();
    }
  };
}
```

```typescript
export async function recognizeText(imagePath: string): Promise<OCRResult> {
  console.log(`[OCR] 辨識中: ${imagePath}`);

  const imageBuffer = await fs.readFile(imagePath);

  const [result] = await client.documentTextDetection({
    image: { content: imageBuffer.toString('base64') },
    imageContext: {
      languageHints: ['zh-TW', 'en', 'ja'],
    },
  });

  const fullTextAnnotation = result.fullTextAnnotation;

  if (!fullTextAnnotation) {
    return { fullText: '', blocks: [], languages: [] };
  }

  const blocks: OCRResult['blocks'] = [];
  const languages = new Set<string>();

  for (const page of fullTextAnnotation.pages ?? []) {
    for (const block of page.blocks ?? []) {
      let blockText = '';
      let totalConfidence = 0;
      let symbolCount = 0;

      for (const paragraph of block.paragraphs ?? []) {
        for (const word of paragraph.words ?? []) {
          for (const symbol of word.symbols ?? []) {
            blockText += symbol.text ?? '';
            totalConfidence += symbol.confidence ?? 0;
            symbolCount++;
          }
          blockText += ' ';
        }
        blockText += '\n';
      }

      const vertices = block.boundingBox?.vertices ?? [];
      const x = vertices[0]?.x ?? 0;
      const y = vertices[0]?.y ?? 0;
      const w = (vertices[2]?.x ?? 0) - x;
      const h = (vertices[2]?.y ?? 0) - y;

      blocks.push({
        text: blockText.trim(),
        confidence: symbolCount > 0 ? totalConfidence / symbolCount : 0,
        boundingBox: { x, y, width: w, height: h },
      });

      // 收集語言
      for (const prop of block.property?.detectedLanguages ?? []) {
        if (prop.languageCode) languages.add(prop.languageCode);
      }
    }
  }

  console.log(`[OCR] 完成: ${blocks.length} 個區塊，${fullTextAnnotation.text?.length ?? 0} 字`);

  return {
    fullText: fullTextAnnotation.text ?? '',
    blocks,
    languages: [...languages],
  };
}
```

```typescript
// src/services/image-retry.ts

/**
 * 帶重試機制的函數包裝器
 * 適用於所有外部 API 呼叫（DALL-E、Replicate、Cloudinary 等）
 */
export async function withRetry<T>(
  fn: () => Promise<T>,
  options: {
    maxRetries?: number;
    initialDelay?: number;
    maxDelay?: number;
    backoffFactor?: number;
    retryableErrors?: string[];
    onRetry?: (error: Error, attempt: number) => void;
  } = {}
): Promise<T> {
  const {
    maxRetries = 3,
    initialDelay = 1000,
    maxDelay = 30000,
    backoffFactor = 2,
    retryableErrors = ['ECONNRESET', 'ETIMEDOUT', 'ENOTFOUND', '429', '500', '502', '503'],
    onRetry,
  } = options;

  let lastError: Error | null = null;

  for (let attempt = 0; attempt <= maxRetries; attempt++) {
    try {
      return await fn();
    } catch (err: any) {
      lastError = err;

      // 判斷是否可重試
      const errorString = String(err.message ?? err);
      const isRetryable = retryableErrors.some((code) =>
        errorString.includes(code)
      );

      if (!isRetryable || attempt === maxRetries) {
        throw err;
      }

      const delay = Math.min(
        initialDelay * Math.pow(backoffFactor, attempt),
        maxDelay
      );

      console.warn(
        `[Retry] 第 ${attempt + 1} 次重試，等待 ${delay}ms`,
        err.message
      );

      onRetry?.(err, attempt + 1);
      await new Promise((r) => setTimeout(r, delay));
    }
  }

  throw lastError;
}
```

```typescript
   * 等待直到可以發送下一個請求
   */
  async acquire(): Promise<void> {
    const now = Date.now();

    // 清除過期的時間戳
    this.timestamps = this.timestamps.filter(
      (t) => now - t < this.windowMs
    );

    if (this.timestamps.length >= this.maxRequests) {
      const oldest = this.timestamps[0];
      const waitTime = this.windowMs - (now - oldest) + 100; // 多等 100ms 安全邊際
      console.log(`[Rate Limit] 等待 ${waitTime}ms...`);
      await new Promise((r) => setTimeout(r, waitTime));
    }

    this.timestamps.push(Date.now());
  }
}

export const cloudinaryRateLimiter = new RateLimiter(500, 3600_000); // 500 次/時
```

## 11. 成本控制與監控

```typescript
// src/services/image-cost-tracker.ts
import fs from 'fs/promises';
import path from 'path';

interface CostEntry {
  timestamp: string;
  service: string;
  operation: string;
  cost: number;
  details?: string;
}

/**
 * 圖片處理成本追蹤器
 * 追蹤每個 API 呼叫的費用，避免帳單爆炸
 */
export class ImageCostTracker {
  private entries: CostEntry[] = [];
  private logPath: string;
  private dailyBudget: number;

  constructor(
    logPath: string = '/tmp/openclaw-image-costs.json',
    dailyBudget: number = 5.0 // 每日上限 $5
  ) {
    this.logPath = logPath;
    this.dailyBudget = dailyBudget;
  }

  /**
   * 記錄一筆費用
   */
  async track(
    service: string,
    operation: string,
    cost: number,
    details?: string
  ): Promise<void> {
    const entry: CostEntry = {
      timestamp: new Date().toISOString(),
      service,
      operation,
      cost,
      details,
    };

    this.entries.push(entry);

    // 寫入檔案
    try {
      let existing: CostEntry[] = [];
      try {
        const data = await fs.readFile(this.logPath, 'utf-8');
        existing = JSON.parse(data);
      } catch {
        // 檔案不存在，用空陣列
      }
      existing.push(entry);
      await fs.writeFile(this.logPath, JSON.stringify(existing, null, 2));
    } catch (err) {
      console.error(`[Cost Tracker] 寫入失敗:`, err);
    }
  }

  /**
   * 檢查今日是否超過預算
   */
  async checkBudget(): Promise<{
    todaySpent: number;
    remaining: number;
    isOverBudget: boolean;
  }> {
    let allEntries: CostEntry[] = [];

    try {
      const data = await fs.readFile(this.logPath, 'utf-8');
      allEntries = JSON.parse(data);
    } catch {
      return { todaySpent: 0, remaining: this.dailyBudget, isOverBudget: false };
    }

    const today = new Date().toISOString().split('T')[0];
    const todayEntries = allEntries.filter((e) =>
      e.timestamp.startsWith(today)
    );
    const todaySpent = todayEntries.reduce((sum, e) => sum + e.cost, 0);

    return {
      todaySpent: Math.round(todaySpent * 1000) / 1000,
      remaining: Math.round((this.dailyBudget - todaySpent) * 1000) / 1000,
      isOverBudget: todaySpent >= this.dailyBudget,
    };
  }

  /**
   * 取得費用摘要
   */
  async getSummary(days: number = 7): Promise<{
    byService: Record<string, number>;
    byDay: Record<string, number>;
    total: number;
  }> {
    let allEntries: CostEntry[] = [];

    try {
      const data = await fs.readFile(this.logPath, 'utf-8');
      allEntries = JSON.parse(data);
    } catch {
      return { byService: {}, byDay: {}, total: 0 };
    }

    const cutoff = new Date();
    cutoff.setDate(cutoff.getDate() - days);

    const recent = allEntries.filter(
      (e) => new Date(e.timestamp) >= cutoff
    );

    const byService: Record<string, number> = {};
    const byDay: Record<string, number> = {};
    let total = 0;

    for (const entry of recent) {
      byService[entry.service] = (byService[entry.service] ?? 0) + entry.cost;
      const day = entry.timestamp.split('T')[0];
      byDay[day] = (byDay[day] ?? 0) + entry.cost;
      total += entry.cost;
    }

    return {
      byService,
      byDay,
      total: Math.round(total * 1000) / 1000,
    };
  }
}

// 匯出全域實例
export const costTracker = new ImageCostTracker();
```

## 附錄 C：常見問題排查

### DALL-E 生成失敗

| 錯誤訊息 | 原因 | 解法 |
|----------|------|------|
| `content_policy_violation` | Prompt 觸發安全過濾 | 調整 prompt，避免敏感詞 |
| `rate_limit_exceeded` | 超過速率限制 | 加速率限制器，等待重試 |
| `billing_hard_limit_reached` | API 額度用完 | 加錢或換帳號 |
| `invalid_size` | 尺寸不在允許範圍 | 只能用 1024x1024/1024x1792/1792x1024 |

### Replicate 問題

| 錯誤 | 原因 | 解法 |
|------|------|------|
| 冷啟動慢（30s+） | 模型需要載入 GPU | 用 Replicate 的 "always-on" 付費方案 |
| `NSFW content detected` | 安全過濾觸發 | 調整 prompt 或用自託管模型 |
| 結果品質差 | CFG Scale 或步數不對 | SDXL: 步數 25-40、CFG 7-9 |

### Sharp 處理問題

| 錯誤 | 原因 | 解法 |
|------|------|------|
| `Input file is missing` | 路徑不存在 | 檢查檔案路徑和權限 |
| `unsupported image format` | 格式不支援 | 確認是 JPEG/PNG/WebP/TIFF/AVIF |
| 記憶體不足 | 圖片太大 | 先 resize 再處理，或用 `sharp.cache(false)` |
| AVIF 編碼慢 | AVIF 本身就慢 | 降低品質或改用 WebP |
