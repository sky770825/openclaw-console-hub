## 1. Google Ads 帳號設定與基本操作

### 1.1 帳號架構概念

```
Google Ads 帳號架構：

帳號（Account）
  └── 廣告活動（Campaign）        ← 預算、地區、排程
        └── 廣告群組（Ad Group）   ← 關鍵字分組、出價
              └── 廣告（Ad）       ← 文案、素材、到達頁
```

**建議命名規範：**

| 層級 | 命名格式 | 範例 |
|------|----------|------|
| 活動 | `[類型]-[產品]-[地區]-[月份]` | `Search-牙醫洗牙-台北-202603` |
| 群組 | `[關鍵字主題]` | `洗牙費用` |
| 廣告 | `[變體編號]-[賣點]` | `V1-首次優惠` |

### 1.2 Google Ads API 設定（TypeScript）

```typescript
// google-ads-setup.ts
// Google Ads API 帳號管理工具

import { GoogleAdsApi, enums } from 'google-ads-api';

// ============================
// 1. API 客戶端初始化
// ============================

interface GoogleAdsConfig {
  clientId: string;
  clientSecret: string;
  developerToken: string;
  refreshToken: string;
  loginCustomerId: string; // MCC 管理帳號 ID
}

function createGoogleAdsClient(config: GoogleAdsConfig): GoogleAdsApi {
  return new GoogleAdsApi({
    client_id: config.clientId,
    client_secret: config.clientSecret,
    developer_token: config.developerToken,
  });
}

// ============================
// 2. 取得帳號下所有廣告活動
// ============================

interface CampaignInfo {
  id: string;
  name: string;
  status: string;
  budget: number;
  type: string;
}

async function listCampaigns(
  client: GoogleAdsApi,
  customerId: string,
  refreshToken: string
): Promise<CampaignInfo[]> {
  const customer = client.Customer({
    customer_id: customerId,
    refresh_token: refreshToken,
  });

  const campaigns = await customer.query(`
    SELECT
      campaign.id,
      campaign.name,
      campaign.status,
      campaign.campaign_budget,
      campaign.advertising_channel_type
    FROM campaign
    WHERE campaign.status != 'REMOVED'
    ORDER BY campaign.name
  `);

  return campaigns.map((row: any) => ({
    id: row.campaign.id.toString(),
    name: row.campaign.name,
    status: row.campaign.status,
    budget: Number(row.campaign.campaign_budget),
    type: row.campaign.advertising_channel_type,
  }));
}

// ============================
// 3. 建立搜尋廣告活動
// ============================

interface CreateCampaignParams {
  name: string;
  dailyBudgetMicros: number; // 微單位，1 元 = 1,000,000
  startDate: string;         // YYYY-MM-DD
  endDate?: string;
  targetLocations: number[]; // 地理位置 ID（台北=1012825）
}

async function createSearchCampaign(
  client: GoogleAdsApi,
  customerId: string,
  refreshToken: string,
  params: CreateCampaignParams
): Promise<string> {
  const customer = client.Customer({
    customer_id: customerId,
    refresh_token: refreshToken,
  });

  // 先建立預算
  const budgetResult = await customer.campaignBudgets.create([
    {
      name: `Budget-${params.name}-${Date.now()}`,
      amount_micros: params.dailyBudgetMicros,
      delivery_method: enums.BudgetDeliveryMethod.STANDARD,
    },
  ]);
  const budgetResourceName = budgetResult.results[0].resource_name;

  // 建立活動
  const campaignResult = await customer.campaigns.create([
    {
      name: params.name,
      campaign_budget: budgetResourceName,
      advertising_channel_type: enums.AdvertisingChannelType.SEARCH,
      status: enums.CampaignStatus.PAUSED, // 先暫停，設定完再啟用
      start_date: params.startDate.replace(/-/g, ''),
      end_date: params.endDate?.replace(/-/g, ''),
      network_settings: {
        target_google_search: true,
        target_search_network: false,
        target_content_network: false,
      },
      geo_target_type_setting: {
        positive_geo_target_type: enums.PositiveGeoTargetType.PRESENCE,
        negative_geo_target_type: enums.NegativeGeoTargetType.PRESENCE,
      },
    },
  ]);

  return campaignResult.results[0].resource_name;
}

// ============================
// 4. 新增關鍵字
// ============================

interface KeywordParams {
  adGroupId: string;
  keywords: Array<{
    text: string;
    matchType: 'EXACT' | 'PHRASE' | 'BROAD';
    cpcBidMicros?: number;
  }>;
}

async function addKeywords(
  client: GoogleAdsApi,
  customerId: string,
  refreshToken: string,
  params: KeywordParams
): Promise<void> {
  const customer = client.Customer({
    customer_id: customerId,
    refresh_token: refreshToken,
  });

  const matchTypeMap = {
    EXACT: enums.KeywordMatchType.EXACT,
    PHRASE: enums.KeywordMatchType.PHRASE,
    BROAD: enums.KeywordMatchType.BROAD,
  };

  const criteria = params.keywords.map((kw) => ({
    ad_group: `customers/${customerId}/adGroups/${params.adGroupId}`,
    keyword: {
      text: kw.text,
      match_type: matchTypeMap[kw.matchType],
    },
    cpc_bid_micros: kw.cpcBidMicros ?? 0,
    status: enums.AdGroupCriterionStatus.ENABLED,
  }));

  await customer.adGroupCriteria.create(criteria);
}

// ============================
// 5. 取得廣告成效報表
// ============================

interface AdPerformance {
  campaignName: string;
  adGroupName: string;
  impressions: number;
  clicks: number;
  cost: number;
  conversions: number;
  ctr: number;
  cpc: number;
  conversionRate: number;
  date: string;
}

async function getPerformanceReport(
  client: GoogleAdsApi,
  customerId: string,
  refreshToken: string,
  startDate: string,
  endDate: string
): Promise<AdPerformance[]> {
  const customer = client.Customer({
    customer_id: customerId,
    refresh_token: refreshToken,
  });

  const rows = await customer.query(`
    SELECT
      campaign.name,
      ad_group.name,
      metrics.impressions,
      metrics.clicks,
      metrics.cost_micros,
      metrics.conversions,
      metrics.ctr,
      metrics.average_cpc,
      segments.date
    FROM ad_group
    WHERE segments.date BETWEEN '${startDate}' AND '${endDate}'
    ORDER BY metrics.cost_micros DESC
  `);

  return rows.map((row: any) => ({
    campaignName: row.campaign.name,
    adGroupName: row.ad_group.name,
    impressions: Number(row.metrics.impressions),
    clicks: Number(row.metrics.clicks),
    cost: Number(row.metrics.cost_micros) / 1_000_000,
    conversions: Number(row.metrics.conversions),
    ctr: Number(row.metrics.ctr),
    cpc: Number(row.metrics.average_cpc) / 1_000_000,
    conversionRate:
      Number(row.metrics.clicks) > 0
        ? Number(row.metrics.conversions) / Number(row.metrics.clicks)
        : 0,
    date: row.segments.date,
  }));
}
```

```typescript
// google-ads-conversion.ts
// Google Ads 轉換追蹤 — 前端 + 後端

// ============================
// 前端：gtag 轉換追蹤
// ============================

/** 初始化 Google Ads 追蹤碼（放在 <head> 裡） */
function initGoogleAdsTag(conversionId: string): string {
  return `
<!-- Google Ads Tag -->
<script async src="https://www.googletagmanager.com/gtag/js?id=${conversionId}"></script>
<script>
  window.dataLayer = window.dataLayer || [];
  function gtag(){dataLayer.push(arguments);}
  gtag('js', new Date());
  gtag('config', '${conversionId}');
</script>`;
}

/** 發送轉換事件（結帳完成頁呼叫） */
function sendConversion(
  conversionId: string,
  conversionLabel: string,
  value: number,
  currency: string = 'TWD',
  transactionId?: string
): void {
  if (typeof window !== 'undefined' && (window as any).gtag) {
    (window as any).gtag('event', 'conversion', {
      send_to: `${conversionId}/${conversionLabel}`,
      value,
      currency,
      transaction_id: transactionId,
    });
  }
}

// ============================
// 後端：Google Ads Offline Conversion
// ============================

import { GoogleAdsApi } from 'google-ads-api';

interface OfflineConversion {
  gclid: string;           // Google Click ID（從 URL 取得）
  conversionAction: string; // 轉換動作 resource name
  conversionDateTime: string; // ISO 格式
  conversionValue: number;
  currencyCode: string;
}

async function uploadOfflineConversion(
  client: GoogleAdsApi,
  customerId: string,
  refreshToken: string,
  conversions: OfflineConversion[]
): Promise<void> {
  const customer = client.Customer({
    customer_id: customerId,
    refresh_token: refreshToken,
  });

  await customer.conversionUploads.uploadClickConversions({
    customer_id: customerId,
    conversions: conversions.map((c) => ({
      gclid: c.gclid,
      conversion_action: c.conversionAction,
      conversion_date_time: c.conversionDateTime,
      conversion_value: c.conversionValue,
      currency_code: c.currencyCode,
    })),
    partial_failure: true,
  });
}
```

## 3. LINE LAP 廣告投放

### 3.1 LINE LAP 架構與 API 操作

```typescript
// line-lap.ts
// LINE Ads Platform 廣告投放管理

import axios, { AxiosInstance } from 'axios';

// ============================
// 1. LINE LAP API 客戶端
// ============================

interface LineLapConfig {
  accessToken: string;
  adAccountId: string;
}

class LineLapClient {
  private http: AxiosInstance;
  private adAccountId: string;

  constructor(config: LineLapConfig) {
    this.adAccountId = config.adAccountId;
    this.http = axios.create({
      baseURL: 'https://ads.line.me/api/v3',
      headers: {
        Authorization: `Bearer ${config.accessToken}`,
        'Content-Type': 'application/json',
      },
    });
  }

  // ============================
  // 2. 廣告群組（Campaign）管理
  // ============================

  async listCampaigns(): Promise<any[]> {
    const res = await this.http.get(`/adaccounts/${this.adAccountId}/campaigns`);
    return res.data.campaigns ?? [];
  }

  async createCampaign(params: {
    name: string;
    objective: 'WEBSITE_VISIT' | 'WEBSITE_CONVERSION' | 'APP_INSTALL' | 'VIDEO_VIEW';
    dailyBudget: number;
    startDate: string; // YYYYMMDD
    endDate?: string;
  }): Promise<any> {
    const res = await this.http.post(
      `/adaccounts/${this.adAccountId}/campaigns`,
      {
        name: params.name,
        campaignObjective: params.objective,
        budgetMode: 'DAILY',
        dailyBudgetAmount: params.dailyBudget,
        startDate: params.startDate,
        endDate: params.endDate,
        status: 'PAUSED',
      }
    );
    return res.data;
  }

  // ============================
  // 3. 廣告組合（Ad Group）管理
  // ============================

  async createAdGroup(
    campaignId: string,
    params: {
      name: string;
      bidAmount: number;
      targetingAge?: { min: number; max: number };
      targetingGender?: 'MALE' | 'FEMALE' | 'ALL';
      targetingRegions?: string[]; // 台灣地區代碼
      targetingInterests?: string[];
      targetingAudienceIds?: string[]; // 自訂受眾
    }
  ): Promise<any> {
    const targeting: any = {};

    if (params.targetingAge) {
      targeting.ageGroup = {
        minAge: params.targetingAge.min,
        maxAge: params.targetingAge.max,
      };
    }
    if (params.targetingGender && params.targetingGender !== 'ALL') {
      targeting.gender = params.targetingGender;
    }
    if (params.targetingRegions) {
      targeting.regionTarget = { regions: params.targetingRegions };
    }
    if (params.targetingInterests) {
      targeting.interestTarget = { interests: params.targetingInterests };
    }
    if (params.targetingAudienceIds) {
      targeting.audienceTarget = { audienceIds: params.targetingAudienceIds };
    }

    const res = await this.http.post(
      `/adaccounts/${this.adAccountId}/campaigns/${campaignId}/adgroups`,
      {
        name: params.name,
        bidMethod: 'CPC',
        bidAmount: params.bidAmount,
        targeting,
        status: 'PAUSED',
      }
    );
    return res.data;
  }

  // ============================
  // 4. 成效報表
  // ============================

  async getReport(params: {
    granularity: 'DAILY' | 'HOURLY' | 'TOTAL';
    startDate: string;
    endDate: string;
    level: 'CAMPAIGN' | 'ADGROUP' | 'AD';
  }): Promise<any> {
    const res = await this.http.get(
      `/adaccounts/${this.adAccountId}/reports`,
      {
        params: {
          granularity: params.granularity,
          startDate: params.startDate,
          endDate: params.endDate,
          reportLevel: params.level,
        },
      }
    );
    return res.data;
  }
}

// ============================
// 5. LINE Tag（追蹤碼）安裝
// ============================

/** 產生 LINE Tag 初始化 HTML */
function generateLineTag(tagId: string): string {
  return `
<!-- LINE Tag Code -->
<script>
(function(g,d,o){
  g._ltq=g._ltq||[];g._lt=g._lt||function(){g._ltq.push(arguments)};
  var h=d.getElementsByTagName(o)[0];
  var s=d.createElement(o);s.async=1;
  s.src='https://s.yimg.jp/images/listing/tool/cv/ytag.js';
  h.parentNode.insertBefore(s,h);
})(window,document,'script');
_lt('init', {
  customerType: 'lap',
  tagId: '${tagId}'
});
_lt('send', 'pv', ['${tagId}']);
</script>
<noscript>
<img height="1" width="1" style="display:none"
src="https://tr.line.me/tag.gif?c_t=lap&t_id=${tagId}&e=pv&noscript=1"/>
</noscript>

}

/** LINE Tag 轉換事件 */
function sendLineConversion(tagId: string, eventType: string, value?: number): void {
  if (typeof window !== 'undefined' && (window as any)._lt) {
    const params: any = {};
    if (value !== undefined) params.value = value;
    (window as any)._lt('send', eventType, [tagId], params);
  }
}
```

## 4. GA4 電商追蹤設定

### 4.1 GA4 電商事件完整實作

```typescript
// ga4-ecommerce.ts
// GA4 增強型電子商務追蹤（Enhanced Ecommerce）

declare const gtag: (...args: any[]) => void;

// ============================
// 1. 商品資料型別
// ============================

interface GA4Item {
  item_id: string;
  item_name: string;
  item_brand?: string;
  item_category?: string;
  item_category2?: string;
  item_variant?: string;
  price: number;
  quantity?: number;
  coupon?: string;
  discount?: number;
  index?: number;
}

// ============================
// 2. 商品列表曝光（view_item_list）
// ============================

function trackViewItemList(
  listId: string,
  listName: string,
  items: GA4Item[]
): void {
  gtag('event', 'view_item_list', {
    item_list_id: listId,
    item_list_name: listName,
    items: items.map((item, index) => ({
      ...item,
      index: item.index ?? index,
    })),
  });
}

// ============================
// 3. 商品點擊（select_item）
// ============================

function trackSelectItem(
  listId: string,
  listName: string,
  item: GA4Item
): void {
  gtag('event', 'select_item', {
    item_list_id: listId,
    item_list_name: listName,
    items: [item],
  });
}

// ============================
// 4. 瀏覽商品（view_item）
// ============================

function trackViewItem(item: GA4Item, value: number, currency: string = 'TWD'): void {
  gtag('event', 'view_item', {
    currency,
    value,
    items: [item],
  });
}

// ============================
// 5. 加入購物車（add_to_cart）
// ============================

function trackGA4AddToCart(
  items: GA4Item[],
  value: number,
```

## 6. 廣告 ROI 計算與優化

### 6.1 核心指標計算

```typescript
// roi-calculator.ts
// 廣告 ROI 計算引擎

// ============================
// 1. 型別定義
// ============================

interface AdSpend {
  platform: 'google' | 'facebook' | 'line' | 'other';
  campaignName: string;
  spend: number;         // 花費（TWD）
  impressions: number;   // 曝光數
  clicks: number;        // 點擊數
  conversions: number;   // 轉換數
  revenue: number;       // 營收（TWD）
  period: string;        // YYYY-MM
}

interface ROIReport {
  platform: string;
  campaignName: string;
  period: string;
  spend: number;
  revenue: number;
  profit: number;
  roas: number;          // Return on Ad Spend
  roi: number;           // Return on Investment (%)
  ctr: number;           // Click-Through Rate (%)
  cpc: number;           // Cost per Click
  cpa: number;           // Cost per Acquisition
  conversionRate: number; // Conversion Rate (%)
  aov: number;           // Average Order Value
  ltv: number;           // 預估客戶終身價值
}

// ============================
// 2. 核心計算函數
// ============================

function calculateROI(data: AdSpend, avgOrderValue?: number): ROIReport {
  const profit = data.revenue - data.spend;
  const roas = data.spend > 0 ? data.revenue / data.spend : 0;
  const roi = data.spend > 0 ? (profit / data.spend) * 100 : 0;
  const ctr = data.impressions > 0 ? (data.clicks / data.impressions) * 100 : 0;
  const cpc = data.clicks > 0 ? data.spend / data.clicks : 0;
  const cpa = data.conversions > 0 ? data.spend / data.conversions : 0;
  const conversionRate = data.clicks > 0
    ? (data.conversions / data.clicks) * 100
    : 0;
  const aov = data.conversions > 0
    ? data.revenue / data.conversions
    : avgOrderValue ?? 0;

  // 簡化 LTV 估算：AOV x 平均回購次數（假設 3 次）x 平均留存期（假設 12 個月）
  const estimatedRepurchaseRate = 3;
  const ltv = aov * estimatedRepurchaseRate;

  return {
    platform: data.platform,
    campaignName: data.campaignName,
    period: data.period,
    spend: Math.round(data.spend),
    revenue: Math.round(data.revenue),
    profit: Math.round(profit),
    roas: Math.round(roas * 100) / 100,
    roi: Math.round(roi * 100) / 100,
    ctr: Math.round(ctr * 100) / 100,
    cpc: Math.round(cpc * 100) / 100,
    cpa: Math.round(cpa),
    conversionRate: Math.round(conversionRate * 100) / 100,
    aov: Math.round(aov),
    ltv: Math.round(ltv),
  };
}

// ============================
// 3. 多平台匯總
// ============================

interface PlatformSummary {
  platform: string;
  totalSpend: number;
  totalRevenue: number;
  totalProfit: number;
  avgRoas: number;
  avgCpa: number;
  totalConversions: number;
}

function summarizeByPlatform(reports: ROIReport[]): PlatformSummary[] {
  const grouped = new Map<string, ROIReport[]>();
  for (const r of reports) {
    const existing = grouped.get(r.platform) ?? [];
    existing.push(r);
    grouped.set(r.platform, existing);
  }

  const summaries: PlatformSummary[] = [];
  for (const [platform, items] of grouped) {
    const totalSpend = items.reduce((s, i) => s + i.spend, 0);
    const totalRevenue = items.reduce((s, i) => s + i.revenue, 0);
    const totalConversions = items.reduce(
      (s, i) => s + (i.spend > 0 && i.cpa > 0 ? Math.round(i.spend / i.cpa) : 0),
      0
    );

    summaries.push({
      platform,
      totalSpend,
      totalRevenue,
      totalProfit: totalRevenue - totalSpend,
      avgRoas: totalSpend > 0 ? totalRevenue / totalSpend : 0,
      avgCpa: totalConversions > 0 ? totalSpend / totalConversions : 0,
      totalConversions,
    });
  }

  return summaries.sort((a, b) => b.avgRoas - a.avgRoas);
}

// ============================
// 4. 預算分配建議
// ============================

interface BudgetAllocation {
  platform: string;
  currentBudget: number;
  suggestedBudget: number;
  reason: string;
}

function suggestBudgetAllocation(
  summaries: PlatformSummary[],
  totalBudget: number
): BudgetAllocation[] {
  // 以 ROAS 為權重分配預算
  const totalWeight = summaries.reduce(
    (s, p) => s + Math.max(p.avgRoas, 0.1),
    0
  );

  return summaries.map((s) => {
    const weight = Math.max(s.avgRoas, 0.1) / totalWeight;
    const suggested = Math.round(totalBudget * weight);

    let reason: string;
    if (s.avgRoas >= 5) {
      reason = `ROAS ${s.avgRoas.toFixed(1)}x 表現優異，建議加碼`;
    } else if (s.avgRoas >= 2) {
      reason = `ROAS ${s.avgRoas.toFixed(1)}x 穩定獲利，維持投放`;
    } else if (s.avgRoas >= 1) {
      reason = `ROAS ${s.avgRoas.toFixed(1)}x 接近打平，需優化素材與受眾`;
    } else {
      reason = `ROAS ${s.avgRoas.toFixed(1)}x 虧損中，建議縮減或暫停`;
    }

    return {
      platform: s.platform,
      currentBudget: s.totalSpend,
      suggestedBudget: suggested,
      reason,
    };
  });
}

// ============================
// 5. ROAS 達標警報
// ============================

interface AlertRule {
  platform: string;
  campaignPattern?: string;  // 正則匹配活動名稱
  minRoas: number;           // 最低 ROAS
  maxCpa: number;            // 最高 CPA
}

interface Alert {
  level: 'warning' | 'critical';
  message: string;
  campaign: string;
  platform: string;
  metric: string;
  currentValue: number;
  threshold: number;
}

function checkAlerts(reports: ROIReport[], rules: AlertRule[]): Alert[] {
  const alerts: Alert[] = [];

  for (const report of reports) {
    for (const rule of rules) {
      if (rule.platform !== report.platform) continue;
      if (
        rule.campaignPattern &&
        !new RegExp(rule.campaignPattern).test(report.campaignName)
      )
        continue;

      // ROAS 低於門檻
      if (report.roas < rule.minRoas && report.spend > 0) {
        alerts.push({
          level: report.roas < rule.minRoas * 0.5 ? 'critical' : 'warning',
          message: `ROAS ${report.roas.toFixed(2)}x 低於目標 ${rule.minRoas}x`,
          campaign: report.campaignName,
          platform: report.platform,
          metric: 'ROAS',
          currentValue: report.roas,
          threshold: rule.minRoas,
        });
      }

      // CPA 超過門檻
      if (report.cpa > rule.maxCpa && report.cpa > 0) {
        alerts.push({
          level: report.cpa > rule.maxCpa * 2 ? 'critical' : 'warning',
          message: `CPA $${report.cpa} 超過上限 $${rule.maxCpa}`,
          campaign: report.campaignName,
          platform: report.platform,
          metric: 'CPA',
          currentValue: report.cpa,
          threshold: rule.maxCpa,
        });
      }
    }
  }

  return alerts.sort((a, b) =>
    a.level === 'critical' && b.level !== 'critical' ? -1 : 1
  );
}
```

```typescript
// ============================
// 2. 流量分配（deterministic hash）
// ============================

/** 根據用戶 ID 或 cookie 確定性分配變體 */
function assignVariant(
  userId: string,
  testId: string,
  variants: Variant[]
): Variant {
  const hash = crypto
    .createHash('md5')
    .update(`${testId}:${userId}`)
    .digest('hex');
  // 取前 8 位 hex 轉數字，映射到 0-100
  const bucket = parseInt(hash.substring(0, 8), 16) % 100;

  let cumulative = 0;
  for (const variant of variants) {
    cumulative += variant.weight;
    if (bucket < cumulative) {
      return variant;
    }
  }
  return variants[variants.length - 1];
}

// ============================
// 3. 統計顯著性計算（Z-test）
// ============================

/** 標準常態分布累積分布函數近似 */
function normalCDF(z: number): number {
  const a1 = 0.254829592;
  const a2 = -0.284496736;
  const a3 = 1.421413741;
  const a4 = -1.453152027;
  const a5 = 1.061405429;
  const p = 0.3275911;
  const sign = z < 0 ? -1 : 1;
  const x = Math.abs(z) / Math.SQRT2;
  const t = 1.0 / (1.0 + p * x);
  const y =
    1.0 - ((((a5 * t + a4) * t + a3) * t + a2) * t + a1) * t * Math.exp(-x * x);
  return 0.5 * (1.0 + sign * y);
}

/** 計算兩組轉換率的 Z-test 信賴水準 */
function calculateSignificance(
  controlVisitors: number,
  controlConversions: number,
  testVisitors: number,
  testConversions: number
): { zScore: number; pValue: number; confidence: number; isSignificant: boolean } {
  const p1 = controlVisitors > 0 ? controlConversions / controlVisitors : 0;
  const p2 = testVisitors > 0 ? testConversions / testVisitors : 0;
  const pPooled =
    (controlConversions + testConversions) /
    (controlVisitors + testVisitors);

  if (pPooled === 0 || pPooled === 1) {
    return { zScore: 0, pValue: 1, confidence: 0, isSignificant: false };
  }

  const se = Math.sqrt(
    pPooled * (1 - pPooled) * (1 / controlVisitors + 1 / testVisitors)
  );

  const zScore = se > 0 ? (p2 - p1) / se : 0;
  const pValue = 2 * (1 - normalCDF(Math.abs(zScore))); // 雙尾檢定
  const confidence = (1 - pValue) * 100;

  return {
    zScore: Math.round(zScore * 1000) / 1000,
    pValue: Math.round(pValue * 10000) / 10000,
    confidence: Math.round(confidence * 100) / 100,
    isSignificant: pValue < 0.05, // 95% 信賴水準
  };
}

// ============================
// 4. 最少樣本數計算
// ============================

/** 計算達到統計顯著性所需的最少樣本數 */
function calculateMinSampleSize(
  baselineRate: number,     // 原始轉換率（例如 0.03 = 3%）
  minimumDetectableEffect: number, // 最小可偵測效果（例如 0.2 = 提升 20%）
  confidenceLevel: number = 0.95,
  power: number = 0.8
): number {
  // Z-scores
  const zAlpha = confidenceLevel === 0.95 ? 1.96 : confidenceLevel === 0.99 ? 2.576 : 1.645;
  const zBeta = power === 0.8 ? 0.842 : power === 0.9 ? 1.282 : 0.674;

  const p1 = baselineRate;
  const p2 = baselineRate * (1 + minimumDetectableEffect);
  const pAvg = (p1 + p2) / 2;

  const numerator = Math.pow(
    zAlpha * Math.sqrt(2 * pAvg * (1 - pAvg)) +
      zBeta * Math.sqrt(p1 * (1 - p1) + p2 * (1 - p2)),
    2
  );
  const denominator = Math.pow(p2 - p1, 2);

  return Math.ceil(numerator / denominator);
}

import { Router, Request, Response } from 'express';

function createABTestRouter(): Router {
  const router = Router();

  // 記憶體儲存（正式環境用 DB）
  const tests = new Map<string, ABTestConfig>();
  const events = new Map<string, Array<{ variantId: string; type: 'visit' | 'conversion'; revenue?: number; timestamp: number }>>();

  // 建立測試
  router.post('/tests', (req: Request, res: Response) => {
    const config: ABTestConfig = req.body;
    tests.set(config.testId, config);
    events.set(config.testId, []);
    res.json({ success: true, testId: config.testId });
  });

  // 取得分配（前端呼叫）
  router.get('/assign/:testId', (req: Request, res: Response) => {
    const test = tests.get(req.params.testId);
    if (!test || test.status !== 'active') {
      res.json({ variant: null });
      return;
    }

    const userId =
      (req.cookies?.ab_uid as string) ??
      (req.query.uid as string) ??
      crypto.randomUUID();

    const variant = assignVariant(userId, test.testId, test.variants);

    // 記錄 visit
    const evts = events.get(test.testId) ?? [];
    evts.push({
      variantId: variant.id,
      type: 'visit',
      timestamp: Date.now(),
    });

    res.json({ variant, userId });
  });

  // 記錄轉換
  router.post('/convert/:testId', (req: Request, res: Response) => {
    const { variantId, revenue } = req.body;
    const evts = events.get(req.params.testId) ?? [];
    evts.push({
      variantId,
      type: 'conversion',
      revenue,
      timestamp: Date.now(),
    });
    res.json({ success: true });
  });

  // 取得結果
  router.get('/results/:testId', (req: Request, res: Response) => {
    const test = tests.get(req.params.testId);
    const evts = events.get(req.params.testId) ?? [];
    if (!test) {
      res.status(404).json({ error: 'Test not found' });
      return;
    }

    const results: TestResult[] = test.variants.map((variant) => {
      const variantEvents = evts.filter((e) => e.variantId === variant.id);
      const visitors = variantEvents.filter((e) => e.type === 'visit').length;
      const conversions = variantEvents.filter(
        (e) => e.type === 'conversion'
      ).length;
      const revenue = variantEvents
        .filter((e) => e.type === 'conversion')
        .reduce((s, e) => s + (e.revenue ?? 0), 0);

      return {
        variantId: variant.id,
        variantName: variant.name,
        visitors,
        conversions,
        conversionRate: visitors > 0 ? conversions / visitors : 0,
        revenue,
        revenuePerVisitor: visitors > 0 ? revenue / visitors : 0,
        improvement: 0,
        isSignificant: false,
        confidence: 0,
      };
    });

    // 計算相對於第一組（對照組）的提升 & 顯著性
    if (results.length >= 2) {
      const control = results[0];
      for (let i = 1; i < results.length; i++) {
        const test = results[i];
        if (control.conversionRate > 0) {
          test.improvement =
            ((test.conversionRate - control.conversionRate) /
              control.conversionRate) *
            100;
        }
        const sig = calculateSignificance(
          control.visitors,
          control.conversions,
          test.visitors,
          test.conversions
        );
        test.isSignificant = sig.isSignificant;
        test.confidence = sig.confidence;
      }
    }

    res.json({ testId: test.testId, testName: test.testName, results });
  });

  return router;
}
```

```typescript
function parseDateRange(range: string): {
  startDate: string;
  endDate: string;
} {
  const now = new Date();
  switch (range) {
    case 'yesterday': {
      const yesterday = new Date(now);
      yesterday.setDate(yesterday.getDate() - 1);
      const d = yesterday.toISOString().split('T')[0];
      return { startDate: d, endDate: d };
    }
    case 'last7days': {
      const end = new Date(now);
      end.setDate(end.getDate() - 1);
      const start = new Date(end);
      start.setDate(start.getDate() - 6);
      return {
        startDate: start.toISOString().split('T')[0],
        endDate: end.toISOString().split('T')[0],
      };
    }
    case 'last30days': {
      const end = new Date(now);
      end.setDate(end.getDate() - 1);
      const start = new Date(end);
      start.setDate(start.getDate() - 29);
      return {
        startDate: start.toISOString().split('T')[0],
        endDate: end.toISOString().split('T')[0],
      };
    }
    default:
      return {
        startDate: now.toISOString().split('T')[0],
        endDate: now.toISOString().split('T')[0],
      };
  }
}
```
