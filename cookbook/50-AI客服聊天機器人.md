### 3.2 RAG Pipeline `src/rag-pipeline.ts`

```typescript
// src/rag-pipeline.ts

import { ChatMessage, KnowledgeChunk, RAGResult } from './types';
import { searchKnowledge } from './knowledge-store';
import { callLLM } from './llm-client';
import { detectLanguage } from './i18n';

/**
 * RAG Pipeline：Query Rewrite → 向量搜尋 → Rerank → Prompt 組裝 → LLM 生成
 */
export async function ragPipeline(
  userQuery: string,
  conversationHistory: ChatMessage[],
  language: string
): Promise<RAGResult> {
  // Step 1: Query 改寫 — 把對話上下文融入搜尋關鍵字
  const rewrittenQuery = await rewriteQuery(userQuery, conversationHistory);

  // Step 2: 向量搜尋知識庫
  const chunks = await searchKnowledge(rewrittenQuery, {
    topK: 8,
    language,
    minSimilarity: 0.35,
  });

  // Step 3: Rerank — 用交叉比對重新排序
  const rankedChunks = rerank(rewrittenQuery, chunks);

  // Step 4: 取前 4 筆最相關的結果
  const topChunks = rankedChunks.slice(0, 4);

  // Step 5: 組裝 Prompt
  const systemPrompt = buildSystemPrompt(topChunks, language);
  const messages = buildMessages(systemPrompt, conversationHistory, userQuery);

  // Step 6: 呼叫 LLM 生成回覆
  const llmResponse = await callLLM(messages);

  // Step 7: 計算信心分數
  const confidence = calculateConfidence(topChunks, llmResponse.content);

  return {
    answer: llmResponse.content,
    sources: topChunks,
    confidence,
    tokensUsed: llmResponse.tokensUsed,
    model: llmResponse.model,
  };
}

/**
 * Query 改寫：融合對話上下文，讓搜尋更精準
 */
async function rewriteQuery(
  query: string,
  history: ChatMessage[]
): Promise<string> {
  if (history.length === 0) return query;

  // 取最近 3 輪對話作為上下文
  const recentHistory = history.slice(-6);
  const contextSummary = recentHistory
    .map((m) => `${m.role}: ${m.content}`)
    .join('\n');

  const rewritePrompt = [
    {
      role: 'system' as const,
      content:
        '你是搜尋查詢改寫器。根據對話上下文，將使用者的最新問題改寫成一個獨立的搜尋查詢。只輸出改寫後的查詢，不要解釋。',
    },
    {
      role: 'user' as const,
      content: `對話上下文:\n${contextSummary}\n\n最新問題: ${query}\n\n改寫後的搜尋查詢:`,
    },
  ];

  try {
    const result = await callLLM(rewritePrompt, { maxTokens: 100 });
    return result.content.trim() || query;
  } catch {
    return query; // 改寫失敗就用原始查詢
  }
}

/**
 * Rerank：根據關鍵字重疊度 + 語義相似度 重新排序
 */
function rerank(query: string, chunks: KnowledgeChunk[]): KnowledgeChunk[] {
  const queryTokens = new Set(
    query
      .toLowerCase()
      .split(/[\s,，。.!！?？]+/)
      .filter(Boolean)
  );

  return chunks
    .map((chunk) => {
      const contentTokens = new Set(
        chunk.content
          .toLowerCase()
          .split(/[\s,，。.!！?？]+/)
          .filter(Boolean)
      );

      // 關鍵字重疊率
      let overlap = 0;
      for (const token of queryTokens) {
        if (contentTokens.has(token)) overlap++;
      }
      const keywordScore = queryTokens.size > 0 ? overlap / queryTokens.size : 0;

      // 綜合分數：向量相似度 60% + 關鍵字 40%
      const combinedScore = (chunk.similarity || 0) * 0.6 + keywordScore * 0.4;

      return { ...chunk, similarity: combinedScore };
    })
    .sort((a, b) => (b.similarity || 0) - (a.similarity || 0));
}

/**
 * 組裝 System Prompt
 */
function buildSystemPrompt(chunks: KnowledgeChunk[], language: string): string {
  const knowledgeContext = chunks
    .map((c, i) => `[來源 ${i + 1}] ${c.metadata.title || c.metadata.source}\n${c.content}`)
    .join('\n\n---\n\n');

  const langInstruction: Record<string, string> = {
    'zh-TW': '請用繁體中文回覆。',
    'zh-CN': '请用简体中文回复。',
    en: 'Please reply in English.',
    ja: '日本語で回答してください。',
    ko: '한국어로 답변해 주세요.',
  };

  return `你是一位專業的 AI 客服助理。請根據以下知識庫內容回答使用者問題。

規則：
1. 只根據知識庫內容回答，不要編造資訊。
2. 如果知識庫沒有相關資訊，誠實告知使用者並建議轉接人工客服。
3. 回覆要簡潔、友善、專業。
4. 如果問題模糊，主動追問釐清。
5. ${langInstruction[language] || langInstruction['zh-TW']}

知識庫內容：
${knowledgeContext}`;
}

/**
 * 組裝完整訊息陣列
 */
function buildMessages(
  systemPrompt: string,
  history: ChatMessage[],
  currentQuery: string
): Array<{ role: 'system' | 'user' | 'assistant'; content: string }> {
  const messages: Array<{ role: 'system' | 'user' | 'assistant'; content: string }> = [
    { role: 'system', content: systemPrompt },
  ];

  // 最近 10 則對話
  const recentHistory = history.slice(-10);
  for (const msg of recentHistory) {
    messages.push({ role: msg.role, content: msg.content });
  }

  messages.push({ role: 'user', content: currentQuery });

  return messages;
}

/**
 * 計算回覆信心分數（0~1）
 */
function calculateConfidence(chunks: KnowledgeChunk[], answer: string): number {
  if (chunks.length === 0) return 0.1;

  // 因子 1：最高相似度
  const maxSimilarity = Math.max(...chunks.map((c) => c.similarity || 0));

  // 因子 2：來源數量（越多來源支撐越可靠）
  const sourceFactor = Math.min(chunks.length / 4, 1);

  // 因子 3：回覆中是否包含「不確定」「無法」等低信心關鍵字
  const uncertainKeywords = ['不確定', '無法確認', '建議聯繫', '沒有相關', 'not sure', 'I cannot'];
  const hasUncertainty = uncertainKeywords.some((kw) =>
    answer.toLowerCase().includes(kw.toLowerCase())
  );
  const certaintyFactor = hasUncertainty ? 0.3 : 1.0;

  const confidence = maxSimilarity * 0.5 + sourceFactor * 0.3 + certaintyFactor * 0.2;
  return Math.round(confidence * 100) / 100;
}
```

## 4. OpenAI / Gemini API 串接

### 4.1 LLM Client `src/llm-client.ts`

```typescript
// src/llm-client.ts

import OpenAI from 'openai';
import { GoogleGenerativeAI } from '@google/generative-ai';

// ---- 初始化 ----

const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });

const genAI = new GoogleGenerativeAI(process.env.GEMINI_API_KEY || '');

interface LLMResponse {
  content: string;
  tokensUsed: number;
  model: string;
}

interface LLMOptions {
  maxTokens?: number;
  temperature?: number;
  provider?: 'openai' | 'gemini';
}

/**
 * 統一 LLM 呼叫介面：OpenAI 為主，Gemini 為備援
 */
export async function callLLM(
  messages: Array<{ role: 'system' | 'user' | 'assistant'; content: string }>,
  options: LLMOptions = {}
): Promise<LLMResponse> {
  const { maxTokens = 1024, temperature = 0.3, provider } = options;

  // 明確指定 provider 就用指定的，否則預設 openai，失敗 fallback gemini
  if (provider === 'gemini') {
    return callGemini(messages, maxTokens, temperature);
  }

  try {
    return await callOpenAI(messages, maxTokens, temperature);
  } catch (error) {
    console.warn('[LLM] OpenAI failed, falling back to Gemini:', (error as Error).message);
    return callGemini(messages, maxTokens, temperature);
  }
}

/**
 * OpenAI Chat Completion
 */
async function callOpenAI(
  messages: Array<{ role: 'system' | 'user' | 'assistant'; content: string }>,
  maxTokens: number,
  temperature: number
): Promise<LLMResponse> {
  const model = process.env.OPENAI_MODEL || 'gpt-4o';

  const response = await openai.chat.completions.create({
    model,
    messages,
    max_tokens: maxTokens,
    temperature,
  });

  const choice = response.choices[0];
  return {
    content: choice.message.content || '',
    tokensUsed: response.usage?.total_tokens || 0,
    model,
  };
}

/**
 * Gemini Chat
 */
async function callGemini(
  messages: Array<{ role: 'system' | 'user' | 'assistant'; content: string }>,
  maxTokens: number,
  temperature: number
): Promise<LLMResponse> {
  const modelName = process.env.GEMINI_MODEL || 'gemini-2.5-flash';
  const model = genAI.getGenerativeModel({
    model: modelName,
    generationConfig: { maxOutputTokens: maxTokens, temperature },
  });

  // 轉換訊息格式：Gemini 用 system instruction + contents
  const systemMsg = messages.find((m) => m.role === 'system');
  const chatMessages = messages.filter((m) => m.role !== 'system');

  const chat = model.startChat({
    systemInstruction: systemMsg ? { role: 'user', parts: [{ text: systemMsg.content }] } : undefined,
    history: chatMessages.slice(0, -1).map((m) => ({
      role: m.role === 'assistant' ? 'model' : 'user',
      parts: [{ text: m.content }],
    })),
  });

  const lastMessage = chatMessages[chatMessages.length - 1];
  const result = await chat.sendMessage(lastMessage.content);
  const responseText = result.response.text();

  return {
    content: responseText,
    tokensUsed: responseText.length, // Gemini 不直接回傳 token 數
    model: modelName,
  };
}

// ---- Embedding ----

/**
 * OpenAI Embedding
 */
export async function embedText(text: string): Promise<number[]> {
  const model = process.env.OPENAI_EMBEDDING_MODEL || 'text-embedding-3-small';

  try {
    const response = await openai.embeddings.create({
      model,
      input: text.slice(0, 8000), // 截斷過長文字
    });
    return response.data[0].embedding;
  } catch (error) {
    console.warn('[Embed] OpenAI failed, trying Gemini:', (error as Error).message);
    return embedTextGemini(text);
  }
}

/**
 * Gemini Embedding（備援）
 */
async function embedTextGemini(text: string): Promise<number[]> {
  const model = genAI.getGenerativeModel({ model: 'text-embedding-004' });
  const result = await model.embedContent(text.slice(0, 8000));
  return result.embedding.values;
}
```

## 5. 知識庫建置與向量搜尋

### 5.1 Supabase 資料表（SQL）

```sql
-- 啟用 pgvector 擴充
create extension if not exists vector;

-- 知識庫 chunks
create table knowledge_chunks (
  id uuid primary key default gen_random_uuid(),
  content text not null,
  embedding vector(1536),   -- OpenAI text-embedding-3-small 維度
  metadata jsonb default '{}',
  category text default 'general',
  language text default 'zh-TW',
  source text,
  title text,
  created_at timestamptz default now(),
  updated_at timestamptz default now()
);

-- 向量搜尋索引（IVFFlat，資料超過 1 萬筆時效能更好）
create index on knowledge_chunks
  using ivfflat (embedding vector_cosine_ops) with (lists = 100);

-- 全文搜尋索引
create index idx_knowledge_content_gin on knowledge_chunks
  using gin (to_tsvector('simple', content));

-- 分類索引
create index idx_knowledge_category on knowledge_chunks (category);
create index idx_knowledge_language on knowledge_chunks (language);

-- 向量搜尋函數
create or replace function match_knowledge(
  query_embedding vector(1536),
  match_count int default 5,
  match_threshold float default 0.35,
  filter_language text default null,
  filter_category text default null
)
returns table (
  id uuid,
  content text,
  metadata jsonb,
  similarity float
)
language plpgsql
as $$
begin
  return query
  select
    kc.id,
    kc.content,
    kc.metadata,
    1 - (kc.embedding <=> query_embedding) as similarity
  from knowledge_chunks kc
  where 1 - (kc.embedding <=> query_embedding) > match_threshold
    and (filter_language is null or kc.language = filter_language)
    and (filter_category is null or kc.category = filter_category)
  order by kc.embedding <=> query_embedding
  limit match_count;
end;
$$;

-- 對話紀錄
create table conversations (
  id uuid primary key default gen_random_uuid(),
  session_id text not null,
  platform text not null,
  user_id text not null,
  messages jsonb default '[]',
  language text default 'zh-TW',
  metadata jsonb default '{}',
  created_at timestamptz default now(),
  updated_at timestamptz default now()
);

create index idx_conversations_session on conversations (session_id);
create index idx_conversations_user on conversations (user_id, platform);

-- 轉人工紀錄
create table escalations (
  id uuid primary key default gen_random_uuid(),
  session_id text not null,
  user_id text not null,
  platform text not null,
  reason text,
  conversation_snapshot jsonb,
  status text default 'pending', -- pending / assigned / resolved
  assigned_to text,
  resolved_at timestamptz,
  created_at timestamptz default now()
);

create index idx_escalations_status on escalations (status);
```

### 5.2 知識庫存取 `src/knowledge-store.ts`

```typescript
// src/knowledge-store.ts

import { createClient } from '@supabase/supabase-js';
import { KnowledgeChunk } from './types';
import { embedText } from './llm-client';

const supabase = createClient(
  process.env.SUPABASE_URL || '',
  process.env.SUPABASE_SERVICE_KEY || ''
);

interface SearchOptions {
  topK?: number;
  language?: string;
  category?: string;
  minSimilarity?: number;
}

/**
 * 向量搜尋知識庫
 */
export async function searchKnowledge(
  query: string,
  options: SearchOptions = {}
): Promise<KnowledgeChunk[]> {
  const { topK = 5, language, category, minSimilarity = 0.35 } = options;

  // 1. 取得查詢向量
  const queryEmbedding = await embedText(query);

  // 2. 呼叫 Supabase RPC 搜尋
  const { data, error } = await supabase.rpc('match_knowledge', {
    query_embedding: queryEmbedding,
    match_count: topK,
    match_threshold: minSimilarity,
    filter_language: language || null,
    filter_category: category || null,
  });

  if (error) {
    console.error('[Knowledge] Search error:', error.message);
    return [];
  }

  return (data || []).map((row: any) => ({
    id: row.id,
    content: row.content,
    metadata: row.metadata || {},
    similarity: row.similarity,
  }));
}

/**
 * 新增知識到知識庫
 */
export async function addKnowledge(
  content: string,
  metadata: {
    source: string;
    category?: string;
    language?: string;
    title?: string;
    url?: string;
  }
): Promise<string | null> {
  const embedding = await embedText(content);

  const { data, error } = await supabase
    .from('knowledge_chunks')
    .insert({
      content,
      embedding,
      metadata: {
        source: metadata.source,
        title: metadata.title || '',
        url: metadata.url || '',
      },
      category: metadata.category || 'general',
      language: metadata.language || 'zh-TW',
      source: metadata.source,
      title: metadata.title || '',
    })
    .select('id')
    .single();

  if (error) {
    console.error('[Knowledge] Insert error:', error.message);
    return null;
  }

  return data.id;
}

/**
 * 批次匯入知識（從 JSON 陣列）
 */
export async function batchImportKnowledge(
  items: Array<{
    content: string;
    source: string;
    category?: string;
    language?: string;
    title?: string;
  }>,
  concurrency: number = 3
): Promise<{ success: number; failed: number }> {
  let success = 0;
  let failed = 0;

  // 併發控制
  for (let i = 0; i < items.length; i += concurrency) {
    const batch = items.slice(i, i + concurrency);
    const results = await Promise.allSettled(
      batch.map((item) => addKnowledge(item.content, item))
    );

    for (const r of results) {
      if (r.status === 'fulfilled' && r.value) success++;
      else failed++;
    }

    console.log(`[BatchImport] Progress: ${i + batch.length}/${items.length}`);
  }

  return { success, failed };
}

/**
 * 文件切片器：將長文件切成適合 embedding 的 chunks
 */
export function splitDocument(
  text: string,
  options: { chunkSize?: number; overlap?: number } = {}
): string[] {
  const { chunkSize = 800, overlap = 100 } = options;
  const chunks: string[] = [];

  // 優先以段落為分界
  const paragraphs = text.split(/\n\n+/);
  let currentChunk = '';

  for (const para of paragraphs) {
    if (currentChunk.length + para.length > chunkSize && currentChunk.length > 0) {
      chunks.push(currentChunk.trim());
      // 保留 overlap
      const words = currentChunk.split('');
      currentChunk = words.slice(-overlap).join('') + '\n\n' + para;
    } else {
      currentChunk += (currentChunk ? '\n\n' : '') + para;
    }
  }

  if (currentChunk.trim()) {
    chunks.push(currentChunk.trim());
  }

  return chunks;
}
```

```typescript
/**
 * 取得或建立 session
 */
export async function getOrCreateSession(
  platform: 'line' | 'telegram' | 'web',
  userId: string
): Promise<ConversationSession> {
  const sessionKey = `session:${platform}:${userId}`;

  // 先查 Redis
  const cached = await redis.get(sessionKey);
  if (cached) {
    const session = JSON.parse(cached) as ConversationSession;
    session.lastActiveAt = Date.now();
    await redis.setex(sessionKey, SESSION_TTL, JSON.stringify(session));
    return session;
  }

  // Redis 沒有，查 Supabase（上次的對話）
  const { data } = await supabase
    .from('conversations')
    .select('*')
    .eq('user_id', userId)
    .eq('platform', platform)
    .order('updated_at', { ascending: false })
    .limit(1)
    .single();

  if (data && Date.now() - new Date(data.updated_at).getTime() < SESSION_TTL * 1000) {
    const session: ConversationSession = {
      sessionId: data.session_id,
      platform,
      userId,
      messages: data.messages || [],
      language: data.language || 'zh-TW',
      createdAt: new Date(data.created_at).getTime(),
      lastActiveAt: Date.now(),
      metadata: data.metadata || {},
    };
    await redis.setex(sessionKey, SESSION_TTL, JSON.stringify(session));
    return session;
  }

  // 建立新 session
  const session: ConversationSession = {
    sessionId: uuidv4(),
    platform,
    userId,
    messages: [],
    language: 'zh-TW',
    createdAt: Date.now(),
    lastActiveAt: Date.now(),
    metadata: {},
  };

  await redis.setex(sessionKey, SESSION_TTL, JSON.stringify(session));
  return session;
}

  userId: string,

  session.messages.push(message);

  if (session.messages.length > MAX_HISTORY) {
    session.messages = session.messages.slice(-MAX_HISTORY);

  session.lastActiveAt = Date.now();
  await redis.setex(sessionKey, SESSION_TTL, JSON.stringify(session));
}

  userId: string

  if (!cached) return;

  const session = JSON.parse(cached) as ConversationSession;

  await supabase.from('conversations').upsert(

      session_id: session.sessionId,
      platform: session.platform,
      user_id: session.userId,
      messages: session.messages,
      language: session.language,
```

```typescript
const CONFIDENCE_THRESHOLD = 0.4;       // 低於此值自動轉人工
const CONSECUTIVE_LOW_THRESHOLD = 3;     // 連續 N 次低信心也轉
const EXPLICIT_KEYWORDS = ['轉人工', '真人客服', '找人', 'human', 'agent', 'talk to human', 'real person'];

interface EscalationCheck {
  shouldEscalate: boolean;
  reason: string;
}

/**
 * 判斷是否需要轉人工
 */
export function checkEscalation(
  userMessage: string,
  confidence: number,
  session: ConversationSession
): EscalationCheck {
  // 1. 使用者明確要求轉人工
  const lowerMsg = userMessage.toLowerCase();
  for (const keyword of EXPLICIT_KEYWORDS) {
    if (lowerMsg.includes(keyword)) {
      return { shouldEscalate: true, reason: `使用者主動要求轉人工（關鍵字: ${keyword}）` };
    }
  }

  // 2. 單次信心分數過低
  if (confidence < CONFIDENCE_THRESHOLD) {
    return { shouldEscalate: true, reason: `AI 信心分數過低 (${confidence})` };
  }

  // 3. 連續低信心
  const recentLowConfidence = (session.metadata.consecutiveLowConfidence as number) || 0;
  if (recentLowConfidence >= CONSECUTIVE_LOW_THRESHOLD) {
    return { shouldEscalate: true, reason: `連續 ${recentLowConfidence} 次低信心回覆` };
  }

  return { shouldEscalate: false, reason: '' };
}

/**
```
```typescript
 * 統一訊息處理入口（所有平台共用）
 */
export async function handleIncomingMessage(msg: PlatformMessage): Promise<string> {
  try {
    // 1. 取得 / 建立 session
    const session = await getOrCreateSession(msg.platform, msg.userId);

    // 2. 偵測語言（第一則訊息時）
    if (session.messages.length === 0) {
      const detectedLang = detectLanguage(msg.text);
      await updateLanguage(msg.platform, msg.userId, detectedLang);
      session.language = detectedLang;
    }

    // 3. 檢查特殊指令
    const command = parseCommand(msg.text);
    if (command) {
      return await handleCommand(command, msg, session);
    }

    // 4. 記錄使用者訊息
    const userMessage: ChatMessage = {
      role: 'user',
      content: msg.text,
      timestamp: msg.timestamp,
    };
    await addMessage(msg.platform, msg.userId, userMessage);

    // 5. RAG Pipeline 生成回覆
    const result = await ragPipeline(
      msg.text,
      session.messages,
      session.language
    );

    // 6. 檢查是否需要轉人工
    const escalationCheck = checkEscalation(msg.text, result.confidence, session);

    if (escalationCheck.shouldEscalate) {
      await escalateToHuman(session, escalationCheck.reason, msg.text);
      await triggerEscalationWorkflow({
        sessionId: session.sessionId,
        userId: session.userId,
        platform: session.platform,
        reason: escalationCheck.reason,
        lastQuery: msg.text,
        conversationPreview: session.messages
          .slice(-5)
          .map((m) => `${m.role}: ${m.content}`)
          .join('\n'),
      });
      // 仍然回覆 AI 的答案，加上轉人工提示
      return result.answer + '\n\n' + t('transfer', session.language as SupportedLanguage);
    }

    // 7. 記錄 AI 回覆
    const assistantMessage: ChatMessage = {
      role: 'assistant',
      content: result.answer,
      timestamp: Date.now(),
    };
    await addMessage(msg.platform, msg.userId, assistantMessage);

    // 8. 更新連續低信心計數
    if (result.confidence < 0.5) {
      const count = ((session.metadata.consecutiveLowConfidence as number) || 0) + 1;
      session.metadata.consecutiveLowConfidence = count;
    } else {
      session.metadata.consecutiveLowConfidence = 0;
    }

    return result.answer;
  } catch (error) {
    console.error('[MessageHandler] Error:', error);
    const lang = detectLanguage(msg.text);
    return t('error', lang);
  }
}
```

### 12.2 Rate Limiter `src/rate-limiter.ts`

```typescript
// src/rate-limiter.ts

import Redis from 'ioredis';
import { Request, Response, NextFunction } from 'express';

const redis = new Redis(process.env.REDIS_URL || 'redis://localhost:6379');

interface RateLimitConfig {
  windowMs: number;   // 時間窗口（毫秒）
  maxRequests: number; // 最大請求數
  keyPrefix: string;
}

const defaultConfig: RateLimitConfig = {
  windowMs: 60 * 1000,  // 1 分鐘
  maxRequests: 20,       // 20 次
  keyPrefix: 'rl',
};

/**
 * 滑動窗口 Rate Limiter
 */
export function rateLimiter(config: Partial<RateLimitConfig> = {}) {
  const cfg = { ...defaultConfig, ...config };

  return async (req: Request, res: Response, next: NextFunction): Promise<void> => {
    // 從 header 或 body 取得使用者 ID
    const userId =
      (req.body?.events?.[0]?.source?.userId) ||           // LINE
      (req.body?.message?.from?.id) ||                      // Telegram
      req.headers['x-user-id'] ||                           // Web
      req.ip;

    const key = `${cfg.keyPrefix}:${userId}`;
    const now = Date.now();
    const windowStart = now - cfg.windowMs;

    try {
      // 移除過期的紀錄
      await redis.zremrangebyscore(key, 0, windowStart);

      // 計算窗口內請求數
      const count = await redis.zcard(key);

      if (count >= cfg.maxRequests) {
        res.status(429).json({
          error: 'Too many requests',
          retryAfter: Math.ceil(cfg.windowMs / 1000),
        });
        return;
      }

      // 新增本次請求
      await redis.zadd(key, now, `${now}:${Math.random()}`);
      await redis.expire(key, Math.ceil(cfg.windowMs / 1000));

      next();
    } catch (error) {
      // Redis 掛了就放行（不能因為 rate limiter 擋住所有人）
      console.warn('[RateLimiter] Redis error, allowing request:', error);
      next();
    }
  };
}
```

```typescript
interface Metrics {
```
