## 1. 會員系統架構選擇

### 四種方案比較

| 方案 | 難度 | 成本/月 | 開發時間 | 適合場景 | 彈性 |
|------|------|---------|---------|---------|------|
| 自建 Node.js + Supabase | 高 | 0-25 USD | 2-4 週 | 需高度客製、有工程師 | ★★★★★ |
| WordPress + WooCommerce | 低 | 10-50 USD (主機) | 3-7 天 | 電商、部落格、小型網站 | ★★★ |

### LINE Login 整合

台灣 B2C 最實用的登入方式，用戶不用記密碼。

**申請流程**：
1. LINE Developers Console → 建立 Provider
2. 建立 LINE Login Channel（不是 Messaging API）
3. 設定 Callback URL：`https://你的網域/auth/line/callback`
4. 取得 Channel ID + Channel Secret

**Node.js 串接**：

```typescript
// routes/auth-line.ts
import express from 'express';
import axios from 'axios';

const router = express.Router();

const LINE_CLIENT_ID = process.env.LINE_LOGIN_CHANNEL_ID!;
const LINE_CLIENT_SECRET = process.env.LINE_LOGIN_CHANNEL_SECRET!;
const REDIRECT_URI = process.env.LINE_LOGIN_CALLBACK_URL!;

// 步驟 1：導向 LINE 授權頁
router.get('/auth/line', (req, res) => {
  const state = crypto.randomUUID(); // 防 CSRF
  // 存 state 到 session 或 cookie
  res.cookie('line_state', state, { httpOnly: true, maxAge: 300000 });

  const url = new URL('https://access.line.me/oauth2/v2.1/authorize');
  url.searchParams.set('response_type', 'code');
  url.searchParams.set('client_id', LINE_CLIENT_ID);
  url.searchParams.set('redirect_uri', REDIRECT_URI);
  url.searchParams.set('state', state);
  url.searchParams.set('scope', 'profile openid email');
  res.redirect(url.toString());
});

// 步驟 2：LINE 回來後換 token
router.get('/auth/line/callback', async (req, res) => {
  const { code, state } = req.query;

  // 驗證 state
  if (state !== req.cookies.line_state) {
    return res.status(403).json({ error: 'Invalid state' });
  }

  // 換 access token
  const tokenRes = await axios.post('https://api.line.me/oauth2/v2.1/token',
    new URLSearchParams({
      grant_type: 'authorization_code',
      code: code as string,
      redirect_uri: REDIRECT_URI,
      client_id: LINE_CLIENT_ID,
      client_secret: LINE_CLIENT_SECRET,
    }),
    { headers: { 'Content-Type': 'application/x-www-form-urlencoded' } }
  );

  const { access_token, id_token } = tokenRes.data;

  // 步驟 3：取得用戶資料
  const profileRes = await axios.get('https://api.line.me/v2/profile', {
    headers: { Authorization: `Bearer ${access_token}` }
  });

  const { userId, displayName, pictureUrl } = profileRes.data;

  // 步驟 4：找或建立用戶（upsert）
  const { data: user } = await supabase
    .from('users')
    .upsert({
      line_user_id: userId,
      display_name: displayName,
      avatar_url: pictureUrl,
      provider: 'line',
      last_login: new Date().toISOString(),
    }, { onConflict: 'line_user_id' })
    .select()
    .single();

  // 步驟 5：發 JWT，導回前端
  const jwt = signJwt({ userId: user.id, role: user.role });
  res.redirect(`${process.env.FRONTEND_URL}/auth/callback?token=${jwt}`);
});

export default router;
```

### 2.2 註冊/登入流程

```typescript
// services/auth.service.ts
import bcrypt from 'bcrypt';
import jwt from 'jsonwebtoken';
import crypto from 'crypto';
import { supabase } from '../lib/supabase';

const SALT_ROUNDS = 12;
const JWT_SECRET = process.env.JWT_SECRET!;
const JWT_EXPIRES = '7d';

// ===== 密碼強度驗證 =====
function validatePassword(password: string): { valid: boolean; errors: string[] } {
  const errors: string[] = [];
  if (password.length < 8) errors.push('密碼至少 8 個字元');
  if (!/[A-Z]/.test(password)) errors.push('需包含大寫英文');
  if (!/[a-z]/.test(password)) errors.push('需包含小寫英文');
  if (!/[0-9]/.test(password)) errors.push('需包含數字');
  if (/^(.)\1{2,}/.test(password)) errors.push('不能有連續 3 個相同字元');
  return { valid: errors.length === 0, errors };
}

// ===== 註冊 =====
export async function register(email: string, password: string, displayName: string) {
  // 1. 驗證密碼強度
  const pwCheck = validatePassword(password);
  if (!pwCheck.valid) {
    throw new Error(`密碼不符合規則：${pwCheck.errors.join('、')}`);
  }

  // 2. 檢查 email 是否重複
  const { data: existing } = await supabase
    .from('users')
    .select('id')
    .eq('email', email.toLowerCase())
    .single();

  if (existing) throw new Error('此 Email 已被註冊');

  // 3. hash 密碼
  const passwordHash = await bcrypt.hash(password, SALT_ROUNDS);

  // 4. 建立用戶
  const { data: user, error } = await supabase
    .from('users')
    .insert({
      email: email.toLowerCase(),
      password_hash: passwordHash,
      display_name: displayName,
      role: 'member',
      membership_tier: 'free',
    })
    .select()
    .single();

  if (error) throw error;

  // 5. 發送驗證信
  await sendVerificationEmail(user.id, user.email);

  // 6. 發 JWT
  const token = signJwt(user);

  return { user: sanitizeUser(user), token };
}

// ===== 登入 =====
export async function login(email: string, password: string) {
  // 1. 找用戶
  const { data: user } = await supabase
    .from('users')
    .select('*')
    .eq('email', email.toLowerCase())
    .single();

  if (!user) throw new Error('帳號或密碼錯誤');

  // 2. 檢查鎖定
  if (user.locked_until && new Date(user.locked_until) > new Date()) {
    const minutes = Math.ceil((new Date(user.locked_until).getTime() - Date.now()) / 60000);
    throw new Error(`帳號已鎖定，請 ${minutes} 分鐘後再試`);
  }

  // 3. 驗證密碼
  const valid = await bcrypt.compare(password, user.password_hash);

  if (!valid) {
    // 失敗次數 +1，超過 5 次鎖 15 分鐘
    const attempts = (user.failed_login_attempts || 0) + 1;
    const updates: Record<string, any> = { failed_login_attempts: attempts };

    if (attempts >= 5) {
      updates.locked_until = new Date(Date.now() + 15 * 60 * 1000).toISOString();
      updates.failed_login_attempts = 0;
    }

    await supabase.from('users').update(updates).eq('id', user.id);
    throw new Error('帳號或密碼錯誤');
  }

  // 4. 登入成功，清除失敗次數
  await supabase.from('users').update({
    failed_login_attempts: 0,
    locked_until: null,
    last_login: new Date().toISOString(),
    login_count: (user.login_count || 0) + 1,
  }).eq('id', user.id);

  // 5. 發 JWT
  const token = signJwt(user);

  return { user: sanitizeUser(user), token };
}

// ===== JWT 工具 =====
function signJwt(user: any) {
  return jwt.sign(
    { userId: user.id, role: user.role, tier: user.membership_tier },
    JWT_SECRET,
    { expiresIn: JWT_EXPIRES }
  );
}

function sanitizeUser(user: any) {
  const { password_hash, failed_login_attempts, locked_until, ...safe } = user;
  return safe;
}
```

### 2.3 JWT vs Session-based Auth

| 比較 | JWT | Session |
|------|-----|---------|
| 存儲位置 | Client (localStorage/cookie) | Server (DB/Redis) |
| 擴展性 | 好（stateless） | 需 Redis 共享 session |
| 撤銷能力 | 差（要搭黑名單） | 好（刪 session 即可） |
| 適合場景 | API、SPA、行動 App | 傳統網站、需強制登出 |
| 安全風險 | XSS 竊取 token | CSRF 攻擊 |
| 推薦用法 | 放 httpOnly cookie | 放 httpOnly cookie |

**推薦做法**：JWT 放在 httpOnly cookie，不放 localStorage。

```typescript
// middleware/auth.ts
import { Request, Response, NextFunction } from 'express';
import jwt from 'jsonwebtoken';

export function authMiddleware(req: Request, res: Response, next: NextFunction) {
  // 從 cookie 或 header 取 token
  const token = req.cookies?.token || req.headers.authorization?.replace('Bearer ', '');

  if (!token) {
    return res.status(401).json({ error: '未登入' });
  }

  try {
    const payload = jwt.verify(token, process.env.JWT_SECRET!) as any;
    req.user = payload; // { userId, role, tier }
    next();
  } catch (err) {
    return res.status(401).json({ error: 'Token 無效或過期' });
  }
}

// 角色檢查 middleware
export function requireRole(...roles: string[]) {
  return (req: Request, res: Response, next: NextFunction) => {
    if (!req.user || !roles.includes(req.user.role)) {
      return res.status(403).json({ error: '權限不足' });
    }
    next();
  };
}

// 使用範例
// router.get('/admin/users', authMiddleware, requireRole('admin'), getUsers);
// router.get('/premium/content', authMiddleware, requireRole('premium', 'admin'), getPremiumContent);
```

### 2.4 Email 驗證流程

```typescript
// services/email-verification.ts
import crypto from 'crypto';
import { supabase } from '../lib/supabase';

export async function sendVerificationEmail(userId: string, email: string) {
  const token = crypto.randomBytes(32).toString('hex');
  const expiresAt = new Date(Date.now() + 24 * 60 * 60 * 1000); // 24 小時

  await supabase.from('email_verifications').insert({
    user_id: userId,
    token,
    expires_at: expiresAt.toISOString(),
  });

  const verifyUrl = `${process.env.FRONTEND_URL}/verify-email?token=${token}`;

  // 用 n8n webhook 寄信（或 nodemailer / SendGrid / Resend）
  await fetch(`${process.env.N8N_WEBHOOK_URL}/send-email`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      to: email,
      subject: '請驗證您的 Email',
      html: `
        <h2>歡迎加入！</h2>
        <p>請點擊以下連結驗證您的 Email：</p>
        <a href="${verifyUrl}" style="display:inline-block;padding:12px 24px;background:#4F46E5;color:white;border-radius:8px;text-decoration:none;">
          驗證 Email
        </a>
        <p>此連結 24 小時內有效。</p>
      `,
    }),
  });
}

export async function verifyEmail(token: string) {
  const { data: record } = await supabase
    .from('email_verifications')
    .select('*')
    .eq('token', token)
    .eq('used', false)
    .gt('expires_at', new Date().toISOString())
    .single();

  if (!record) throw new Error('驗證連結無效或已過期');

  // 標記已使用
  await supabase.from('email_verifications')
    .update({ used: true })
    .eq('id', record.id);

  // 更新用戶 email_verified
  await supabase.from('users')
    .update({ email_verified: true })
    .eq('id', record.user_id);

  return { success: true };
}
```

```typescript
// services/password-reset.ts
export async function requestPasswordReset(email: string) {

        <a href="${resetUrl}" style="display:inline-block;padding:12px 24px;background:#4F46E5;color:white;border-radius:8px;text-decoration:none;">

        </a>
        <p>此連結 1 小時內有效。</p>
      `,
    }),

export async function resetPassword(token: string, newPassword: string) {
  const pwCheck = validatePassword(newPassword);
  if (!pwCheck.valid) {
    throw new Error(`密碼不符合規則：${pwCheck.errors.join('、')}`);
  }

  const { data: record } = await supabase
    .from('password_resets')
    .select('*')
    .eq('token', token)
    .eq('used', false)
    .gt('expires_at', new Date().toISOString())
    .single();

  if (!record) throw new Error('重設連結無效或已過期');

  const passwordHash = await bcrypt.hash(newPassword, 12);

  await supabase.from('password_resets')
    .update({ used: true })
    .eq('id', record.id);

  await supabase.from('users')
    .update({ password_hash: passwordHash })
    .eq('id', record.user_id);

  return { success: true };
}
```

```typescript
import { OAuth2Client } from 'google-auth-library';

const googleClient = new OAuth2Client(
  process.env.GOOGLE_CLIENT_ID,
  process.env.GOOGLE_CLIENT_SECRET,
  `${process.env.API_BASE_URL}/auth/google/callback`

router.get('/auth/google', (req, res) => {

router.get('/auth/google/callback', async (req, res) => {
  const { code } = req.query;
  const { tokens } = await googleClient.getToken(code as string);
  googleClient.setCredentials(tokens);

  // 取 Google 用戶資訊
  const ticket = await googleClient.verifyIdToken({
    idToken: tokens.id_token!,
    audience: process.env.GOOGLE_CLIENT_ID,
  });

  const { data: user } = await supabase
    .from('users')

      google_id: payload.sub,
      email: payload.email,
      display_name: payload.name,
      avatar_url: payload.picture,
      email_verified: payload.email_verified ?? false,
      provider: 'google',
    }, { onConflict: 'google_id' })
    .select()
    .single();

  const jwt = signJwt(user);
  res.redirect(`${process.env.FRONTEND_URL}/auth/callback?token=${jwt}`);
});
```

## 3. Supabase Auth 快速方案

最省時間的方案。Supabase Auth 免費版支援 50,000 月活用戶，足夠大部分接案專案。

### 3.1 設定步驟

```bash
# 1. 安裝
npm install @supabase/supabase-js
```

### 3.3 Row Level Security (RLS)

Supabase 最強大的功能：資料庫層級的權限控制。

```sql
-- 啟用 RLS
ALTER TABLE profiles ENABLE ROW LEVEL SECURITY;
ALTER TABLE transactions ENABLE ROW LEVEL SECURITY;

-- 用戶只能讀自己的 profile
CREATE POLICY "users_read_own_profile" ON profiles
  FOR SELECT USING (auth.uid() = id);

-- 用戶只能改自己的 profile
CREATE POLICY "users_update_own_profile" ON profiles
  FOR UPDATE USING (auth.uid() = id)
  WITH CHECK (auth.uid() = id);

-- 用戶只能看自己的交易
CREATE POLICY "users_read_own_transactions" ON transactions
  FOR SELECT USING (auth.uid() = user_id);

-- Admin 看全部
CREATE POLICY "admin_read_all_profiles" ON profiles
  FOR SELECT USING (
    EXISTS (
      SELECT 1 FROM profiles WHERE id = auth.uid() AND role = 'admin'
    )
  );

-- Service role 不受 RLS 限制（server 端用 service_role key）
-- 金流 webhook 回調用 service_role key 寫入交易
```

```typescript
  const { data: { user } } = await supabaseAdmin.auth.getUser(token);

     new Date(profile.membership_expires_at) > new Date());

  if (!isPremium) return res.status(403).json({ error: '此功能僅限付費會員' });

  req.user = { ...user, profile };
```

```typescript
// services/ecpay.service.ts
import crypto from 'crypto';

const {
  ECPAY_MERCHANT_ID,
  ECPAY_HASH_KEY,
  ECPAY_HASH_IV,
  ECPAY_API_URL,
  API_BASE_URL,
  FRONTEND_URL,
} = process.env;

// ===== 產生 CheckMacValue（綠界驗證用） =====
function generateCheckMacValue(params: Record<string, string>): string {
  // 1. 依照 key 排序
  const sorted = Object.keys(params)
    .sort((a, b) => a.toLowerCase().localeCompare(b.toLowerCase()))
    .map(key => `${key}=${params[key]}`)
    .join('&');

  // 2. 前後加上 HashKey 和 HashIV
  let raw = `HashKey=${ECPAY_HASH_KEY}&${sorted}&HashIV=${ECPAY_HASH_IV}`;

  // 3. URL encode（小寫）
  raw = encodeURIComponent(raw).toLowerCase();

  // 4. 特殊字元替換（綠界規定）
  raw = raw.replace(/%2d/g, '-')
           .replace(/%5f/g, '_')
           .replace(/%2e/g, '.')
           .replace(/%21/g, '!')
           .replace(/%2a/g, '*')
           .replace(/%28/g, '(')
           .replace(/%29/g, ')')
           .replace(/%20/g, '+');

  // 5. SHA256 + 大寫
  return crypto.createHash('sha256').update(raw).digest('hex').toUpperCase();
}

// ===== 建立訂單（產生送給綠界的 form data） =====
export function createEcpayOrder(params: {
  orderId: string;
  amount: number;
  description: string;
  paymentMethod?: 'Credit' | 'ATM' | 'CVS' | 'BARCODE' | 'ALL';
}) {
  const now = new Date();
  const merchantTradeDate = now.toISOString()
    .replace('T', ' ').replace(/\.\d+Z/, '')
    .replace(/-/g, '/');

  const formData: Record<string, string> = {
    MerchantID: ECPAY_MERCHANT_ID!,
    MerchantTradeNo: params.orderId,     // 不超過 20 字元
    MerchantTradeDate: merchantTradeDate, // yyyy/MM/dd HH:mm:ss
    PaymentType: 'aio',
    TotalAmount: String(params.amount),   // 整數，不含小數
    TradeDesc: encodeURIComponent('線上購物'),
    ItemName: params.description,          // 多項用 # 分隔
    ReturnURL: `${API_BASE_URL}/api/payment/ecpay/notify`,   // 背景通知（server-to-server）
    ClientBackURL: `${FRONTEND_URL}/payment/result`,          // 用戶點「返回商店」
    OrderResultURL: `${FRONTEND_URL}/payment/result`,         // 付款完直接導向
    ChoosePayment: params.paymentMethod || 'ALL',
    EncryptType: '1',                      // SHA256
    NeedExtraPaidInfo: 'Y',
  };

  // 計算 CheckMacValue
  formData.CheckMacValue = generateCheckMacValue(formData);

  return {
    apiUrl: `${ECPAY_API_URL}/Cashier/AioCheckOut/V5`,
    formData,
  };
}

router.post('/api/payment/ecpay/notify', async (req, res) => {
  const params = req.body;

  // 1. 驗證 CheckMacValue
  const receivedMac = params.CheckMacValue;
  const { CheckMacValue, ...rest } = params;
  const expectedMac = generateCheckMacValue(rest);

  if (receivedMac !== expectedMac) {
    console.error('ECPay CheckMacValue mismatch');
    return res.send('0|CheckMacValue Error');
  }

  // 2. 更新訂單狀態
  const isPaid = params.RtnCode === '1';

  await supabase.from('transactions').update({
    status: isPaid ? 'paid' : 'failed',
    gateway_trade_no: params.TradeNo,
    payment_method: params.PaymentType,
    paid_at: isPaid ? new Date().toISOString() : null,
    metadata: params,
  }).eq('order_no', params.MerchantTradeNo);

  // 3. 付款成功 → 開通權限
  if (isPaid) {
    const { data: tx } = await supabase
      .from('transactions')
      .select('user_id, metadata')
      .eq('order_no', params.MerchantTradeNo)
      .single();

      await activateMembership(tx.user_id, 'basic', 30); // 開通 30 天
    }
  }

  res.send('1|OK');
});
```

```typescript
// services/linepay.service.ts
import crypto from 'crypto';
import axios from 'axios';

const {
  LINEPAY_CHANNEL_ID,
  LINEPAY_CHANNEL_SECRET,
  LINEPAY_API_URL,
  API_BASE_URL,
} = process.env;

// LINE Pay v3 簽章
function generateLinePaySignature(
  channelSecret: string,
  uri: string,
  body: string,
  nonce: string
): string {
  const message = channelSecret + uri + body + nonce;
  return crypto.createHmac('sha256', channelSecret)
    .update(message)
    .digest('base64');
}

function linePayHeaders(uri: string, body: object) {
  const nonce = crypto.randomUUID();
  const bodyStr = JSON.stringify(body);

  return {
    'Content-Type': 'application/json',
    'X-LINE-ChannelId': LINEPAY_CHANNEL_ID!,
    'X-LINE-Authorization-Nonce': nonce,
    'X-LINE-Authorization': generateLinePaySignature(
      LINEPAY_CHANNEL_SECRET!, uri, bodyStr, nonce
    ),
  };
}

// ===== Step 1: Request（建立付款請求） =====
export async function linePayRequest(params: {
  orderId: string;
  amount: number;
  productName: string;
}) {
  const uri = '/v3/payments/request';
  const body = {
    amount: params.amount,
    currency: 'TWD',
    orderId: params.orderId,
    packages: [{
      id: 'pkg-001',
      amount: params.amount,
      products: [{
        name: params.productName,
        quantity: 1,
        price: params.amount,
      }],
    }],
    redirectUrls: {
      confirmUrl: `${API_BASE_URL}/api/payment/linepay/confirm`,
      cancelUrl: `${API_BASE_URL}/api/payment/linepay/cancel`,
    },
  };

  const res = await axios.post(`${LINEPAY_API_URL}${uri}`, body, {
    headers: linePayHeaders(uri, body),
  });

  // res.data.info.paymentUrl.web → 導向 LINE Pay 付款頁
  // res.data.info.transactionId → 存起來，confirm 時要用
  return res.data;
}

// ===== Step 2: Confirm（用戶付完回來確認） =====
export async function linePayConfirm(transactionId: string, amount: number) {
  const uri = `/v3/payments/requests/${transactionId}/confirm`;
  const body = {
    amount,
    currency: 'TWD',
  };

  const res = await axios.post(`${LINEPAY_API_URL}${uri}`, body, {
    headers: linePayHeaders(uri, body),
  });

  return res.data;
}

// ===== Step 3: Refund（退款） =====
export async function linePayRefund(transactionId: string, refundAmount?: number) {

  const body = refundAmount ? { refundAmount } : {};

  const res = await axios.post(`${LINEPAY_API_URL}${uri}`, body, {
    headers: linePayHeaders(uri, body),

}

router.get('/api/payment/linepay/confirm', async (req, res) => {
  const { transactionId, orderId } = req.query;

  // 查訂單金額
  const { data: tx } = await supabase
    .from('transactions')

  try {
```

```typescript
function aesEncrypt(data: string): string {
  const cipher = crypto.createCipheriv(
    'aes-256-cbc',
    NEWEBPAY_HASH_KEY!,
    NEWEBPAY_HASH_IV!
  );
  let encrypted = cipher.update(data, 'utf8', 'hex');
  encrypted += cipher.final('hex');
  return encrypted;
}

function sha256Hash(data: string): string {
  return crypto.createHash('sha256')
    .update(`HashKey=${NEWEBPAY_HASH_KEY}&${data}&HashIV=${NEWEBPAY_HASH_IV}`)
    .digest('hex')
    .toUpperCase();
}
```

**Webhook 處理（重要！Stripe 所有狀態變化都靠 webhook）**：

```typescript
// routes/stripe-webhook.ts
import Stripe from 'stripe';

const stripe = new Stripe(process.env.STRIPE_SECRET_KEY!);

router.post('/api/payment/stripe/webhook',
  express.raw({ type: 'application/json' }), // 必須用 raw body
  async (req, res) => {
    const sig = req.headers['stripe-signature']!;

    let event: Stripe.Event;
    try {
      event = stripe.webhooks.constructEvent(
        req.body,
        sig,
        process.env.STRIPE_WEBHOOK_SECRET!
      );
    } catch (err: any) {
      console.error('Stripe webhook 驗證失敗:', err.message);
      return res.status(400).send(`Webhook Error: ${err.message}`);
    }

    // 處理不同事件
    switch (event.type) {
      case 'checkout.session.completed': {
        const session = event.data.object as Stripe.Checkout.Session;
        const userId = session.metadata?.userId;

        if (session.mode === 'payment') {
          // 一次性付款成功
          await supabase.from('transactions').update({
            status: 'paid',
            paid_at: new Date().toISOString(),
          }).eq('gateway_trade_no', session.id);
        }

        if (session.mode === 'subscription') {
          // 訂閱建立成功
          await supabase.from('subscriptions').insert({
            user_id: userId,
            plan: 'pro',
            gateway: 'stripe',
            gateway_subscription_id: session.subscription as string,
            amount: session.amount_total! / 100,
            status: 'active',
          });
          await activateMembership(userId!, 'pro', 30);
        }
        break;
      }

      case 'invoice.payment_succeeded': {
        // 訂閱續費成功
        const invoice = event.data.object as Stripe.Invoice;
        const subId = invoice.subscription as string;

        await supabase.from('subscriptions').update({
          status: 'active',
          current_period_end: new Date(
            (invoice.lines.data[0]?.period?.end ?? 0) * 1000
          ).toISOString(),
        }).eq('gateway_subscription_id', subId);
        break;
      }

      case 'invoice.payment_failed': {
        // 續費失敗
        const invoice = event.data.object as Stripe.Invoice;
        await supabase.from('subscriptions').update({
          status: 'past_due',
        }).eq('gateway_subscription_id', invoice.subscription as string);
        break;
      }

      case 'customer.subscription.deleted': {
        // 訂閱取消
        const sub = event.data.object as Stripe.Subscription;
        await supabase.from('subscriptions').update({
          status: 'cancelled',
          cancelled_at: new Date().toISOString(),
        }).eq('gateway_subscription_id', sub.id);

        // 停權
        const { data: subRecord } = await supabase
          .from('subscriptions')
          .select('user_id')
          .eq('gateway_subscription_id', sub.id)
          .single();
        if (subRecord) {
          await supabase.from('profiles').update({
            membership_tier: 'free',
            role: 'member',
          }).eq('id', subRecord.user_id);
        }
        break;
      }

      case 'charge.refunded': {
        // 退款
        const charge = event.data.object as Stripe.Charge;
        await supabase.from('transactions').update({
          status: 'refunded',
          refunded_at: new Date().toISOString(),
        }).eq('gateway_trade_no', charge.payment_intent as string);
        break;
      }
    }

    res.json({ received: true });
  }
);
```

## 5. 金流安全注意事項

### 5.1 基本安全清單

| 項目 | 必要性 | 說明 |
|------|--------|------|
| HTTPS | 必須 | 所有金流 API 都要求 HTTPS，否則不給過審 |
| CheckMacValue 驗證 | 必須 | 綠界/藍新每筆回調都要驗，防止偽造 |
| Webhook 簽章驗證 | 必須 | Stripe 用 stripe-signature，LINE Pay 用 HMAC |
| 金額 server 端驗證 | 必須 | 永遠不信前端傳來的金額，用 server 端存的金額 |
| 訂單防重複 | 必須 | 同一筆訂單只能成功一次 |
| 幂等處理 | 必須 | Webhook 可能重複送，處理邏輯要幂等 |
| API Key 安全 | 必須 | HashKey/Secret 只放 server 端，絕不放前端 |
| PCI DSS | 建議 | 不直接處理卡號（用綠界/Stripe 的表單或 SDK） |

### 5.2 防重複扣款

```typescript
// 幂等鎖 — 確保同一筆 webhook 只處理一次
async function processPaymentOnce(orderNo: string, callback: () => Promise<void>) {
  // 1. 用資料庫鎖（Supabase/PostgreSQL）
  const { data: tx } = await supabase
    .from('transactions')
    .select('status')
    .eq('order_no', orderNo)
    .single();

  // 已處理過就跳過
  if (tx?.status === 'paid' || tx?.status === 'refunded') {
    console.log(`Order ${orderNo} already processed, skipping`);
    return;
  }

  // 2. 用 UPDATE ... WHERE status = 'pending' 確保只有一次能成功
  const { data: updated, error } = await supabase
    .from('transactions')
    .update({ status: 'processing' })
    .eq('order_no', orderNo)
    .eq('status', 'pending')  // 只有 pending 才能變 processing
    .select();

  if (!updated || updated.length === 0) {
    console.log(`Order ${orderNo} already being processed`);
    return;
  }

  // 3. 執行實際處理
  try {
    await callback();
  } catch (err) {
    // 失敗回滾
    await supabase.from('transactions')
      .update({ status: 'pending' })
      .eq('order_no', orderNo);
    throw err;
  }
}
```

```typescript
// services/membership.service.ts
export async function activateMembership(
  userId: string,
  tier: 'basic' | 'pro',
  days: number
) {
  const { data: user } = await supabase
    .from('profiles')
    .select('membership_tier, membership_expires_at')
    .eq('id', userId)
    .single();

  let expiresAt: Date;

  // 如果還沒過期，從到期日往後加
  if (user?.membership_expires_at && new Date(user.membership_expires_at) > new Date()) {
    expiresAt = new Date(user.membership_expires_at);
    expiresAt.setDate(expiresAt.getDate() + days);
  } else {
    // 已過期或從沒買過，從今天算
    expiresAt = new Date();
    expiresAt.setDate(expiresAt.getDate() + days);
  }

  await supabase.from('profiles').update({
    membership_tier: tier,
    role: tier === 'pro' ? 'premium' : 'member',
    membership_expires_at: expiresAt.toISOString(),
  }).eq('id', userId);

  return { tier, expiresAt };
}

// 檢查會員是否有效
export async function checkMembership(userId: string) {
  const { data: user } = await supabase
    .from('profiles')
    .select('membership_tier, membership_expires_at, role')
    .eq('id', userId)
    .single();

  if (!user) return { active: false, tier: 'free' };

  const isActive = user.membership_tier !== 'free' &&
    user.membership_expires_at &&
    new Date(user.membership_expires_at) > new Date();

  // 自動降級過期會員
  if (!isActive && user.membership_tier !== 'free') {
    await supabase.from('profiles').update({
      membership_tier: 'free',
      role: 'member',
    }).eq('id', userId);

    return { active: false, tier: 'free', expired: true };
  }

  return {
    active: isActive,
    tier: user.membership_tier,
    expiresAt: user.membership_expires_at,
    daysLeft: isActive
      ? Math.ceil((new Date(user.membership_expires_at).getTime() - Date.now()) / 86400000)
      : 0,
  };
}
```

### 8.5 觸發 n8n 的通用函式

在 server 端統一用這個函式觸發 n8n workflow：

```typescript
// utils/n8n-trigger.ts
export async function triggerN8nWebhook(
  path: string,
  data: Record<string, any>
) {
  const url = `${process.env.N8N_WEBHOOK_URL}/${path}`;
  try {
    const res = await fetch(url, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(data),
    });
    if (!res.ok) {
      console.error(`n8n webhook failed: ${path}`, await res.text());
    }
    return res.ok;
  } catch (err) {
    console.error(`n8n webhook error: ${path}`, err);
    return false;
  }
}

// 使用範例
await triggerN8nWebhook('new-member', { email, displayName });
await triggerN8nWebhook('payment-success', { userId, email, plan, amount, expiresAt });
await triggerN8nWebhook('payment-refunded', { userId, email, amount, orderId });
```
