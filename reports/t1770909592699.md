## 進階配置

### 1. 多模型混合策略

```yaml
# clawflow.config.yaml
models:
  providers:
    - name: ollama
      type: ollama
      baseUrl: http://localhost:11434
      models:
        - mistral:latest      # 快速推理
        - llama3.2:latest     # 均衡
        - neural-chat:latest  # 對話優化
    
    - name: openrouter
      type: openai
      apiKey: ${OPENROUTER_KEY}
      models:
        - openrouter/qwen/qwen-3-max  # 複雜任務
        - openrouter/deepseek/deepseek-r1:free

  routing:
    # 自動路由策略
    simple-task: ollama/mistral:latest
    complex-task: openrouter/qwen-3-max
    chat: ollama/neural-chat:latest
    default: ollama/llama3.2:latest
```

## Multi-Agent 編排

### 1. Agent 間通訊

```javascript
// agents/researcher.js
const researcher = new Agent({
  name: 'researcher',
  async run(task) {
    const results = await this.agents.search('web-search', {
      query: task.query,
      count: 5
    });
    
    await this.publish('research:complete', { results, task });
    return results;
  }
});

// agents/writer.js
const writer = new Agent({
  name: 'writer',
  async run(task) {
    this.subscribe('research:complete', async ({ results }) => {
      const article = await this.llm.chat([
        { role: 'system', content: '撰寫文章' },
        { role: 'user', content: `基於研究結果: ${JSON.stringify(results)}` }
      ]);
      
      return article;
    });
  }
});
```

### 2. 工作流程編排

```yaml
workflows:
  content-pipeline:
    name: 內容生產流程
    steps:
      - id: research
        agent: researcher
        input: ${input.topic}
        timeout: 300
      
      - id: draft
        agent: writer
        input: ${research.results}
        dependsOn: research
        timeout: 600
      
      - id: review
        agent: reviewer
        input: ${draft.article}
        dependsOn: draft
        parallel: true
      
      - id: publish
        agent: publisher
        input: ${review.feedback}
        dependsOn: review
    
    onSuccess: notify-slack
    onFailure: alert-admin
```

```javascript
// 根據任務複雜度選擇模型
const selectModel = (task) => {
```

### 2. 批次處理

```javascript
// 批次處理 100 個項目
const items = await getItems();
const batchSize = 10;

for (let i = 0; i < items.length; i += batchSize) {
  const batch = items.slice(i, i + batchSize);
  
  const results = await Promise.all(
    batch.map(item => processItem(item))
  );
  
  await saveBatch(results);
}
```

### 3. 快取策略

```javascript
// 實作智能快取
class SmartCache {
  async get(key, fetchFn, ttl = 300) {
    const cached = await this.cache.get(key);
    if (cached) return cached;
    
    const result = await fetchFn();
    await this.cache.set(key, result, ttl);
    return result;
  }
}

// 使用
const result = await cache.get(
  `github:user:${username}`,
  () => this.tools.github.getUser(username),
  3600
);
```

## 生產環境部署

### 1. Docker 部署

```dockerfile
FROM node:18-alpine

WORKDIR /app
COPY package*.json ./
RUN npm install --production

COPY . .

ENV NODE_ENV=production
ENV LOG_LEVEL=warn

EXPOSE 3000
CMD ["clawflow", "start"]
```

### 2. Kubernetes 部署

```yaml
apiVersion: apps/v1
kind: Deployment
metadata:
  name: clawflow-agent
spec:
  replicas: 3
  selector:
    matchLabels:
      app: clawflow-agent
  template:
    metadata:
      labels:
        app: clawflow-agent
    spec:
      containers:
      - name: clawflow
        image: clawflow:latest
        resources:
          requests:

            cpu: "250m"
```
