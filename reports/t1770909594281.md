### Docker 容器化規格

#### 1. Dockerfile 結構

```dockerfile
# Dockerfile (多階段建置)

# Stage 1: Builder
FROM node:20-alpine AS builder
WORKDIR /build

# 複製 package 檔案
COPY package*.json ./

# 安裝依賴
RUN npm ci --only=production && \
    npm cache clean --force

# 複製源代碼
COPY . .

# 編譯/打包 (如需要)
RUN npm run build || true

# Stage 2: Runtime
FROM node:20-alpine

# 安裝系統依賴
RUN apk add --no-cache \
    curl \
    tini  # 作為 init 進程

WORKDIR /app

# 建立非 root 用戶
RUN addgroup -g 1001 -S nodejs && \
    adduser -S nodejs -u 1001

# 從 builder 複製編譯結果
COPY --from=builder --chown=nodejs:nodejs /build/node_modules ./node_modules
COPY --from=builder --chown=nodejs:nodejs /build/dist ./dist
COPY --from=builder --chown=nodejs:nodejs /build/package*.json ./

# 複製環境配置範本
COPY .env.example .env

# 暴露埠
EXPOSE 3000 3011

# 健康檢查
HEALTHCHECK --interval=30s --timeout=3s --start-period=40s --retries=3 \
  CMD curl -f http://localhost:3000/health || exit 1

# 切換至非 root 用戶
USER nodejs

# 啟動應用
ENTRYPOINT ["/sbin/tini", "--"]
CMD ["node", "dist/index.js"]
```

#### 2. Docker-Compose 配置

```yaml
# docker-compose.yml

version: '3.8'

services:
  # OpenClaw Gateway
  openclaw:
    build:
      context: .
      dockerfile: Dockerfile
    container_name: openclaw-gateway
    ports:
      - "3000:3000"
      - "3011:3011"
    environment:
      NODE_ENV: production
      LOG_LEVEL: info
      ANTHROPIC_API_KEY: ${ANTHROPIC_API_KEY}
      OPENROUTER_API_KEY: ${OPENROUTER_API_KEY}
    volumes:
      - ./workspace:/app/workspace
      - openclaw-logs:/app/logs
    restart: unless-stopped
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:3000/health"]
      interval: 30s
      timeout: 10s
      retries: 3
      start_period: 40s
    networks:
      - openclaw-net

  # Ollama 本地 LLM (可選)
  ollama:
    image: ollama/ollama:latest
    container_name: ollama-service
    ports:
      - "11434:11434"
    volumes:
      - ollama-data:/root/.ollama
    environment:
      - OLLAMA_MODELS=/root/.ollama/models
    restart: unless-stopped
    networks:
      - openclaw-net
    profiles:
      - with-ollama

  # PostgreSQL 資料庫 (可選)
  postgres:
    image: postgres:15-alpine
    container_name: openclaw-db
    environment:
      POSTGRES_DB: openclaw
      POSTGRES_USER: openclaw
      POSTGRES_PASSWORD: ${DB_PASSWORD:-changeme}
    volumes:
      - postgres-data:/var/lib/postgresql/data
    restart: unless-stopped
    networks:
      - openclaw-net
    profiles:
      - with-db
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U openclaw"]
      interval: 10s
      timeout: 5s
      retries: 5

  # Redis 快取 (可選)
  redis:
    image: redis:7-alpine
    container_name: openclaw-cache
    ports:
      - "6379:6379"
    volumes:
      - redis-data:/data
    restart: unless-stopped
    networks:
      - openclaw-net
    profiles:
      - with-cache
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 10s
      timeout: 5s
      retries: 5

volumes:
  openclaw-logs:
  ollama-data:
  postgres-data:
  redis-data:

networks:
  openclaw-net:
    driver: bridge
```

#### 3. 使用方式

**快速啟動 (只有 OpenClaw)**
```bash
docker-compose up -d
```

**完整堆棧 (含 Ollama + PostgreSQL + Redis)**
```bash
docker-compose --profile with-ollama --profile with-db --profile with-cache up -d
```

**檢查狀態**
```bash
docker-compose ps
docker-compose logs -f openclaw
```

**停止服務**
```bash
docker-compose down
docker-compose down -v  # 刪除 volumes
```

#### 7. CI/CD 整合

```yaml
# .github/workflows/docker-build.yml

name: Docker Build & Push

on:
  push:
    branches: [main]
    tags: ['v*']

jobs:
  build:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      
      - name: Build Docker image
        run: docker build -t openclaw:latest .
      
      - name: Run Trivy security scan
        uses: aquasecurity/trivy-action@master
        with:
          image-ref: 'openclaw:latest'
          format: 'sarif'
      
      - name: Push to Docker Hub
        if: github.event_name == 'push' && startsWith(github.ref, 'refs/tags')
        run: |
          echo ${{ secrets.DOCKER_PASSWORD }} | docker login -u ${{ secrets.DOCKER_USERNAME }} --password-stdin
          docker tag openclaw:latest ${{ secrets.DOCKER_USERNAME }}/openclaw:latest
          docker push ${{ secrets.DOCKER_USERNAME }}/openclaw:latest
```
