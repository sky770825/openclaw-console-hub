#### 4. 自動化測試套件

```bash
#!/bin/bash
# run-all-tests.sh

SCRIPT="./unified-monitor.sh"
PASS=0
FAIL=0

run_test() {
  local name=$1
  local cmd=$2
  
  echo ""
  echo "========== $name =========="
  
  if eval "$cmd"; then
    echo "✅ $name PASSED"
    ((PASS++))
  else
    echo "❌ $name FAILED"
    ((FAIL++))
  fi
}

# 執行所有測試
run_test "Quick Mode" "$SCRIPT quick > /dev/null"
run_test "Detailed Mode" "$SCRIPT detailed > /dev/null"
run_test "Expert Mode (timeout 40s)" "timeout 40 $SCRIPT expert > /dev/null"
run_test "Help Text" "$SCRIPT --help > /dev/null || true"
run_test "Invalid Mode" "$SCRIPT invalid 2>&1 | grep -q 'Unknown mode' || true"

# 報告
echo ""
echo "========== 測試報告 =========="
echo "✅ 通過: $PASS"
echo "❌ 失敗: $FAIL"
echo "📊 總計: $((PASS + FAIL))"

if [ $FAIL -eq 0 ]; then
  echo "🎉 所有測試通過！"
  exit 0
else
  echo "⚠️  有 $FAIL 個測試失敗"
  exit 1
fi
```
