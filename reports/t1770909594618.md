| 平台 | 類型 | 優勢 | 成本 |

| **本地 (Mac)** | 本機運行 | 零成本，全功能 | $0 |

#### 3. 統一部署腳本

```bash
#!/bin/bash
# deploy.sh

set -e

# 0️⃣ 配置
REPO_URL="https://github.com/openclaw/openclaw"
TARGET_PLATFORM="${1:-vps}"  # vps | railway | vercel | local
ENVIRONMENT="${2:-production}"
VERBOSE="${VERBOSE:-0}"

# 顏色輸出
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
NC='\033[0m'

log() { echo -e "${GREEN}[$(date +'%H:%M:%S')]${NC} $1"; }
warn() { echo -e "${YELLOW}[WARNING]${NC} $1"; }
error() { echo -e "${RED}[ERROR]${NC} $1"; exit 1; }

# 1️⃣ 環境檢測
detect_platform() {
  log "🔍 檢測執行環境..."
  
  case "$TARGET_PLATFORM" in
    vps|local)
      [[ ! -f /etc/os-release ]] && error "不支援的系統"
      source /etc/os-release
      OS=$NAME
      log "✅ 檢測到: $OS"
      ;;
    railway)
      [[ -z "$RAILWAY_ENVIRONMENT" ]] && warn "非 Railway 環境"
      log "✅ Railway 環境檢測"
      ;;
    vercel)
      [[ -z "$VERCEL_ENV" ]] && warn "非 Vercel 環境"
      log "✅ Vercel 環境檢測"
      ;;
    *)
      error "未知平台: $TARGET_PLATFORM"
      ;;
  esac
}

# 2️⃣ 依賴安裝
install_dependencies() {
  log "📦 安裝依賴..."
  
  case "$OS" in
    Ubuntu*|Debian*)
      sudo apt-get update
      sudo apt-get install -y \
        nodejs npm \
        curl git \
        build-essential python3
      ;;
    macOS)
      command -v brew >/dev/null || error "需要安裝 Homebrew"
      brew install node
      ;;
  esac
  
  npm install -g yarn pnpm
  log "✅ 依賴安裝完成"
}

# 3️⃣ 代碼克隆
clone_repo() {
  log "📥 克隆代碼倉庫..."
  
  WORK_DIR="/opt/openclaw"
  [[ ! -d "$WORK_DIR" ]] && sudo mkdir -p "$WORK_DIR"
  
  if [ -d "$WORK_DIR/.git" ]; then
    cd "$WORK_DIR"
    git pull
  else
    sudo git clone "$REPO_URL" "$WORK_DIR"
  fi
  
  cd "$WORK_DIR"
  log "✅ 代碼克隆完成"
}

# 4️⃣ 配置生成
generate_config() {
  log "⚙️  生成配置..."
  
  CONFIG_DIR="$WORK_DIR/.openclaw/config"
  mkdir -p "$CONFIG_DIR"
  
  # 從環境變數讀取密鑰
  cat > "$CONFIG_DIR/env.production" << EOF
# OpenClaw Production Configuration
NODE_ENV=production
PORT=${PORT:-3000}

# API Keys (從環境變數注入)
ANTHROPIC_API_KEY=$ANTHROPIC_API_KEY
OPENROUTER_API_KEY=$OPENROUTER_API_KEY
TELEGRAM_BOT_TOKEN=$TELEGRAM_BOT_TOKEN

# Database
DATABASE_URL=$DATABASE_URL

# Deployment Info
DEPLOYMENT_PLATFORM=$TARGET_PLATFORM
DEPLOYMENT_TIME=$(date -u +%Y-%m-%dT%H:%M:%SZ)
EOF
  
  log "✅ 配置生成完成"
}

# 5️⃣ 資料庫初始化
init_database() {
  log "🗄️  初始化資料庫..."
  
  if [ -f "$WORK_DIR/scripts/init-db.sh" ]; then
    bash "$WORK_DIR/scripts/init-db.sh"
  else
    warn "找不到資料庫初始化腳本"
  fi
  
  log "✅ 資料庫初始化完成"
}

# 6️⃣ 安裝 npm 依賴
install_npm_deps() {
  log "📚 安裝 npm 依賴..."
  
  cd "$WORK_DIR"
  npm install --production
  
  log "✅ npm 依賴安裝完成"
}

# 7️⃣ 平台特定配置
platform_specific_config() {
  log "🔧 平台特定配置..."
  
  case "$TARGET_PLATFORM" in
    vps)
      setup_vps_systemd
      ;;
    railway)
      setup_railway_config
      ;;
    vercel)
      setup_vercel_config
      ;;
    local)
      log "⏭️  略過本地特定配置"
      ;;
  esac
  
  log "✅ 平台配置完成"
}

# VPS: Systemd 服務
setup_vps_systemd() {
  log "  📋 設定 systemd 服務..."
  
  sudo tee /etc/systemd/system/openclaw.service > /dev/null << EOF
[Unit]
Description=OpenClaw Gateway Service
After=network.target

[Service]
Type=simple
User=openclaw
WorkingDirectory=$WORK_DIR
ExecStart=/usr/bin/node bin/gateway.js
Restart=always
RestartSec=10
StandardOutput=journal
StandardError=journal

[Install]
WantedBy=multi-user.target
EOF
  
  sudo systemctl daemon-reload
  sudo systemctl enable openclaw
  sudo systemctl start openclaw
  
  log "  ✅ Systemd 服務設定完成"
}

# Railway: railway.json
setup_railway_config() {
  log "  📋 設定 Railway 配置..."
  
  cat > "$WORK_DIR/railway.json" << EOF
{
  "build": {
    "builder": "dockerfile"
  },
  "deploy": {
    "numReplicas": 1,
    "sleepApplication": false,
    "restartPolicyMaxRetries": 10
  }
}
EOF
  
  log "  ✅ Railway 配置完成"
}

# Vercel: vercel.json
setup_vercel_config() {
  log "  📋 設定 Vercel 配置..."
  
  cat > "$WORK_DIR/vercel.json" << EOF
{
  "version": 2,
  "builds": [
    { "src": "api/**/*.js", "use": "@vercel/node" }
  ],
  "routes": [
    { "src": "/(.*)", "dest": "api/index.js" }
  ],
  "env": {
    "NODE_ENV": "production"
  }
}
EOF
  
  log "  ✅ Vercel 配置完成"
}

# 8️⃣ 啟動服務
start_service() {
  log "🚀 啟動服務..."
  
  case "$TARGET_PLATFORM" in
    vps)
      sudo systemctl status openclaw
      ;;
    railway)
      log "  ⏭️  Railway 自動部署"
      ;;
    vercel)
      log "  ⏭️  Vercel 自動部署"
      ;;
    local)
      log "  啟動本地開發服務..."
      npm run dev &
      ;;
  esac
  
  log "✅ 服務啟動完成"
}

# 9️⃣ 健康檢查
health_check() {
  log "🏥 執行健康檢查..."
  
  sleep 3  # 等待服務啟動
  
  HEALTH_URL="${HEALTH_URL:-http://localhost:3000/health}"
  
  for i in {1..10}; do
    if curl -s "$HEALTH_URL" | grep -q '"ok"'; then
      log "✅ 健康檢查通過"
      return 0
    fi
    warn "  嘗試 $i/10..."
    sleep 2
  done
  
  error "健康檢查失敗"
}

# 🔟 部署摘要
deployment_summary() {
  log ""
  log "========================================="
  log "  ✨ 部署完成！"
  log "========================================="
  log "平台: $TARGET_PLATFORM"
  log "環境: $ENVIRONMENT"
  log "時間: $(date)"
  log "位置: $WORK_DIR"
  log ""
  
  case "$TARGET_PLATFORM" in
    vps)
      log "檢查狀態: sudo systemctl status openclaw"
      log "查看日誌: sudo journalctl -u openclaw -f"
      ;;
    railway)
      log "部署 URL: $RAILWAY_PUBLIC_DOMAIN"
      ;;
    vercel)
      log "部署 URL: https://$VERCEL_URL"
      ;;
  esac
  
  log "=========================================
}

# 主程序流程
main() {
  log "🎯 OpenClaw 部署開始"
  log "目標: $TARGET_PLATFORM / $ENVIRONMENT"
  
  detect_platform
  install_dependencies
  clone_repo
  generate_config
  init_database
  install_npm_deps
  platform_specific_config
  start_service
  health_check
  deployment_summary
  
  log "✅ 部署流程完成！"
}

# 錯誤捕捉
trap 'error "部署被中斷"' INT TERM

main "$@"
```

#### 4. 快速使用

```bash
# VPS (DigitalOcean/Linode)
curl -fsSL https://raw.githubusercontent.com/openclaw/openclaw/main/deploy.sh | bash -s vps production

# Railway (網頁部署)
git clone https://github.com/openclaw/openclaw
cd openclaw
railway up

# Vercel (無伺服器)
npm install -g vercel
cd openclaw
vercel deploy

# 本地開發
bash deploy.sh local development
```
