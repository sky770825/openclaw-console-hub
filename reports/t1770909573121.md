### JavaScript 渲染
```typescript
// 使用 Playwright（CoDEX 推薦）
async function scrapeWithJS(url: string, options: JSScraperOptions) {
  const browser = await chromium.launch({
    headless: options.headless ?? true,
    proxy: options.proxy
  });
  
  const page = await browser.newPage();
  
  // 設置 UA
  await page.setUserAgent(options.userAgent);
  
  // 設置視窗大小
  await page.setViewportSize({ width: 1920, height: 1080 });
  
  try {
    // 導航到頁面
    await page.goto(url, { waitUntil: 'networkidle' });
    
    // 等待特定元素
    if (options.waitFor?.selector) {
      await page.waitForSelector(options.waitFor.selector, {
        timeout: options.waitFor.timeout ?? 5000
      });
    }
    
    // 執行點擊序列
    for (const button of options.clickButtons || []) {
      await page.click(button);
      await page.waitForLoadState('networkidle');
    }
    
    // 滾動頁面（加載延遲內容）
    if (options.scrollPage) {
      await page.evaluate(() => {
        window.scrollBy(0, document.body.scrollHeight);
      });
    }
    
    // 獲取內容
    const content = await page.content();
    return content;
  } finally {
    await browser.close();
  }
}
```

## 資料清洗模塊

### HTML 清洗
```typescript
async function cleanHTML(html: string, rules: CleaningRules) {
  const { JSDOM } = require('jsdom');
  const dom = new JSDOM(html);
  const doc = dom.window.document;
  
  // 移除腳本
  if (rules.removeScripts) {
    doc.querySelectorAll('script').forEach(el => el.remove());
  }
  
  // 移除樣式
  if (rules.removeStyles) {
    doc.querySelectorAll('style, link[rel="stylesheet"]').forEach(el => el.remove());
  }
  
  // 移除特定標籤
  for (const tag of rules.removeTags || []) {
    doc.querySelectorAll(tag).forEach(el => el.remove());
  }
  
  // 正規化空白
  if (rules.normalizeWhitespace) {
    const text = doc.body.textContent;
    return text.replace(/\s+/g, ' ').trim();
  }
  
  return doc.body.innerHTML;
}
```

```typescript
async function extractStructuredData(html: string, schema: DataSchema) {
  const { JSDOM } = require('jsdom');
  const dom = new JSDOM(html);
  
  if (schema.type === 'array') {
    // 抽取多個物件
    const results = [];
    // 實現邏輯...
    return results;
  } else {
    // 抽取單個物件
    const obj = {};
    for (const [fieldName, fieldDef] of Object.entries(schema.fields)) {
      let elements = dom.window.document.querySelectorAll(fieldDef.selector);
      
      if (fieldDef.multiple) {
        obj[fieldName] = Array.from(elements).map(el => 
          fieldDef.transform ? fieldDef.transform(el.textContent) : el.textContent
        );
      } else {
        obj[fieldName] = fieldDef.transform 
          ? fieldDef.transform(elements[0]?.textContent)
          : elements[0]?.textContent;
      }
    }
    return obj;
  }
}
```

### 去重邏輯
```typescript
function deduplicateResults(results: ScraperResult[]) {
  const seen = new Set<string>();
  const deduped = [];
  
  for (const result of results) {
    // 使用內容雜湊去重
    const hash = crypto
      .createHash('sha256')
      .update(result.content)
      .digest('hex');
    
    if (!seen.has(hash)) {
      seen.add(hash);
      deduped.push(result);
    }
  }
  
  return deduped;
}
```

### 指數退避策略
```typescript
async function scrapeWithRetry(
  url: string,
  options: ScrapeOptions,
  maxRetries = 3
): Promise<ScraperResult> {
  for (let attempt = 0; attempt < maxRetries; attempt++) {
    try {
      const result = await scrape(url, options);
      
      // 檢測是否被阻擋
      if (result.statusCode === 429 || result.statusCode === 403) {
        throw new Error('IP blocked');
      }
      
      return result;
    } catch (error) {
      const delay = Math.pow(2, attempt) * 1000 + Math.random() * 1000;
      
      if (attempt < maxRetries - 1) {
        await sleep(delay);
        
        // 嘗試換代理
        if (options.proxy?.url) {
          options.proxy = getNextProxy();
        }
      } else {
        throw error;
      }
    }
  }
}
```
