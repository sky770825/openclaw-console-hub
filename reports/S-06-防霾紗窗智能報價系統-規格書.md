## 3. 數據模型

### 3.1 biz_window_screen規格

```javascript
// 標準規格
{
  "width": 1200,        // 寬度 (mm)
  "height": 1500,       // 高度 (mm)
  "material": "standard",  // 標準/高級/強固
  "color": "white",     // 顏色
  "installationType": "sliding",  // 滑軌/固定
  "customization": {
    "handle": true,     // 手柄
    "latch": true,      // 門栓
    "lockSet": false    // 鎖組
  }
}

// 計算面積
area = (width * height) / 1000000  // m²

// 定價邏輯
baseCost = materialCost[material] * area
installationCost = installationType === 'sliding' ? 800 : 500
customizationCost = sum(customization options)
laborCost = area * 150  // 每 m² 150 元工資

subtotal = baseCost + installationCost + customizationCost + laborCost
margin = subtotal * profitMargin  // 50% - 70%
finalPrice = subtotal + margin
```

### 4.2 LINEBot 實現

```javascript
// skills/line-quote-bot/index.js
const { CommunicationBroker } = require('@openclaw/communication-abstraction');
const { QuoteEngine } = require('../quote-engine');
const { PDFGenerator } = require('../pdf-generator');

class LineQuoteBot {
  constructor(config) {
    this.broker = new CommunicationBroker(config.communication);
    this.quoteEngine = new QuoteEngine(config.pricing);
    this.pdfGenerator = new PDFGenerator();
    this.crmDB = new GoogleSheetsAdapter(config.crm);
  }

  async handleMessage(userId, messageText) {
    const session = await this.getOrCreateSession(userId);

    switch (session.state) {
      case 'START':
        return this.askWindowType(userId);
      
      case 'WINDOW_TYPE_SELECTED':
        return this.askSpecification(userId, messageText);
      
      case 'SPECIFICATION_INPUT':
        return this.calculateQuote(userId, messageText, session);
      
      case 'QUOTE_CONFIRMED':
        return this.collectCustomerInfo(userId, messageText, session);
      
      case 'COMPLETED':
        return this.showFollowUpOptions(userId);
    }
  }

  async askWindowType(userId) {
    const buttons = [
      { label: '🪟 標準窗', text: 'standard_window' },
      { label: '🚪 推拉門', text: 'sliding_door' },
      { label: '🪟 其他', text: 'other' }
    ];

    await this.broker.sendButton(
      'line',
      userId,
      '防霾biz_window_screen報價助手',
      buttons
    );

    await this.updateSession(userId, { state: 'WINDOW_TYPE_SELECTED' });
  }

  async askSpecification(userId, windowType) {
    const message = `
好的！我來幫您計算 ${this.getWindowTypeName(windowType)} 的報價。

請提供以下規格（使用格式: 寬度,高度,材料,顏色）
例: 1200,1500,standard,white

📏 寬度 (mm)
📏 高度 (mm)
🎨 材料 (standard/premium/strong)
🎨 顏色 (white/brown/black)
    `;

    await this.broker.sendMessage('line', userId, message);
    await this.updateSession(userId, { 
      state: 'SPECIFICATION_INPUT',
      windowType 
    });
  }

  async calculateQuote(userId, specText, session) {
    try {
      const spec = this.parseSpecification(specText);
      const quote = await this.quoteEngine.calculate(spec);
      
      const summary = `
✅ 規格已收到！

摘要:
📐 寬 ${spec.width}mm × 高 ${spec.height}mm
🎨 材料: ${spec.material} | 顏色: ${spec.color}
⚙️ 面積: ${(spec.width * spec.height / 1000000).toFixed(2)} m²

💰 預計報價: NT$ ${quote.total.toLocaleString()}

請確認? [確認報價] [修改規格] [追加清單]
      `;

      await this.broker.sendMessage('line', userId, summary);
      
      // 儲存會話狀態
      await this.updateSession(userId, {
        state: 'QUOTE_CONFIRMED',
        quote,
        spec
      });

    } catch (error) {
      await this.broker.sendMessage('line', userId, `❌ 規格解析失敗: ${error.message}`);
    }
  }

  async collectCustomerInfo(userId, action, session) {
    if (action === '確認報價') {
      const message = `
太好了！最後需要您的聯繫方式:

請回覆格式: 姓名,電話,信箱
例: 王小明,0987654321,wang@example.com
      `;

      await this.broker.sendMessage('line', userId, message);
      await this.updateSession(userId, {
        state: 'CUSTOMER_INFO_COLLECTING'
      });
    }
  }

  async generateAndSendQuote(userId, session) {
    const { quote, spec } = session;
    const customerInfo = session.customerInfo;

    // 建立報價記錄
    const quotation = {
      quotationId: `QT-${Date.now()}`,
      customerId: userId,
      spec,
      quote,
      customerInfo,
      createdAt: new Date().toISOString(),
      expiresAt: new Date(Date.now() + 14 * 86400000).toISOString(),
      status: 'pending'
    };

    // 生成 PDF
    const pdfPath = await this.pdfGenerator.generate(quotation);
    
    // 上傳到 Google Drive
    const fileId = await this.uploadToGoogleDrive(pdfPath);
    quotation.pdfUrl = `https://drive.google.com/file/d/${fileId}/view`;

    // 存儲到 CRM Sheets
    await this.crmDB.appendValues('Sheet1!A1', [[
      quotation.quotationId,
      customerInfo.name,
      customerInfo.phone,
      quote.total,
      'pending',
      new Date().toISOString()
    ]]);

    // 發送報價單與相關訊息
    const message = `
✨ 報價單已生成！

${customerInfo.name} 您好！

報價單編號: ${quotation.quotationId}
報價金額: NT$ ${quote.total.toLocaleString()}
有效期: ${quotation.expiresAt.split('T')[0]}

📄 報價單: [點此查看]
💬 有問題? 直接詢問或聯絡業務
📞 業務電話: 0800-123-456

期待您的回復！🙏
    `;

    await this.broker.sendMessage('line', userId, message);
    
    // 業務員收到通知
    await this.notifyTeam(quotation);

    // 更新會話
    await this.updateSession(userId, {
      state: 'COMPLETED',
      quotationId: quotation.quotationId
    });
  }

  async notifyTeam(quotation) {
    const teamMessage = `
🔔 新的報價詢問

客戶: ${quotation.customerInfo.name}
電話: ${quotation.customerInfo.phone}
金額: NT$ ${quotation.quote.total.toLocaleString()}
報價單: ${quotation.quotationId}

請跟進...
    `;

    // 發送給業務員群組
    await this.broker.sendMessage('telegram', process.env.SALES_TEAM_CHAT_ID, teamMessage);
  }

  parseSpecification(text) {
    const [width, height, material, color] = text.split(',').map(s => s.trim());
    return {
      width: parseInt(width),
      height: parseInt(height),
      material: material || 'standard',
      color: color || 'white',
      installationType: 'sliding',
      customization: {}
    };
  }

  getWindowTypeName(type) {
    const names = {
      'standard_window': '標準窗',
      'sliding_door': '推拉門',
      'other': '其他'
    };
    return names[type] || type;
  }
}

module.exports = LineQuoteBot;
```

## 6. PDF 報價單生成

```javascript
// lib/pdf-generator.js
const PDFDocument = require('pdfkit');
const fs = require('fs');

class PDFGenerator {
  async generate(quotation) {
    const doc = new PDFDocument();
    const filename = `/tmp/${quotation.quotationId}.pdf`;
    const stream = fs.createWriteStream(filename);

    doc.pipe(stream);

    // 標題
    doc.fontSize(20).text('防霾biz_window_screen報價單', { align: 'center' });
    doc.fontSize(10).text(`報價編號: ${quotation.quotationId}`, { align: 'right' });

    // 客戶資訊
    doc.fontSize(12).text('客戶資訊', { underline: true });
    doc.fontSize(10);
    doc.text(`姓名: ${quotation.customerInfo.name}`);
    doc.text(`電話: ${quotation.customerInfo.phone}`);
    doc.text(`信箱: ${quotation.customerInfo.email}`);

    // 規格
    doc.fontSize(12).text('biz_window_screen規格', { underline: true });
    const spec = quotation.spec;
    doc.fontSize(10);
    doc.text(`尺寸: ${spec.width}mm × ${spec.height}mm`);
    doc.text(`材料: ${spec.material}`);
    doc.text(`顏色: ${spec.color}`);

    // 價格明細
    doc.fontSize(12).text('價格明細', { underline: true });
    doc.fontSize(10);
    doc.text(`小計: NT$ ${quotation.quote.subtotal.toLocaleString()}`);
    doc.text(`稅金: NT$ ${quotation.quote.tax.toLocaleString()}`);
    doc.text(`優惠: NT$ ${quotation.quote.discount.toLocaleString()}`);
    
    doc.fontSize(14).text(`總計: NT$ ${quotation.quote.total.toLocaleString()}`, {
      weight: 'bold'
    });

    // 有效期
    doc.fontSize(10);
    doc.text(`有效期至: ${quotation.expiresAt.split('T')[0]}`);
    doc.text(`報價生成時間: ${new Date().toLocaleString('zh-TW')}`);

    // 簽署線
    doc.moveTo(50, doc.y + 20).lineTo(550, doc.y + 20).stroke();
    doc.text('客戶簽章');

    doc.end();

    return new Promise((resolve) => {
      stream.on('finish', () => resolve(filename));
    });
  }
}

module.exports = PDFGenerator;
```
