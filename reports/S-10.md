### 1. ReportGenerator
```python
class ReportGenerator:
    def __init__(self):
        self.template_engine = Jinja2Templates()
        self.data_aggregator = DataAggregator()
    
    def generate(self, report_type, config):
        # 1. иҒҡеҗҲж•ёж“ҡ
        data = self.data_aggregator.fetch(config.sources)
        
        # 2. жҮүз”ЁжЁЎжқҝ
        template = self.template_engine.load(config.template)
        content = template.render(data=data, config=config)
        
        # 3. ијёеҮәж јејҸ
        if config.format == 'pdf':
            return self._to_pdf(content)
        elif config.format == 'html':
            return self._to_html(content)
        return content
```

```python
from apscheduler import Scheduler

scheduler = Scheduler()

@scheduler.schedule(cron="0 9 * * *")  # жҜҸж—Ҙ 9:00
def daily_report():
    report = generator.generate('daily', config)
    sender.send_email(report)
    sender.send_line(report.summary)

@scheduler.schedule(cron="0 9 * * 1")  # жҜҸйҖұдёҖ 9:00
def weekly_report():
    report = generator.generate('weekly', config)
    sender.send_email(report, to_all=True)
```
