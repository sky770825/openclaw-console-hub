### 3.2 核心類別設計

```python
class RealPriceQuerySkill:
    def __init__(self):
        self.api_client = MOIApiClient()
        self.cache = SQLiteCache()
        self.analyzer = PriceAnalyzer()
        self.reporter = WeeklyReporter()
    
    def query(self, city, district, road=None, 
              date_range=None, building_type=None):
        """
        查詢實價登錄資料
        
        Args:
            city: 縣市名稱
            district: 鄉鎮區名稱
            road: 路段名稱（可選）
            date_range: 日期範圍 tuple (start, end)
            building_type: 建物類型（住宅/商辦/店面）
        """
        # 1. 檢查快取
        cache_key = f"{city}-{district}-{road}-{date_range}"
        cached = self.cache.get(cache_key)
        if cached and not self._is_stale(cached):
            return cached
        
        # 2. 呼叫 API
        data = self.api_client.fetch(city, district, road, date_range)
        
        # 3. 儲存快取
        self.cache.set(cache_key, data)
        
        return data
    
    def generate_weekly_report(self, city, district):
        """生成週報"""
        # 取得近 7 天資料
        end_date = datetime.now()
        start_date = end_date - timedelta(days=7)
        
        data = self.query(city, district, date_range=(start_date, end_date))
        
        report = self.reporter.generate(data, {
            'city': city,
            'district': district,
            'period': f"{start_date.date()} ~ {end_date.date()}"
        })
        
        return report
    
    def track_competitor(self, project_name):
        """追蹤競品建案"""
        # 查詢該建案歷史成交紀錄
        history = self.api_client.search_by_project(project_name)
        
        # 分析價格趨勢
        trend = self.analyzer.analyze_trend(history)
        
        return {
            'project': project_name,
            'transactions': history,
            'trend': trend,
            'avg_price_per_ping': self._calc_avg(history)
        }
```

## 4. 資料模型

```sql
-- 實價登錄原始資料
CREATE TABLE real_price_transactions (
    id INTEGER PRIMARY KEY,
    case_id TEXT UNIQUE,
    city TEXT NOT NULL,
    district TEXT NOT NULL,
    road TEXT,
    address TEXT,
    transaction_date DATE,
    total_price INTEGER,
    price_per_ping REAL,
    building_area_ping REAL,
    land_area_ping REAL,
    building_type TEXT,
    floor TEXT,
    total_floors INTEGER,
    building_age REAL,
    room_count INTEGER,
    layout TEXT,
    parking_space BOOLEAN,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 週報記錄
CREATE TABLE weekly_reports (
    id INTEGER PRIMARY KEY,
    city TEXT NOT NULL,
    district TEXT NOT NULL,
    report_date DATE,
    report_content TEXT,
    transaction_count INTEGER,
    avg_price_per_ping REAL,
    price_change_percent REAL,
    generated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 競品追蹤清單
CREATE TABLE competitor_tracking (
    id INTEGER PRIMARY KEY,
    project_name TEXT NOT NULL,
    city TEXT,
    district TEXT,
    target_price_per_ping REAL,
    alert_threshold REAL, -- 漲跌幅百分比觸發通知
    is_active BOOLEAN DEFAULT 1,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
```

## 6. OpenClaw 技能腳本

```bash
#!/bin/bash
# skills/real-price-query/main.sh

COMMAND="$1"
shift

 case "$COMMAND" in
    "查詢")
        city="$1"
        district="$2"
        road="$3"
        python3 "$SKILL_ROOT/query.py" "$city" "$district" "$road"
        ;;
    "週報")
        city="$1"
        district="$2"
        python3 "$SKILL_ROOT/weekly_report.py" "$city" "$district"
        ;;
    "追蹤")
        project="$1"
        python3 "$SKILL_ROOT/track.py" "$project"
        ;;
    *)
        echo "用法:"
        echo "  實價查詢 [縣市] [鄉鎮] [路段]"
        echo "  實價週報 [縣市] [鄉鎮]"
        echo "  實價追蹤 [建案名稱]"
        ;;
esac
```
