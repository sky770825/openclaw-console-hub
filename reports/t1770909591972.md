## 方案概覽

使用 TypeDoc (TypeScript) + Swagger/OpenAPI (REST API) 實現自動化文件生成。

---

## 1. TypeScript API 文件 (TypeDoc)

### 安裝與配置

```bash
npm install -D typedoc @typedoc/plugin-markdown
```

### typedoc.json 設定

```json
{
  "entryPoints": ["src/index.ts"],
  "out": "docs/api",
  "plugin": ["@typedoc/plugin-markdown"],
  "markdown": {
    "outputFileStrategy": "modules",
    "readme": "none"
  },
  "includeVersion": true,
  "categorizeByGroup": true
}
```

### 執行

```bash
npx typedoc --out docs/api/typescript
```

## 2. REST API 文件 (Swagger/OpenAPI)

### 使用 Swagger JSDoc

```bash
npm install -D swagger-jsdoc swagger-ui-express
```

### 程式碼範例

```javascript
// src/api/skills.ts
/**
 * @openapi
 * /api/skills:
 *   get:
 *     tags:
 *       - Skills
 *     summary: 列出所有 Skill
 *     responses:
 *       200:
 *         description: Skill 列表
 *         content:
 *           application/json:
 *             schema:
 *               type: array
 *               items:
 *                 $ref: '#/components/schemas/Skill'
 */
router.get('/api/skills', async (req, res) => {
  // 實作
});
```

### swagger.js

```javascript
const swaggerJsdoc = require('swagger-jsdoc');

const options = {
  definition: {
    openapi: '3.0.0',
    info: {
      title: 'ClawFlow API',
      version: '1.0.0',
      description: '開源 AI Agent 自動化框架 API'
    },
    servers: [
      { url: 'http://localhost:3000', description: 'Development' },
      { url: 'https://api.clawflow.dev', description: 'Production' }
    ]
  },
  apis: ['./src/api/*.ts']
};

module.exports = swaggerJsdoc(options);
```

### Express 整合

```javascript
const swaggerUi = require('swagger-ui-express');
const specs = require('./swagger');

app.use('/api-docs', swaggerUi.serve);
app.get('/api-docs', swaggerUi.setup(specs));
```

## 3. 自動化生成流程

### 建立生成指令

```bash
# package.json
"scripts": {
  "docs:build": "npm run docs:typedoc && npm run docs:api",
  "docs:typedoc": "typedoc",
  "docs:api": "swagger-jsdoc -d swagger.js > docs/api/openapi.yaml",
  "docs:watch": "nodemon --exec npm run docs:build"
}
```

### GitHub Actions CI/CD

```yaml
# .github/workflows/docs.yml
name: Generate API Docs

on:
  push:
    branches: [main]
    paths:
      - 'src/**'
      - 'package.json'

jobs:
  build:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - uses: actions/setup-node@v3
        with:
          node-version: '18'
      
      - run: npm install
      - run: npm run docs:build
      
      - uses: actions/upload-artifact@v3
        with:
          name: api-docs
          path: docs/api/
```

## 4. 文件發佈

### 部署到 GitHub Pages

```yaml
# .github/workflows/deploy-docs.yml
name: Deploy Docs

on:
  push:
    branches: [main]

jobs:
  deploy:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - uses: actions/setup-node@v3
      - run: npm run docs:build
      
      - uses: peaceiris/actions-gh-pages@v3
        with:
          github_token: ${{ secrets.GITHUB_TOKEN }}
          publish_dir: ./docs
```

### 或使用 Vercel

```bash
# vercel.json
{
  "buildCommand": "npm run docs:build",
  "outputDirectory": "docs"
}
```

 * ```
 */
async function runSkill(skillName: string, input: Record<string, any>) {
  // 實作
}
```
