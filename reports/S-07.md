```sql
CREATE TABLE products (
    id INTEGER PRIMARY KEY,
    name TEXT NOT NULL,
    unit TEXT NOT NULL,
    category TEXT,
    min_stock REAL DEFAULT 10,
    optimal_stock REAL DEFAULT 50,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 庫存記錄表
CREATE TABLE inventory_logs (
    id INTEGER PRIMARY KEY,
    product_id INTEGER,
    quantity REAL NOT NULL,
    action_type TEXT, -- 'in'/'out'
    note TEXT,
    recorded_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (product_id) REFERENCES products(id)
);

-- 銷售記錄表
CREATE TABLE sales (
    id INTEGER PRIMARY KEY,
    product_id INTEGER,
    quantity REAL NOT NULL,
    revenue REAL,
    sold_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (product_id) REFERENCES products(id)
);

-- 預警記錄表
CREATE TABLE alerts (
    id INTEGER PRIMARY KEY,
    product_id INTEGER,
    alert_type TEXT, -- 'low_stock'/'expiring'
    status TEXT DEFAULT 'pending', -- 'pending'/'resolved'
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    resolved_at TIMESTAMP,
    FOREIGN KEY (product_id) REFERENCES products(id)
);
```

## 3. 補貨建議演算法

```python
def get_reorder_suggestions(self):
    """
    補貨建議邏輯：
    1. 計算過去 7 天平均日銷量
    2. 預測剩餘庫存可支撐天數
    3. 建議補貨量 = (optimal_stock - current_stock) + (日銷量 × 3)
    """
    suggestions = []
    products = self.get_all_products()
    
    for product in products:
        current = self.get_current_stock(product.id)
        daily_rate = self.calculate_consumption_rate(product.id, days=7)
        
        if daily_rate > 0:
            days_remaining = current / daily_rate
            if days_remaining < 3:  # 少於 3 天庫存
                suggest_qty = (product.optimal_stock - current) + (daily_rate * 3)
                suggestions.append({
                    'product': product.name,
                    'current_stock': current,
                    'days_remaining': round(days_remaining, 1),
                    'suggested_qty': round(suggest_qty, 0),
                    'urgency': 'high' if days_remaining < 1 else 'medium'
                })
    
    return sorted(suggestions, key=lambda x: x['days_remaining'])
```
