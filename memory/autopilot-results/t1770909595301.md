#### 3. Cron 自動化腳本

```bash
#!/bin/bash
# track-api-costs.sh

set -e

COST_DIR="$HOME/.openclaw/workspace/memory/costs"
TODAY=$(date +%Y-%m-%d)
LOG_FILE="$COST_DIR/$TODAY.jsonl"
SUMMARY_FILE="$COST_DIR/summary.csv"

mkdir -p "$COST_DIR"

# 函數: 取得 OpenAI 成本
get_openai_cost() {
  if [ -z "$OPENAI_API_KEY" ]; then
    echo "⚠️  OPENAI_API_KEY 未設定"
    return
  fi
  
  local cost=$(curl -s "https://api.openai.com/dashboard/billing/usage?date=$TODAY" \
    -H "Authorization: Bearer $OPENAI_API_KEY" 2>/dev/null | \
    jq '.total_usage' || echo "error")
  
  if [ "$cost" != "error" ]; then
    echo "{\"date\":\"$TODAY\",\"provider\":\"openai\",\"cost\":$cost,\"timestamp\":\"$(date -u +%Y-%m-%dT%H:%M:%SZ)\"}" >> "$LOG_FILE"
    echo "✅ OpenAI: \$$cost"
  fi
}

# 函數: 取得 OpenRouter 成本
get_openrouter_cost() {
  if [ -z "$OPENROUTER_API_KEY" ]; then
    echo "⚠️  OPENROUTER_API_KEY 未設定"
    return
  fi
  
  local data=$(curl -s "https://openrouter.ai/api/v1/auth/key" \
    -H "Authorization: Bearer $OPENROUTER_API_KEY" 2>/dev/null)
  
  local cost=$(echo "$data" | jq '.data.total_cost // 0')
  local requests=$(echo "$data" | jq '.data.requests // 0')
  
  echo "{\"date\":\"$TODAY\",\"provider\":\"openrouter\",\"cost\":$cost,\"requests\":$requests,\"timestamp\":\"$(date -u +%Y-%m-%dT%H:%M:%SZ)\"}" >> "$LOG_FILE"
  echo "✅ OpenRouter: \$$cost ($requests requests)"
}

# 函數: 計算本地 Ollama 成本（$0）
get_ollama_cost() {
  local ollama_calls=$(find ~/.openclaw/logs -name "*.json" -type f \
    -newermt "$TODAY" | xargs grep -l "ollama" 2>/dev/null | wc -l)
  
  echo "{\"date\":\"$TODAY\",\"provider\":\"local\",\"model\":\"ollama\",\"cost\":0,\"calls\":$ollama_calls,\"timestamp\":\"$(date -u +%Y-%m-%dT%H:%M:%SZ)\"}" >> "$LOG_FILE"
  echo "✅ Ollama (本地): $0 ($ollama_calls calls)"
}

# 主程序
echo "📊 開始記錄 $TODAY 的 API 成本..."

> "$LOG_FILE"  # 清空今日日誌

get_openai_cost
get_openrouter_cost
get_ollama_cost

# 生成摘要
echo "📈 生成每日摘要..."
local_cost=$(jq -s 'map(select(.provider=="openai") | .cost) | add // 0' "$LOG_FILE")
router_cost=$(jq -s 'map(select(.provider=="openrouter") | .cost) | add // 0' "$LOG_FILE")
total_cost=$(echo "$local_cost + $router_cost" | bc)

{
  echo "date,provider,cost_usd,requests,status"
  echo "$TODAY,openai,$local_cost,N/A,logged"
  echo "$TODAY,openrouter,$router_cost,N/A,logged"
  echo "$TODAY,total,$total_cost,--,--"
} >> "$SUMMARY_FILE"

echo ""
echo "✅ 成本追蹤完成"
echo "   OpenAI: \$$local_cost"
echo "   OpenRouter: \$$router_cost"
echo "   Total: \$$total_cost"
echo "   日誌: $LOG_FILE"
```
