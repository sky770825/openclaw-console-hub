## 三、API 設計

```typescript
interface DocGenerationConfig {
  sourceDir: string;         // 程式碼目錄
  outputDir: string;         // 輸出目錄
  formats: ('readme' | 'api' | 'changelog')[];
  template?: string;         // 自訂範本路徑
  apiStyle?: 'openapi' | 'jsdoc';
  includeExamples?: boolean;
}

interface GenerationResult {
  readme?: string;           // README 內容
  apiDocs?: string;          // API 文件
  changelog?: string;        // 變更日誌
  generatedAt: Date;
  filesCreated: string[];
}

// 主要介面
generateDocs(config: DocGenerationConfig): Promise<GenerationResult>
generateREADME(sourceDir: string, template?: string): Promise<string>
generateAPIDocs(sourceDir: string, style: 'openapi' | 'jsdoc'): Promise<string>
generateChangelog(sourceDir: string, versions?: number): Promise<string>
```
