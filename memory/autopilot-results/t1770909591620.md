## install.sh 標準模板

```bash
#!/bin/bash

# 配置
SKILL_NAME="${1:-my-skill}"
SKILL_DIR="./skills/$SKILL_NAME"
COLORS_RESET='\033[0m'
COLORS_GREEN='\033[0;32m'
COLORS_RED='\033[0;31m'
COLORS_YELLOW='\033[1;33m'

# 日誌函式
log_info() {
  echo -e "${COLORS_GREEN}✓${COLORS_RESET} $1"
}

log_error() {
  echo -e "${COLORS_RED}✗${COLORS_RESET} $1"
  exit 1
}

log_warn() {
  echo -e "${COLORS_YELLOW}!${COLORS_RESET} $1"
}

# 檢查 Node.js
check_nodejs() {
  if ! command -v node &> /dev/null; then
    log_error "Node.js 未安裝。請先安裝 Node.js 18+"
  fi
  
  NODE_VERSION=$(node -v | cut -d'v' -f2 | cut -d'.' -f1)
  if [ "$NODE_VERSION" -lt 18 ]; then
    log_error "需要 Node.js 18+，目前版本：$(node -v)"
  fi
  
  log_info "Node.js $(node -v) 已就緒"
}

# 檢查 npm
check_npm() {
  if ! command -v npm &> /dev/null; then
    log_error "npm 未安裝"
  fi
  log_info "npm $(npm -v) 已就緒"
}

# 檢查 ClawFlow
check_clawflow() {
  if ! command -v clawflow &> /dev/null; then
    log_error "ClawFlow CLI 未安裝。執行: npm install -g clawflow"
  fi
  log_info "ClawFlow 已安裝"
}

# 檢查 Skill 目錄
check_skill_dir() {
  if [ ! -d "$SKILL_DIR" ]; then
    log_error "Skill 目錄不存在: $SKILL_DIR"
  fi
  
  if [ ! -f "$SKILL_DIR/skill.yaml" ]; then
    log_error "缺少 skill.yaml: $SKILL_DIR/skill.yaml"
  fi
  
  if [ ! -f "$SKILL_DIR/package.json" ]; then
    log_error "缺少 package.json: $SKILL_DIR/package.json"
  fi
  
  log_info "Skill 結構驗證完成"
}

# 安裝依賴
install_dependencies() {
  log_info "安裝依賴中..."
  
  cd "$SKILL_DIR" || log_error "無法進入 Skill 目錄"
  
  if ! npm install; then
    log_error "依賴安裝失敗"
  fi
  
  cd - > /dev/null
  log_info "依賴安裝完成"
}

# 環境驗證
verify_environment() {
  log_info "驗證環境設定..."
  
  # 檢查 .env 檔案
  if [ ! -f "$SKILL_DIR/.env.example" ]; then
    log_warn "未找到 .env.example，跳過環境設定"
    return 0
  fi
  
  if [ ! -f "$SKILL_DIR/.env" ]; then
    log_warn "缺少 .env 檔案，從 .env.example 建立"
    cp "$SKILL_DIR/.env.example" "$SKILL_DIR/.env"
  fi
  
  log_info "環境驗證完成"
}

# 測試執行
test_execution() {
  log_info "執行測試..."
  
  cd "$SKILL_DIR" || log_error "無法進入 Skill 目錄"
  
  if npm run test 2>/dev/null; then
    log_info "測試通過"
  else
    log_warn "測試失敗或未定義測試腳本"
  fi
  
  cd - > /dev/null
}

# 註冊 Skill
register_skill() {
  log_info "註冊 Skill..."
  
  if ! clawflow skill register "$SKILL_NAME" "$SKILL_DIR"; then
    log_error "Skill 註冊失敗"
  fi
  
  log_info "Skill 已註冊"
}

# 主執行流程
main() {
  echo "🔧 ClawFlow Skill 安裝程式"
  echo "================================"
  
  check_nodejs
  check_npm
  check_clawflow
  check_skill_dir
  install_dependencies
  verify_environment
  test_execution
  register_skill
  
  echo ""
  echo "================================"
  log_info "安裝完成！"
  echo ""
  echo "下一步："
  echo "  1. 執行: clawflow run $SKILL_NAME"
  echo "  2. 查看文件: $SKILL_DIR/README.md"
  echo "  3. 配置: 編輯 $SKILL_DIR/.env"
}

# 錯誤處理
trap 'log_error "安裝程式中斷"' INT TERM

main "$@"
```

## package.json 中的安裝指令

```json
{
  "scripts": {
    "install:all": "npm install && npm run install:skill",
    "install:skill": "node scripts/install-skill.js",
    "preinstall": "node scripts/check-requirements.js",
    "postinstall": "npm run verify:env && npm run verify:structure"
  }
}
```

## 檢查腳本 (check-requirements.js)

```javascript
const fs = require('fs');
const { execSync } = require('child_process');

console.log('🔍 檢查安裝需求...\n');

// 檢查 Node.js
try {
  const nodeVersion = execSync('node -v').toString().trim();
  console.log(`✓ Node.js ${nodeVersion}`);
} catch {
  console.error('✗ Node.js 未安裝');
  process.exit(1);
}

// 檢查 npm
try {
  const npmVersion = execSync('npm -v').toString().trim();
  console.log(`✓ npm ${npmVersion}`);
} catch {
  console.error('✗ npm 未安裝');
  process.exit(1);
}

// 檢查 ClawFlow
try {
  execSync('clawflow --version', { stdio: 'ignore' });
  console.log('✓ ClawFlow CLI');
} catch {
  console.error('✗ ClawFlow 未安裝');
  console.error('  安裝: npm install -g clawflow');
  process.exit(1);
}

// 檢查必要檔案
const requiredFiles = ['skill.yaml', 'package.json'];
for (const file of requiredFiles) {
  if (!fs.existsSync(file)) {
    console.error(`✗ 缺少檔案: ${file}`);
    process.exit(1);
  }
  console.log(`✓ ${file}`);
}

console.log('\n✅ 所有檢查通過！開始安裝依賴...\n');
```

## 使用方式

```bash
# 方式 1: 執行安裝腳本
./scripts/install.sh my-skill

# 方式 2: 使用 npm
npm run install:all

# 方式 3: 使用 ClawFlow CLI
clawflow skill install my-skill
```
