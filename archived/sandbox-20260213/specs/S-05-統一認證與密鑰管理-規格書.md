// 初始化
const CredentialManager = require('@openclaw/credential-manager');
const manager = new CredentialManager({
  envPath: '.env.production',
  encryptionKey: process.env.ENCRYPTION_KEY,
  vaultUrl: 'https://vault.example.com'
});

// 獲取密鑰
const apiKey = manager.get('LINE_ACCESS_TOKEN');
const apiKey = manager.getSecure('GOOGLE_API_KEY');  // 解密存儲的密鑰

// 設定密鑰

// 驗證密鑰格式
manager.validate('GOOGLE_API_KEY', 'format:google_key');

// 列出所有密鑰（遮蔽敏感資訊）
const keys = manager.listKeys({ include_values: false });

// 檢查密鑰過期
const isExpired = manager.isExpired('OAUTH_TOKEN');

// 獲取密鑰中繼資料
const metadata = manager.getMetadata('OAUTH_TOKEN');
// { expiresAt: '2026-02-20', refreshToken: '...', scope: [...] }

### 3.2 Token 刷新 API

```javascript
const TokenManager = require('@openclaw/token-manager');
const tokenMgr = new TokenManager(credentialManager);

// 自動刷新過期 token
await tokenMgr.refreshIfExpired('GOOGLE_OAUTH_TOKEN');

// 手動刷新
const newToken = await tokenMgr.refresh('GOOGLE_OAUTH_TOKEN', {
  clientId: 'xxx',
  clientSecret: 'xxx',
  refreshToken: 'xxx'
});

// 批量刷新
const results = await tokenMgr.refreshBatch([
  'GOOGLE_OAUTH_TOKEN',
  'GITHUB_API_TOKEN'
]);

// 監聽刷新事件
tokenMgr.on('token:refreshed', (key, newToken) => {
  console.log(`Token ${key} refreshed`);
});

tokenMgr.on('token:refresh_failed', (key, error) => {
  console.error(`Failed to refresh ${key}:`, error);
  // 發送告警
});

// 排程自動刷新
tokenMgr.scheduleRefresh({
  interval: '1d',      // 每天
  beforeExpiry: '6h'   // 過期前 6 小時刷新
});
```

### 4.2 密鑰管理器實現

```javascript
// lib/credential-manager.js
const fs = require('fs');
const path = require('path');
const dotenv = require('dotenv');
const Encryption = require('./encryption');

class CredentialManager {
  constructor(config) {
    this.envPath = config.envPath || '.env';
    this.encryption = new Encryption({
      algorithm: config.encryptionAlgorithm || 'aes-256-gcm',
      key: config.encryptionKey
    });
    this.vaultUrl = config.vaultUrl;
    this.cache = new Map();
    this.metadata = new Map();
    
    this.loadEnv();
  }

  loadEnv() {
    if (fs.existsSync(this.envPath)) {
      const envConfig = dotenv.parse(fs.readFileSync(this.envPath));
      
      for (const [key, value] of Object.entries(envConfig)) {
        // 標記需要加密的敏感密鑰
        const isSecure = this.isSensitiveKey(key);
        
        if (isSecure && value.startsWith('enc:')) {
          // 已加密的密鑰，儲存加密形式
          this.cache.set(key, value);
        } else if (isSecure) {
          // 需要加密的密鑰
          const encrypted = this.encryption.encrypt(value);
          this.cache.set(key, `enc:${JSON.stringify(encrypted)}`);
        } else {
          // 公開密鑰
          this.cache.set(key, value);
        }

        // 記錄中繼資料
        this.metadata.set(key, {
          source: 'env_file',
          loadedAt: new Date(),
          isSecure,
          classification: this.classifyKey(key)
        });
      }
    }

    // 從環境變數覆蓋
    for (const [key, value] of Object.entries(process.env)) {
      if (key.startsWith('APP_')) {
        this.cache.set(key, value);
      }
    }
  }

  get(key, options = {}) {
    const value = this.cache.get(key);
    if (!value) {
      throw new Error(`Credential not found: ${key}`);
    }

    if (options.maskSensitive && this.isSensitiveKey(key)) {
      return this.maskValue(value);
    }

    return value;
  }

  getSecure(key) {
    const value = this.cache.get(key);
    if (!value) {
      throw new Error(`Credential not found: ${key}`);
    }

    if (value.startsWith('enc:')) {
      try {
        const encrypted = JSON.parse(value.slice(4));
        return this.encryption.decrypt(encrypted);
      } catch (error) {
        throw new Error(`Failed to decrypt ${key}: ${error.message}`);
      }
    }

    return value;
  }

  set(key, value, options = {}) {
    const encrypt = options.encrypt || this.isSensitiveKey(key);

    if (encrypt) {
      const encrypted = this.encryption.encrypt(value);
      this.cache.set(key, `enc:${JSON.stringify(encrypted)}`);
    } else {
      this.cache.set(key, value);
    }

    this.metadata.set(key, {
      source: 'runtime',
      setAt: new Date(),
      isSecure: encrypt,
      classification: this.classifyKey(key),
      ...options.metadata
    });
  }

  isSensitiveKey(key) {
    const sensitivePatterns = [
      'TOKEN', 'SECRET', 'PASSWORD', 'KEY',
      'API_KEY', 'OAUTH', 'CREDENTIAL'
    ];
    return sensitivePatterns.some(pattern => key.includes(pattern));
  }

  classifyKey(key) {
    if (key.includes('PUBLIC')) return 'PUBLIC';
    if (key.includes('INTERNAL')) return 'INTERNAL';
    if (key.includes('SECRET') || key.includes('PASSWORD')) return 'CONFIDENTIAL';
    if (key.includes('MASTER') || key.includes('ADMIN')) return 'RESTRICTED';
    return 'INTERNAL';
  }

  maskValue(value) {
    if (typeof value !== 'string') return '***';
    if (value.length <= 4) return '***';
    const visible = value.slice(-4);
    return `***${visible}`;
  }

  validate(key, schema) {
    const value = this.cache.get(key);
    if (!value) return false;

    const rules = this.parseSchema(schema);
    for (const rule of rules) {
      if (!this.validateRule(value, rule)) {
        throw new Error(`Validation failed for ${key}: ${rule}`);
      }
    }
    return true;
  }

  validateRule(value, rule) {
    if (rule.startsWith('format:')) {
      const format = rule.slice(7);
      return this.validateFormat(value, format);
    }
    if (rule.startsWith('regex:')) {
      const regex = new RegExp(rule.slice(6));
      return regex.test(value);
    }
    return true;
  }

  validateFormat(value, format) {
    const formats = {
      google_key: /^AIza[0-9A-Za-z-_]{35}$/,
      telegram_token: /^[0-9]{10}:[A-Za-z0-9_-]{35}$/,
      jwt: /^eyJ[A-Za-z0-9_-]*\.[A-Za-z0-9_-]*\.[A-Za-z0-9_-]*$/,
      uuid: /^[0-9a-f]{8}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{12}$/i
    };

    return formats[format] ? formats[format].test(value) : true;
  }

  listKeys(options = {}) {
    const keys = [];
    for (const [key, value] of this.cache.entries()) {
      const meta = this.metadata.get(key);
      keys.push({
        key,
        classification: meta?.classification,
        isSecure: meta?.isSecure,
        ...(options.include_values && { value: this.maskValue(value) }),
        ...(options.include_metadata && { metadata: meta })
      });
    }
    return keys;
  }

  isExpired(key) {
    const meta = this.metadata.get(key);
    if (!meta?.expiresAt) return false;
    return new Date() > new Date(meta.expiresAt);
  }

  getMetadata(key) {
    return this.metadata.get(key);
  }
}

module.exports = CredentialManager;
```

### 4.3 Token 刷新管理器

```javascript
// lib/token-manager.js
const EventEmitter = require('events');

class TokenManager extends EventEmitter {
  constructor(credentialManager) {
    super();
    this.credentialManager = credentialManager;
    this.refreshStrategies = new Map();
    this.activeRefreshes = new Map();
  }

  registerStrategy(tokenType, strategy) {
    this.refreshStrategies.set(tokenType, strategy);
  }

  async refreshIfExpired(key) {
    if (!this.credentialManager.isExpired(key)) {
      return this.credentialManager.getSecure(key);
    }

    return this.refresh(key);
  }

  async refresh(key, options = {}) {
    // 避免重複刷新
    if (this.activeRefreshes.has(key)) {
      return this.activeRefreshes.get(key);
    }

    const refreshPromise = this._performRefresh(key, options);
    this.activeRefreshes.set(key, refreshPromise);

    try {
      const newToken = await refreshPromise;
      this.credentialManager.set(key, newToken, {
        metadata: {
          refreshedAt: new Date(),
          expiresAt: new Date(Date.now() + 3600 * 1000) // 1 hour
        }
      });

      this.emit('token:refreshed', key, newToken);
      return newToken;
    } catch (error) {
      this.emit('token:refresh_failed', key, error);
      throw error;
    } finally {
      this.activeRefreshes.delete(key);
    }
  }

  async _performRefresh(key, options) {
    const strategy = this._getStrategy(key);
    if (!strategy) {
      throw new Error(`No refresh strategy for ${key}`);
    }

    return strategy(options);
  }

  _getStrategy(key) {
    // 根據密鑰類型選擇刷新策略
    if (key.includes('GOOGLE')) return this._googleOAuthRefresh;
    if (key.includes('GITHUB')) return this._githubTokenRefresh;
    if (key.includes('LINE')) return this._lineChannelRefresh;
    return null;
  }

  async _googleOAuthRefresh(options) {
    const { clientId, clientSecret, refreshToken } = options;
    const response = await fetch('https://oauth2.googleapis.com/token', {
      method: 'POST',
      body: JSON.stringify({
        client_id: clientId,
        client_secret: clientSecret,
        refresh_token: refreshToken,
        grant_type: 'refresh_token'
      })
    });
    const data = await response.json();
    return data.access_token;
  }

  scheduleRefresh(config) {
    const interval = this._parseInterval(config.interval);
    const beforeExpiry = this._parseInterval(config.beforeExpiry);

    setInterval(async () => {
      const keys = this.credentialManager.listKeys();
      for (const { key } of keys) {
        const meta = this.credentialManager.getMetadata(key);
        if (meta?.expiresAt) {
          const expiresAt = new Date(meta.expiresAt);
          const now = new Date();
          const timeUntilExpiry = expiresAt - now;

          if (timeUntilExpiry <= beforeExpiry) {
            try {
              await this.refresh(key);
            } catch (error) {
              console.error(`Failed to refresh ${key}:`, error);
            }
          }
        }
      }
    }, interval);
  }

  _parseInterval(str) {
    const match = str.match(/^(\d+)([smhd])$/);
    if (!match) throw new Error(`Invalid interval: ${str}`);
    
    const [, value, unit] = match;
    const multipliers = { s: 1000, m: 60000, h: 3600000, d: 86400000 };
    return parseInt(value) * (multipliers[unit] || 1000);
  }
}

module.exports = TokenManager;
```
